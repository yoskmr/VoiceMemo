### 1. オンボーディング状態管理（TCA Reducer） :large_blue_circle:

01-system-architecture.md セクション2.3準拠。初回起動フラグはUserDefaultsで管理。

```swift
// FeatureSettings/Onboarding/OnboardingReducer.swift

@Reducer
struct OnboardingReducer {
    @ObservableState
    struct State: Equatable {
        var currentStep: Step = .welcome
        var microphonePermissionGranted: Bool = false
        var speechPermissionGranted: Bool = false
        var isComplete: Bool = false

        enum Step: Int, CaseIterable {
            case welcome = 0
            case featureIntro = 1
            case permissions = 2
        }
    }

    enum Action {
        case nextStep
        case previousStep
        case requestMicrophonePermission
        case requestSpeechPermission
        case microphonePermissionResult(Bool)
        case speechPermissionResult(Bool)
        case skipPermissions
        case completeOnboarding
    }

    @Dependency(\.audioPermission) var audioPermission
    @Dependency(\.speechPermission) var speechPermission
    @Dependency(\.userDefaults) var userDefaults

    var body: some ReducerOf<Self> {
        Reduce { state, action in
            switch action {
            case .nextStep:
                let allSteps = State.Step.allCases
                if let currentIndex = allSteps.firstIndex(of: state.currentStep),
                   currentIndex + 1 < allSteps.count {
                    state.currentStep = allSteps[currentIndex + 1]
                }
                return .none

            case .previousStep:
                let allSteps = State.Step.allCases
                if let currentIndex = allSteps.firstIndex(of: state.currentStep),
                   currentIndex > 0 {
                    state.currentStep = allSteps[currentIndex - 1]
                }
                return .none

            case .requestMicrophonePermission:
                return .run { send in
                    let granted = await audioPermission.requestMicrophoneAccess()
                    await send(.microphonePermissionResult(granted))
                }

            case .requestSpeechPermission:
                return .run { send in
                    let granted = await speechPermission.requestSpeechRecognition()
                    await send(.speechPermissionResult(granted))
                }

            case .microphonePermissionResult(let granted):
                state.microphonePermissionGranted = granted
                return .none

            case .speechPermissionResult(let granted):
                state.speechPermissionGranted = granted
                return .none

            case .skipPermissions:
                return .send(.completeOnboarding)

            case .completeOnboarding:
                state.isComplete = true
                return .run { _ in
                    await userDefaults.setBool(true, forKey: "hasCompletedOnboarding")
                }
            }
        }
    }
}
```

### 5. オンボーディングコンテナView :large_blue_circle:

3ページをTabView + PageTabViewStyleで表示。ページインジケータ付き。

```swift
// FeatureSettings/Onboarding/OnboardingView.swift

struct OnboardingView: View {
    let store: StoreOf<OnboardingReducer>

    var body: some View {
        VStack(spacing: 0) {
            // ページコンテンツ
            TabView(selection: $store.currentStep.sending(\.stepChanged)) {
                WelcomePage()
                    .tag(OnboardingReducer.State.Step.welcome)

                FeatureIntroPage()
                    .tag(OnboardingReducer.State.Step.featureIntro)

                PermissionRequestPage(
                    microphoneGranted: store.microphonePermissionGranted,
                    speechGranted: store.speechPermissionGranted,
                    onRequestMicrophone: { store.send(.requestMicrophonePermission) },
                    onRequestSpeech: { store.send(.requestSpeechPermission) },
                    onSkip: { store.send(.skipPermissions) }
                )
                .tag(OnboardingReducer.State.Step.permissions)
            }
            .tabViewStyle(.page(indexDisplayMode: .never))
            .animation(.vmNavigationPush, value: store.currentStep)

            // カスタムページインジケータ
            HStack(spacing: VMSpacing.sm) {
                ForEach(OnboardingReducer.State.Step.allCases, id: \.self) { step in
                    Circle()
                        .fill(step == store.currentStep ? Color.vmPrimary : Color.vmDivider)
                        .frame(width: 8, height: 8)
                        .animation(.vmMicroInteraction, value: store.currentStep)
                }
            }
            .padding(.vertical, VMSpacing.lg)

            // ナビゲーションボタン
            HStack {
                if store.currentStep != .welcome {
                    Button(action: { store.send(.previousStep) }) {
                        Text("戻る")
                            .font(.vmHeadline)
                            .foregroundColor(.vmTextSecondary)
                    }
                }

                Spacer()

                if store.currentStep == .permissions {
                    Button(action: { store.send(.completeOnboarding) }) {
                        Text("始める")
                            .font(.vmHeadline)
                            .foregroundColor(.white)
                            .padding(.horizontal, VMSpacing.xl)
                            .padding(.vertical, VMSpacing.md)
                            .background(Color.vmCTAPrimary)
                            .cornerRadius(VMCornerRadius.medium)
                    }
                } else {
                    Button(action: { store.send(.nextStep) }) {
                        Text("次へ")
                            .font(.vmHeadline)
                            .foregroundColor(.vmPrimary)
                    }
                }
            }
            .padding(.horizontal, VMSpacing.xl)
            .padding(.bottom, VMSpacing.xxl)
        }
        .background(Color.vmBackground)
    }
}
```

## テスト要件

### 正常系
- 初回起動時（`hasCompletedOnboarding` = false）にオンボーディングが表示される
- 2回目起動時（`hasCompletedOnboarding` = true）にオンボーディングが表示されない
- `.nextStep` でwelcome -> featureIntro -> permissions と遷移する
- `.previousStep` でpermissions -> featureIntro -> welcome と遷移する
- `.requestMicrophonePermission` でシステムの権限ダイアログが表示される
- マイク権限が許可された場合、`microphonePermissionGranted` がtrueになる
- 音声認識権限が許可された場合、`speechPermissionGranted` がtrueになる
- `.completeOnboarding` で `isComplete` がtrueになり、UserDefaultsに保存される

### 異常系
- 権限が拒否された場合でもオンボーディングが進行できる
- `.skipPermissions` で権限未取得のまま完了できる
- welcome画面で `.previousStep` を送信しても画面が変わらない

### TCA TestStore
- `.nextStep` で `currentStep` が正しく遷移する
- `.requestMicrophonePermission` -> `.microphonePermissionResult(true)` の一連のフローが正しく動作する
- `.completeOnboarding` でUserDefaultsへの書き込みが実行される
