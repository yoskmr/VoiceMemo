```swift
// FeatureSettings/Settings/SettingsReducer.swift

@Reducer
struct SettingsReducer {
    @ObservableState
    struct State: Equatable {
        // アカウント
        var currentPlan: SubscriptionPlan = .free
        var aiProcessingRemaining: Int = 5
        var aiProcessingLimit: Int = 5

        // 音声認識
        var preferredSTTEngine: STTEngineType = .whisperKit
        var recognitionLanguage: String = "ja-JP"

        // WhisperKitモデル管理
        var whisperKitModels: [WhisperKitModelInfo] = []
        var modelDownloadProgress: Double? = nil

        // プライバシー
        var biometricAuthEnabled: Bool = false
        var iCloudSyncEnabled: Bool = false

        // 表示
        var theme: ThemeType = .system
        var isPro: Bool = false

        // データ
        var audioStorageUsed: String = "0 MB"
        var databaseStorageUsed: String = "0 MB"
        var totalStorageUsed: String = "0 MB"

        // アプリ情報
        var appVersion: String = ""
        var buildNumber: String = ""

        // ナビゲーション
        @Presents var destination: Destination.State?
    }

    @Reducer
    enum Destination {
        case subscription(SubscriptionReducer)
        case themeSettings(ThemeSettingsReducer)
        case whisperKitModelManagement(WhisperKitModelReducer)
        case customDictionary(CustomDictionaryReducer)
        case privacyPolicy
    }

    enum Action {
        case onAppear
        case loadSettings
        case settingsLoaded(UserSettings)

        // STTエンジン
        case sttEngineChanged(STTEngineType)

        // テーマ
        case themeSettingsTapped

        // WhisperKit
        case whisperKitModelManagementTapped
        case modelDownloadRequested(String)
        case modelDeleteRequested(String)
        case modelDownloadProgress(Double)
        case modelDownloadCompleted

        // プライバシー
        case biometricAuthToggled(Bool)
        case iCloudSyncToggled(Bool)

        // ナビゲーション
        case planTapped
        case customDictionaryTapped
        case privacyPolicyTapped
        case destination(PresentationAction<Destination.Action>)

        // データ
        case storageUsageLoaded(audio: String, database: String, total: String)
    }

    @Dependency(\.settingsRepository) var settingsRepository
    @Dependency(\.subscriptionClient) var subscriptionClient
    @Dependency(\.storageManager) var storageManager

    var body: some ReducerOf<Self> {
        Reduce { state, action in
            switch action {
            case .onAppear:
                state.appVersion = Bundle.main.infoDictionary?["CFBundleShortVersionString"] as? String ?? ""
                state.buildNumber = Bundle.main.infoDictionary?["CFBundleVersion"] as? String ?? ""
                state.isPro = subscriptionClient.currentPlan() == .pro
                return .merge(
                    .send(.loadSettings),
                    .run { send in
                        let usage = await storageManager.calculateStorageUsage()
                        await send(.storageUsageLoaded(
                            audio: usage.audioFormatted,
                            database: usage.databaseFormatted,
                            total: usage.totalFormatted
                        ))
                    }
                )

            case .sttEngineChanged(let engine):
                state.preferredSTTEngine = engine
                return .run { _ in
                    await settingsRepository.updateSTTEngine(engine)
                }

            case .biometricAuthToggled(let enabled):
                state.biometricAuthEnabled = enabled
                return .run { _ in
                    await settingsRepository.updateBiometricAuth(enabled)
                }

            case .themeSettingsTapped:
                guard state.isPro else { return .none }
                state.destination = .themeSettings(ThemeSettingsReducer.State(theme: state.theme))
                return .none

            case .planTapped:
                state.destination = .subscription(SubscriptionReducer.State())
                return .none

            case .privacyPolicyTapped:
                state.destination = .privacyPolicy
                return .none

            default:
                return .none
            }
        }
        .ifLet(\.$destination, action: \.destination)
    }
}
```

```swift
// FeatureSettings/WhisperKit/WhisperKitModelReducer.swift

struct WhisperKitModelInfo: Equatable, Identifiable {
    let id: String
    let name: String
    let sizeFormatted: String
    let isDownloaded: Bool
    let isActive: Bool
}

@Reducer
struct WhisperKitModelReducer {
    @ObservableState
    struct State: Equatable {
        var models: [WhisperKitModelInfo] = []
        var downloadingModelID: String? = nil
        var downloadProgress: Double = 0
        var isLoading: Bool = false
    }

    enum Action {
        case onAppear
        case modelsLoaded([WhisperKitModelInfo])
        case downloadModel(String)
        case downloadProgress(Double)
        case downloadCompleted(String)
        case deleteModel(String)
        case modelDeleted(String)
    }
}

// FeatureSettings/WhisperKit/WhisperKitModelView.swift

struct WhisperKitModelView: View {
    let store: StoreOf<WhisperKitModelReducer>

    var body: some View {
        List {
            ForEach(store.models) { model in
                HStack {
                    VStack(alignment: .leading, spacing: VMSpacing.xs) {
                        Text(model.name)
                            .font(.vmHeadline)
                            .foregroundColor(.vmTextPrimary)
                        Text(model.sizeFormatted)
                            .font(.vmCaption1)
                            .foregroundColor(.vmTextSecondary)
                    }

                    Spacer()

                    if model.isActive {
                        Text("使用中")
                            .font(.vmCaption1)
                            .foregroundColor(.vmSuccess)
                    } else if model.isDownloaded {
                        Button("削除") {
                            store.send(.deleteModel(model.id))
                        }
                        .foregroundColor(.vmError)
                        .font(.vmCaption1)
                    } else if store.downloadingModelID == model.id {
                        ProgressView(value: store.downloadProgress)
                            .frame(width: 60)
                    } else {
                        Button("ダウンロード") {
                            store.send(.downloadModel(model.id))
                        }
                        .foregroundColor(.vmPrimary)
                        .font(.vmCaption1)
                    }
                }
                .padding(.vertical, VMSpacing.xs)
            }
        }
        .navigationTitle("WhisperKitモデル")
        .onAppear { store.send(.onAppear) }
    }
}
```

### 4. テーマ設定サブ画面（Pro限定） :large_blue_circle:

```swift
// FeatureSettings/Theme/ThemeSettingsReducer.swift

@Reducer
struct ThemeSettingsReducer {
    @ObservableState
    struct State: Equatable {
        var theme: ThemeType = .system
    }

    enum Action {
        case themeSelected(ThemeType)
    }

    @Dependency(\.settingsRepository) var settingsRepository

    var body: some ReducerOf<Self> {
        Reduce { state, action in
            switch action {
            case .themeSelected(let theme):
                state.theme = theme
                return .run { _ in
                    await settingsRepository.updateTheme(theme)
                }
            }
        }
    }
}

// FeatureSettings/Theme/ThemeSettingsView.swift

struct ThemeSettingsView: View {
    let store: StoreOf<ThemeSettingsReducer>

    var body: some View {
        List {
            ForEach(ThemeType.allCases, id: \.self) { theme in
                Button(action: { store.send(.themeSelected(theme)) }) {
                    HStack {
                        Image(systemName: theme.iconName)
                            .foregroundColor(.vmPrimary)
                            .frame(width: 30)
                        Text(theme.displayName)
                            .font(.vmBody())
                            .foregroundColor(.vmTextPrimary)
                        Spacer()
                        if store.theme == theme {
                            Image(systemName: "checkmark")
                                .foregroundColor(.vmPrimary)
                        }
                    }
                }
            }
        }
        .navigationTitle("テーマ")
    }
}
```

## テスト要件

### 正常系
- 設定画面の全セクション（アカウント/音声認識/プライバシー/表示/データ/その他）が表示される
- STTエンジン変更がUserSettingsに保存される
- テーマ変更がUserSettingsに保存される（Pro限定）
- WhisperKitモデルのダウンロード/削除が動作する
- プラン表示が現在のサブスクリプション状態を反映する
- データ使用量が正しく計算・フォーマットされる
- アプリバージョンがBundle情報から取得される
- プライバシーポリシーリンクがSafariViewControllerで開く

### 異常系
- FreeプランユーザーがテーマタップしてもPro機能であることが表示される
- WhisperKitモデルダウンロード中にネットワークエラーが発生した場合の処理
- ストレージ計算が失敗した場合に「計算中...」が表示される

### TCA TestStore
- `.onAppear` でアプリバージョンが設定され、ストレージ使用量がロードされる
- `.sttEngineChanged(.whisperKit)` で `preferredSTTEngine` が更新される
- `.biometricAuthToggled(true)` で `biometricAuthEnabled` がtrueになる
- `.planTapped` で `destination` が `.subscription` に設定される
- `.themeSettingsTapped` がProユーザーの場合のみ `destination` を設定する
