### 4. D1マイグレーション :large_blue_circle:

03-backend-proxy.mdセクション4.4、セクション5準拠。

```sql
-- migrations/0001_initial.sql

-- ユーザーテーブル
CREATE TABLE IF NOT EXISTS users (
  id TEXT PRIMARY KEY,                           -- usr_xxxxxxxxxxxx
  apple_user_id TEXT UNIQUE,                     -- Apple Sign In後に設定
  email TEXT,                                    -- Private Relay email
  auth_method TEXT NOT NULL DEFAULT 'device',    -- 'device' | 'apple'
  plan TEXT NOT NULL DEFAULT 'free',             -- 'free' | 'pro'
  created_at TEXT NOT NULL,
  updated_at TEXT NOT NULL
);

-- サブスクリプションテーブル
CREATE TABLE IF NOT EXISTS subscriptions (
  id TEXT PRIMARY KEY,
  user_id TEXT NOT NULL REFERENCES users(id),
  product_id TEXT NOT NULL,                      -- io.murmurnote.pro.monthly / yearly
  app_account_token TEXT NOT NULL,               -- UUID(SHA256(user_id))
  original_transaction_id TEXT,
  status TEXT NOT NULL DEFAULT 'active',         -- active | expired | billing_retry | refunded
  expires_at TEXT,
  auto_renew INTEGER NOT NULL DEFAULT 1,
  created_at TEXT NOT NULL,
  updated_at TEXT NOT NULL
);

-- デバイステーブル
CREATE TABLE IF NOT EXISTS user_devices (
  id TEXT PRIMARY KEY,
  user_id TEXT NOT NULL REFERENCES users(id),
  device_id TEXT NOT NULL,                       -- UIDevice.identifierForVendor
  device_model TEXT,
  os_version TEXT,
  app_version TEXT,
  attest_verified INTEGER NOT NULL DEFAULT 0,
  attest_key_id TEXT,                            -- App Attest keyId
  attest_public_key TEXT,                        -- 保存済み公開鍵
  attest_receipt TEXT,                           -- attestation receipt
  attest_counter INTEGER NOT NULL DEFAULT 0,     -- assertion counter
  created_at TEXT NOT NULL,
  updated_at TEXT NOT NULL,
  UNIQUE(user_id, device_id)
);

-- Webhookイベントテーブル（冪等性保証用）
CREATE TABLE IF NOT EXISTS webhook_events (
  id TEXT PRIMARY KEY,
  notification_uuid TEXT UNIQUE NOT NULL,        -- Apple通知UUID
  notification_type TEXT NOT NULL,
  subtype TEXT,
  original_transaction_id TEXT,
  signed_date INTEGER NOT NULL,                  -- ミリ秒タイムスタンプ
  status TEXT NOT NULL DEFAULT 'processed',      -- processed | skipped
  processed_at TEXT NOT NULL,
  created_at TEXT NOT NULL
);

-- 使用量カウンタテーブル（D1トランザクション対応）
CREATE TABLE IF NOT EXISTS usage_counters (
  id TEXT PRIMARY KEY,
  user_id TEXT NOT NULL REFERENCES users(id),
  year_month TEXT NOT NULL,                      -- 'YYYY-MM' (JST基準)
  count INTEGER NOT NULL DEFAULT 0,
  last_used_at TEXT,
  created_at TEXT NOT NULL,
  updated_at TEXT NOT NULL,
  UNIQUE(user_id, year_month)
);

-- インデックス
CREATE INDEX IF NOT EXISTS idx_users_apple_user_id ON users(apple_user_id);
CREATE INDEX IF NOT EXISTS idx_subscriptions_user_id ON subscriptions(user_id);
CREATE INDEX IF NOT EXISTS idx_subscriptions_app_account_token ON subscriptions(app_account_token);
CREATE INDEX IF NOT EXISTS idx_user_devices_user_id ON user_devices(user_id);
CREATE INDEX IF NOT EXISTS idx_user_devices_device_id ON user_devices(device_id);
CREATE INDEX IF NOT EXISTS idx_webhook_events_notification_uuid ON webhook_events(notification_uuid);
CREATE INDEX IF NOT EXISTS idx_webhook_events_original_transaction_id ON webhook_events(original_transaction_id);
CREATE INDEX IF NOT EXISTS idx_usage_counters_user_id_year_month ON usage_counters(user_id, year_month);
```

### 5. Honoアプリ初期構成 :large_blue_circle:

03-backend-proxy.mdセクション3, 4.5準拠。

```typescript
// src/index.ts
import { Hono } from 'hono';
import { cors } from 'hono/cors';
import { logger } from 'hono/logger';
import type { Env } from './types/env';

const app = new Hono<{ Bindings: Env }>();

// グローバルミドルウェア
app.use('*', logger());
app.use('*', cors({
  origin: '*',
  allowMethods: ['GET', 'POST', 'PUT', 'DELETE'],
  allowHeaders: ['Authorization', 'Content-Type', 'X-App-Version', 'X-OS-Version',
                  'X-Device-ID', 'X-Request-ID', 'X-App-Attest-Assertion',
                  'Accept-Language', 'X-Timestamp', 'X-Nonce'],
}));

// ヘルスチェック
app.get('/health', (c) => c.json({ status: 'ok', version: c.env.API_VERSION }));

// APIルーティング（後続タスクで実装）
// app.route('/api/v1/ai', aiRoutes);
// app.route('/api/v1/auth', authRoutes);
// app.route('/api/v1/subscription', subscriptionRoutes);
// app.route('/api/v1', usageRoutes);

// 共通エラーレスポンス
app.notFound((c) => c.json({
  error: { code: 'NOT_FOUND', message: 'エンドポイントが見つかりません' }
}, 404));

app.onError((err, c) => {
  console.error(`Internal error: ${err.message}`);
  return c.json({
    error: { code: 'INTERNAL_ERROR', message: 'サーバー内部エラーが発生しました' }
  }, 500);
});

export default app;
```
