### 1. AppReducer（ルートReducer + タブ管理） :large_blue_circle:

01-system-architecture.md セクション2.2, 2.4準拠。TCA StackStateを活用。

```swift
// VoiceMemoApp/AppReducer.swift

@Reducer
struct AppReducer {
    @ObservableState
    struct State: Equatable {
        // タブ管理
        var selectedTab: Tab = .home

        // 各タブのルート状態
        var home = HomeReducer.State()
        var memoList = MemoListReducer.State()
        var settings = SettingsReducer.State()

        // NavigationStack
        var homePath = StackState<HomePath.State>()
        var memoPath = StackState<MemoPath.State>()
        var settingsPath = StackState<SettingsPath.State>()

        // オンボーディング
        var onboarding: OnboardingReducer.State?
        var hasCompletedOnboarding: Bool = false

        // タブ定義（3タブ構成: 04-ui-design-system.md セクション5.1）
        enum Tab: String, Hashable, CaseIterable {
            case home      // 録音タブ
            case memos     // メモ一覧タブ
            case settings  // 設定タブ

            var icon: String {
                switch self {
                case .home: return "mic.fill"
                case .memos: return "doc.text.fill"
                case .settings: return "gearshape.fill"
                }
            }

            var label: String {
                switch self {
                case .home: return "録音"
                case .memos: return "メモ"
                case .settings: return "設定"
                }
            }
        }
    }

    // MARK: - ナビゲーションパス定義

    @Reducer
    enum HomePath {
        case memoDetail(MemoDetailReducer)
    }

    @Reducer
    enum MemoPath {
        case memoDetail(MemoDetailReducer)
        case search(SearchReducer)
        case emotionTrend(EmotionVisualizationReducer)
    }

    @Reducer
    enum SettingsPath {
        case subscription(SubscriptionReducer)
        case themeSettings(ThemeSettingsReducer)
        case whisperKitModel(WhisperKitModelReducer)
        case customDictionary(CustomDictionaryReducer)
    }

    enum Action {
        // タブ
        case tabSelected(State.Tab)

        // 各タブのアクション
        case home(HomeReducer.Action)
        case memoList(MemoListReducer.Action)
        case settings(SettingsReducer.Action)

        // ナビゲーション
        case homePath(StackActionOf<HomePath>)
        case memoPath(StackActionOf<MemoPath>)
        case settingsPath(StackActionOf<SettingsPath>)

        // Deep Link
        case deepLinkReceived(URL)
        case navigateToMemoDetail(UUID)

        // オンボーディング
        case onboarding(OnboardingReducer.Action)
        case checkOnboardingStatus
    }

    @Dependency(\.userDefaults) var userDefaults

    var body: some ReducerOf<Self> {
        // 各Feature Reducerのスコープ
        Scope(state: \.home, action: \.home) {
            HomeReducer()
        }
        Scope(state: \.memoList, action: \.memoList) {
            MemoListReducer()
        }
        Scope(state: \.settings, action: \.settings) {
            SettingsReducer()
        }

        Reduce { state, action in
            switch action {
            case .tabSelected(let tab):
                state.selectedTab = tab
                return .none

            case .deepLinkReceived(let url):
                return handleDeepLink(url: url, state: &state)

            case .navigateToMemoDetail(let memoID):
                state.selectedTab = .memos
                state.memoPath.append(.memoDetail(MemoDetailReducer.State(memoID: memoID)))
                return .none

            case .checkOnboardingStatus:
                let completed = userDefaults.bool(forKey: "hasCompletedOnboarding")
                state.hasCompletedOnboarding = completed
                if !completed {
                    state.onboarding = OnboardingReducer.State()
                }
                return .none

            case .onboarding(.completeOnboarding):
                state.onboarding = nil
                state.hasCompletedOnboarding = true
                return .none

            default:
                return .none
            }
        }
        .forEach(\.homePath, action: \.homePath)
        .forEach(\.memoPath, action: \.memoPath)
        .forEach(\.settingsPath, action: \.settingsPath)
        .ifLet(\.onboarding, action: \.onboarding) {
            OnboardingReducer()
        }
    }

    // MARK: - Deep Link処理
    private func handleDeepLink(url: URL, state: inout State) -> Effect<Action> {
        guard let components = URLComponents(url: url, resolvingAgainstBaseURL: false) else {
            return .none
        }

        switch components.host {
        case "record":
            state.selectedTab = .home
            return .send(.home(.recordButtonTapped))

        case "memo":
            let pathSegment = String(components.path.dropFirst())
            if let memoID = UUID(uuidString: pathSegment) {
                return .send(.navigateToMemoDetail(memoID))
            }
            return .none

        default:
            return .none
        }
    }
}
```

### 2. AppView（ルートView + TabView） :large_blue_circle:

04-ui-design-system.md セクション5.1準拠の3タブ構成。

```swift
// VoiceMemoApp/AppView.swift

struct AppView: View {
    let store: StoreOf<AppReducer>

    var body: some View {
        TabView(selection: $store.selectedTab.sending(\.tabSelected)) {
            // Tab 1: 録音
            NavigationStack(path: $store.scope(state: \.homePath, action: \.homePath)) {
                HomeView(store: store.scope(state: \.home, action: \.home))
            } destination: { store in
                switch store.case {
                case .memoDetail(let detailStore):
                    MemoDetailView(store: detailStore)
                }
            }
            .tabItem {
                Label(
                    AppReducer.State.Tab.home.label,
                    systemImage: AppReducer.State.Tab.home.icon
                )
            }
            .tag(AppReducer.State.Tab.home)

            // Tab 2: メモ一覧
            NavigationStack(path: $store.scope(state: \.memoPath, action: \.memoPath)) {
                MemoListView(store: store.scope(state: \.memoList, action: \.memoList))
            } destination: { store in
                switch store.case {
                case .memoDetail(let detailStore):
                    MemoDetailView(store: detailStore)
                case .search(let searchStore):
                    SearchView(store: searchStore)
                case .emotionTrend(let trendStore):
                    EmotionVisualizationView(store: trendStore)
                }
            }
            .tabItem {
                Label(
                    AppReducer.State.Tab.memos.label,
                    systemImage: AppReducer.State.Tab.memos.icon
                )
            }
            .tag(AppReducer.State.Tab.memos)

            // Tab 3: 設定
            NavigationStack(path: $store.scope(state: \.settingsPath, action: \.settingsPath)) {
                SettingsView(store: store.scope(state: \.settings, action: \.settings))
            } destination: { store in
                switch store.case {
                case .subscription(let subStore):
                    SubscriptionView(store: subStore)
                case .themeSettings(let themeStore):
                    ThemeSettingsView(store: themeStore)
                case .whisperKitModel(let modelStore):
                    WhisperKitModelView(store: modelStore)
                case .customDictionary(let dictStore):
                    CustomDictionaryView(store: dictStore)
                }
            }
            .tabItem {
                Label(
                    AppReducer.State.Tab.settings.label,
                    systemImage: AppReducer.State.Tab.settings.icon
                )
            }
            .tag(AppReducer.State.Tab.settings)
        }
        .tint(.vmPrimary)
        .onOpenURL { url in
            store.send(.deepLinkReceived(url))
        }
        .onAppear {
            store.send(.checkOnboardingStatus)
        }
        .sheet(item: $store.scope(state: \.onboarding, action: \.onboarding)) { onboardingStore in
            OnboardingView(store: onboardingStore)
                .interactiveDismissDisabled()
        }
    }
}
```

### 3. 画面遷移アニメーション :large_blue_circle:

04-ui-design-system.md セクション7.2準拠。全画面遷移にスプリングカーブ適用。

```swift
// SharedUI/Navigation/VMNavigationTransitions.swift

import SwiftUI

// MARK: - カスタムタブ遷移
struct TabSwitchModifier: ViewModifier {
    @Environment(\.accessibilityReduceMotion) var reduceMotion
    let selectedTab: AppReducer.State.Tab

    func body(content: Content) -> some View {
        content
            .animation(
                reduceMotion ? .none : .vmTabSwitch,
                value: selectedTab
            )
    }
}

// MARK: - ナビゲーション遷移設定
extension View {
    /// Push遷移用スプリングカーブアニメーション（350ms）
    func vmNavigationTransition() -> some View {
        self.navigationTransition(.slide)
    }

    /// モーダル表示用スプリングカーブアニメーション（400ms）
    func vmModalTransition() -> some View {
        self.transition(
            .asymmetric(
                insertion: .move(edge: .bottom).combined(with: .opacity),
                removal: .move(edge: .bottom).combined(with: .opacity)
            )
        )
    }
}

// MARK: - 遷移パラメータ一覧（04-ui-design-system.md セクション7.2）
// | 遷移タイプ      | Duration | Spring                              | 用途              |
// |:---------------|:---------|:------------------------------------|:-----------------|
// | Push / Pop      | 350ms    | response: 0.35, dampingFraction: 0.86| 標準ナビゲーション |
// | Modal Present   | 400ms    | response: 0.4, dampingFraction: 0.82 | シート表示         |
// | Tab Switch      | 200ms    | response: 0.2, dampingFraction: 0.9  | タブ切替           |
// | Card Expand     | 350ms    | response: 0.35, dampingFraction: 0.8 | メモカード→詳細    |
```

- `.deepLinkReceived(URL(string: "voicememo://record")!)` で `selectedTab` が `.home` に変更され、録音開始アクションが送信される
