### 3.1 TCA State / Action / Reducer

```swift
// FeatureMemo/MemoDetail/MemoDetailReducer.swift
import ComposableArchitecture
import Foundation

@Reducer
struct MemoDetailReducer {
    @ObservableState
    struct State: Equatable {
        let memoID: UUID
        var title: String = ""
        var createdAt: Date = Date()
        var updatedAt: Date = Date()
        var durationSeconds: Double = 0
        var audioFilePath: String = ""

        // 文字起こし
        var transcriptionText: String = ""
        var transcriptionLanguage: String = "ja-JP"
        var transcriptionConfidence: Double = 0

        // AI要約（Phase 3で実装、枠のみ）
        var aiSummary: AISummaryState?
        var isAISummaryAvailable: Bool = false

        // 感情分析
        var emotion: EmotionState?

        // タグ
        var tags: [TagItem] = []

        // 音声再生（TASK-0014で実装、プレースホルダ）
        var audioPlayerState: AudioPlayerPlaceholderState = .init()

        // UI状態
        var isLoading: Bool = false
        var errorMessage: String?

        struct AISummaryState: Equatable {
            var summaryText: String
            var keyPoints: [String]
            var providerType: String
            var isOnDevice: Bool
            var generatedAt: Date
        }

        struct EmotionState: Equatable {
            var category: EmotionCategory
            var confidence: Double
            var description: String    // 感情の説明テキスト
        }

        struct TagItem: Equatable, Identifiable {
            let id: UUID
            var name: String
            var source: String  // "ai" or "manual"
        }

        struct AudioPlayerPlaceholderState: Equatable {
            var isAvailable: Bool = false
        }
    }

    enum Action {
        case onAppear
        case memoLoaded(Result<MemoDetail, Error>)
        case editButtonTapped
        case deleteButtonTapped
        case tagTapped(String)
        case shareButtonTapped
        case backButtonTapped

        // Phase 3 で実装
        case regenerateAISummary

        // TASK-0014 で実装
        case audioPlayer(AudioPlayerPlaceholderAction)
    }

    enum AudioPlayerPlaceholderAction {
        case playTapped
        case pauseTapped
    }

    struct MemoDetail: Equatable {
        let id: UUID
        let title: String
        let createdAt: Date
        let updatedAt: Date
        let durationSeconds: Double
        let audioFilePath: String
        let transcriptionText: String
        let transcriptionLanguage: String
        let transcriptionConfidence: Double
        let aiSummary: State.AISummaryState?
        let emotion: State.EmotionState?
        let tags: [State.TagItem]
    }

    @Dependency(\.memoRepository) var memoRepository

    var body: some ReducerOf<Self> {
        Reduce { state, action in
            switch action {
            case .onAppear:
                state.isLoading = true
                let memoID = state.memoID
                return .run { send in
                    let result = await Result {
                        try await memoRepository.fetchMemoDetail(id: memoID)
                    }
                    await send(.memoLoaded(result))
                }

            case let .memoLoaded(.success(detail)):
                state.isLoading = false
                state.title = detail.title
                state.createdAt = detail.createdAt
                state.updatedAt = detail.updatedAt
                state.durationSeconds = detail.durationSeconds
                state.audioFilePath = detail.audioFilePath
                state.transcriptionText = detail.transcriptionText
                state.transcriptionLanguage = detail.transcriptionLanguage
                state.transcriptionConfidence = detail.transcriptionConfidence
                state.aiSummary = detail.aiSummary
                state.isAISummaryAvailable = detail.aiSummary != nil
                state.emotion = detail.emotion
                state.tags = detail.tags
                return .none

            case let .memoLoaded(.failure(error)):
                state.isLoading = false
                state.errorMessage = error.localizedDescription
                return .none

            default:
                return .none
            }
        }
    }
}
```

### 3.2 MemoDetailView（SwiftUI）

```swift
// FeatureMemo/MemoDetail/MemoDetailView.swift
// 設計書 04-ui-design-system.md セクション6.3 準拠

struct MemoDetailView: View {
    @Bindable var store: StoreOf<MemoDetailReducer>

    var body: some View {
        ScrollView {
            VStack(alignment: .leading, spacing: 16) {
                // タイトル（Title2: 22pt）
                Text(store.title)
                    .font(.vmTitle2)
                    .foregroundColor(.vmTextPrimary)

                // メタ情報
                MetaInfoRow(
                    date: store.createdAt,
                    duration: store.durationSeconds
                )

                // 音声プレイヤー（TASK-0014で実体実装）
                AudioPlayerPlaceholder()

                // 感情バッジ（大サイズ）
                if let emotion = store.emotion {
                    EmotionDetailCard(emotion: emotion)
                }

                // タグ一覧
                if !store.tags.isEmpty {
                    TagFlowLayout(tags: store.tags) { tagName in
                        store.send(.tagTapped(tagName))
                    }
                }

                // AI要約カード（枠のみ、Phase 3で実装）
                AISummarySection(summary: store.aiSummary)

                // 文字起こしセクション
                TranscriptionSection(text: store.transcriptionText)

                // プライバシー表示
                PrivacyIndicator()
            }
            .padding(16)
        }
        .background(Color.vmBackground)
        .navigationBarTitleDisplayMode(.inline)
        .toolbar {
            ToolbarItem(placement: .topBarTrailing) {
                HStack(spacing: 12) {
                    Button { store.send(.editButtonTapped) } label: {
                        Image(systemName: "pencil")
                    }
                    Menu {
                        Button { store.send(.shareButtonTapped) } label: {
                            Label("共有", systemImage: "square.and.arrow.up")
                        }
                        Button(role: .destructive) {
                            store.send(.deleteButtonTapped)
                        } label: {
                            Label("削除", systemImage: "trash")
                        }
                    } label: {
                        Image(systemName: "ellipsis.circle")
                    }
                }
            }
        }
        .onAppear { store.send(.onAppear) }
    }
}
```

```swift
@MainActor
func test_onAppear_メモ詳細データのロード() async {
    let memoID = UUID()
    let testDetail = MemoDetailReducer.MemoDetail(
        id: memoID,
        title: "テストメモ",
        createdAt: Date(),
        updatedAt: Date(),
        durationSeconds: 180,
        audioFilePath: "Audio/test.m4a",
        transcriptionText: "テスト文字起こしテキスト全文",
        transcriptionLanguage: "ja-JP",
        transcriptionConfidence: 0.95,
        aiSummary: nil,
        emotion: .init(category: .joy, confidence: 0.85, description: "ポジティブな感情"),
        tags: [.init(id: UUID(), name: "テスト", source: "ai")]
    )

    let store = TestStore(
        initialState: MemoDetailReducer.State(memoID: memoID)
    ) {
        MemoDetailReducer()
    } withDependencies: {
        $0.memoRepository.fetchMemoDetail = { _ in testDetail }
    }

    await store.send(.onAppear) {
        $0.isLoading = true
    }

    await store.receive(.memoLoaded(.success(testDetail))) {
        $0.isLoading = false
        $0.title = "テストメモ"
        $0.transcriptionText = "テスト文字起こしテキスト全文"
        $0.emotion = .init(category: .joy, confidence: 0.85, description: "ポジティブな感情")
        $0.tags = [.init(id: testDetail.tags[0].id, name: "テスト", source: "ai")]
        $0.isAISummaryAvailable = false
    }
}
```
