// SharedUI/Accessibility/DynamicTypeSupport.swift

import SwiftUI

// MARK: - @ScaledMetric for layout values
struct VMScaledMetrics {
    /// カード内パディング（ベース16pt）
    @ScaledMetric(relativeTo: .body) static var cardPadding: CGFloat = 16
    /// アイコンサイズ（ベース24pt）
    @ScaledMetric(relativeTo: .body) static var iconSize: CGFloat = 24
    /// バッジパディング（ベース10pt）
    @ScaledMetric(relativeTo: .caption) static var badgePadding: CGFloat = 10
}

// MARK: - レイアウト適応（アクセシビリティサイズ対応）
struct AdaptiveMetaRow: View {
    @Environment(\.sizeCategory) var sizeCategory
    let date: String
    let duration: String

    var body: some View {
        if sizeCategory.isAccessibilityCategory {
            // アクセシビリティサイズ時: 縦積み
            VStack(alignment: .leading, spacing: VMSpacing.xs) {
                Label(date, systemImage: "calendar")
                Label(duration, systemImage: "mic.fill")
            }
            .font(.vmCaption1)
        } else {
            // 通常サイズ: 横並び
            HStack {
                Label(date, systemImage: "calendar")
                Spacer()
                Label(duration, systemImage: "mic.fill")
            }
            .font(.vmCaption1)
        }
    }
}

// MARK: - 適応型感情バッジ行
struct AdaptiveEmotionRow: View {
    @Environment(\.sizeCategory) var sizeCategory
    let emotions: [EmotionCategory]
    let tags: [String]

    var body: some View {
        if sizeCategory.isAccessibilityCategory {
            VStack(alignment: .leading, spacing: VMSpacing.xs) {
                ForEach(emotions, id: \.self) { emotion in
                    EmotionBadge(emotion: emotion)
                }
                ForEach(tags.prefix(3), id: \.self) { tag in
                    TagChip(text: tag)
                }
            }
        } else {
            HStack(spacing: VMSpacing.sm) {
                ForEach(emotions, id: \.self) { emotion in
                    EmotionBadge(emotion: emotion)
                }
                ForEach(tags.prefix(3), id: \.self) { tag in
                    TagChip(text: tag)
                }
            }
        }
    }
}

### 3. Reduce Motion対応 :large_blue_circle:

04-ui-design-system.md セクション8.5準拠。`accessibilityReduceMotion` 環境変数で条件分岐。

```swift
// SharedUI/Accessibility/MotionAwareModifiers.swift

import SwiftUI

// MARK: - Reduce Motion対応 ViewModifier
struct MotionAwareAnimation: ViewModifier {
    @Environment(\.accessibilityReduceMotion) var reduceMotion
    let animation: Animation

    func body(content: Content) -> some View {
        content.animation(reduceMotion ? .none : animation)
    }
}

extension View {
    /// Reduce Motion対応のアニメーション適用
    func vmMotionAwareAnimation(_ animation: Animation) -> some View {
        modifier(MotionAwareAnimation(animation: animation))
    }
}

// MARK: - Reduce Motion対応の録音パルスアニメーション
struct MotionAwarePulseRing: View {
    @Environment(\.accessibilityReduceMotion) var reduceMotion
    @State private var scale: CGFloat = 1.0
    @State private var opacity: Double = 0.6

    var body: some View {
        Circle()
            .stroke(Color.vmError.opacity(reduceMotion ? 0.3 : opacity), lineWidth: 2)
            .frame(width: 100, height: 100)
            .scaleEffect(reduceMotion ? 1.0 : scale)
            .onAppear {
                guard !reduceMotion else { return }
                withAnimation(
                    .easeInOut(duration: 1.5)
                    .repeatForever(autoreverses: true)
                ) {
                    scale = 1.4
                    opacity = 0.0
                }
            }
    }
}

// MARK: - Reduce Motion対応の感情表示アニメーション
struct MotionAwareEmotionReveal: ViewModifier {
    @Environment(\.accessibilityReduceMotion) var reduceMotion
    let isVisible: Bool
    @State private var appeared = false

    func body(content: Content) -> some View {
        content
            .scaleEffect(reduceMotion ? 1.0 : (appeared ? 1.0 : 0.3))
            .opacity(appeared || reduceMotion ? 1.0 : 0.0)
            .onChange(of: isVisible) { newValue in
                if newValue {
                    if reduceMotion {
                        appeared = true
                    } else {
                        withAnimation(.vmMicroInteraction) {
                            appeared = true
                        }
                    }
                }
            }
    }
}

// MARK: - Reduce Motion対応の画面遷移
struct MotionAwareTransition: ViewModifier {
    @Environment(\.accessibilityReduceMotion) var reduceMotion
    let isVisible: Bool

    func body(content: Content) -> some View {
        content
            .offset(y: reduceMotion ? 0 : (isVisible ? 0 : -20))
            .opacity(isVisible ? 1 : 0)
            .animation(reduceMotion ? .none : .vmNavigationPush, value: isVisible)
    }
}
```

import SwiftUI

// MARK: - 最小タップターゲット保証 ViewModifier
struct MinimumTapTarget: ViewModifier {
    let minSize: CGFloat

    func body(content: Content) -> some View {
        content

### 5. コントラスト比検証ユーティリティ :large_blue_circle:

WCAG 2.1 AA基準（通常テキスト4.5:1、大テキスト3:1）の自動検証。

```swift
// SharedUI/Accessibility/ContrastChecker.swift

import SwiftUI

/// WCAG 2.1 コントラスト比計算ユーティリティ
enum ContrastChecker {
    /// 2色間のコントラスト比を計算
    static func contrastRatio(foreground: Color, background: Color) -> Double {
        let fgLuminance = relativeLuminance(foreground)
        let bgLuminance = relativeLuminance(background)
        let lighter = max(fgLuminance, bgLuminance)
        let darker = min(fgLuminance, bgLuminance)
        return (lighter + 0.05) / (darker + 0.05)
    }

    /// WCAG AA基準を満たすか判定
    static func meetsAA(
        foreground: Color,
        background: Color,
        isLargeText: Bool = false
    ) -> Bool {
        let ratio = contrastRatio(foreground: foreground, background: background)
        let threshold = isLargeText ? 3.0 : 4.5
        return ratio >= threshold
    }

    private static func relativeLuminance(_ color: Color) -> Double {
        // UIColor経由でRGB成分を取得し、相対輝度を計算
        let uiColor = UIColor(color)
        var r: CGFloat = 0, g: CGFloat = 0, b: CGFloat = 0, a: CGFloat = 0
        uiColor.getRed(&r, green: &g, blue: &b, alpha: &a)

        let rLinear = linearize(r)
        let gLinear = linearize(g)
        let bLinear = linearize(b)

        return 0.2126 * rLinear + 0.7152 * gLinear + 0.0722 * bLinear
    }

    private static func linearize(_ value: CGFloat) -> Double {
        let v = Double(value)
        return v <= 0.03928 ? v / 12.92 : pow((v + 0.055) / 1.055, 2.4)
    }
}
```

## テスト要件

### VoiceOver
- 録音ボタンの各状態（idle/recording/paused/processing）で正しいラベル/ヒントが返される
- メモカードのcombined accessibilityLabelにタイトル・日時・時間が含まれる
- 感情バッジのラベルが「感情: {ラベル名}」の形式
- タグチップのラベルが「タグ: {テキスト}」の形式
- 検索バーにisSearchFieldトレイトが設定されている
- 波形ビジュアライザにupdatesFrequentlyトレイトが設定されている

### Dynamic Type
- `Font.vmBody()` のrelativeToが.bodyである
- `Font.vmCallout` のrelativeToが.calloutである
- `Font.vmSubheadline` のrelativeToが.subheadlineである
- `AdaptiveMetaRow` がisAccessibilityCategoryの場合にVStackレイアウトになる
- `AdaptiveMetaRow` が通常サイズの場合にHStackレイアウトになる

### Reduce Motion
- `accessibilityReduceMotion` がtrueの場合、パルスアニメーションが無効になる
- `accessibilityReduceMotion` がtrueの場合、感情表示のスケールエフェクトがスキップされる
- `accessibilityReduceMotion` がfalseの場合、通常のアニメーションが適用される

### タップターゲット
- `.vmMinTapTarget()` 適用後のframeが44x44pt以上
- 小さいコンポーネント（タグチップ、ツールバーボタン等）にタップ領域が拡張されている

### コントラスト比
- `vmTextPrimary` / `vmBackground` のコントラスト比が4.5:1以上
- `vmTextSecondary` / `vmBackground` のコントラスト比が4.5:1以上
- `vmCTAPrimary` / White のコントラスト比が4.5:1以上
- `vmError` / White のコントラスト比が3:1以上（大テキスト/アイコン用途）
- `vmTextTertiary` のコントラスト比が3:1以上（大テキスト用途限定を確認）
