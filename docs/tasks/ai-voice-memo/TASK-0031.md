```markdown
{文字起こし全文}

---
*VoiceMemoアプリからエクスポート*
```

### 2. MarkdownExporter実装 :large_blue_circle:

Domain層にエクスポートロジックを配置し、Feature層から呼び出す。

```swift
// Domain/Export/MarkdownExporter.swift

struct MarkdownExporter {
    /// メモデータからMarkdown文字列を生成
    func export(memo: VoiceMemo) -> String {
        var lines: [String] = []

        // タイトル
        lines.append("# \(memo.title.isEmpty ? "無題のメモ" : memo.title)")
        lines.append("")

        // メタデータ
        let formatter = DateFormatter()
        formatter.dateFormat = "yyyy/MM/dd HH:mm"
        formatter.locale = Locale(identifier: "ja_JP")
        lines.append("**日時**: \(formatter.string(from: memo.createdAt))")
        lines.append("**録音時間**: \(formatDuration(memo.durationSeconds))")

        // 感情
        if let emotion = memo.emotionAnalysis {
            let icon = emotion.primaryEmotion.iconName
            let label = emotion.primaryEmotion.label
            let confidence = Int(emotion.confidence * 100)
            lines.append("**感情**: \(label)（信頼度: \(confidence)%）")
        }
        lines.append("")

        // タグ
        if !memo.tags.isEmpty {
            lines.append("## タグ")
            lines.append(memo.tags.map(\.name).joined(separator: ", "))
            lines.append("")
        }

        // AI要約
        if let summary = memo.aiSummary {
            lines.append("## AI要約")
            lines.append(summary.summaryText)
            lines.append("")

            if !summary.keyPoints.isEmpty {
                lines.append("### キーポイント")
                for point in summary.keyPoints {
                    lines.append("- \(point)")
                }
                lines.append("")
            }
        }

        // 文字起こし
        if let transcription = memo.transcription {
            lines.append("## 文字起こし")
            lines.append(transcription.fullText)
            lines.append("")
        }

        // フッター
        lines.append("---")
        lines.append("*VoiceMemoアプリからエクスポート*")

        return lines.joined(separator: "\n")
    }

    private func formatDuration(_ seconds: Double) -> String {
        let minutes = Int(seconds) / 60
        let secs = Int(seconds) % 60
        return String(format: "%d:%02d", minutes, secs)
    }
}
```

### 3. TCA Reducer統合（共有アクション） :large_blue_circle:

MemoDetailReducerにエクスポートアクションを追加。

```swift
// FeatureMemo/MemoDetail/MemoDetailReducer.swift（追加部分）

@Reducer
struct MemoDetailReducer {
    @ObservableState
    struct State {
        // ... 既存フィールド
        @Presents var shareSheet: ShareSheetState?
        var showUpgradePrompt: Bool = false
    }

    enum Action {
        // ... 既存アクション
        case exportMarkdownTapped
        case shareSheetPresented(ShareSheetState)
        case shareSheetDismissed
        case upgradePromptDismissed
    }

    @Dependency(\.subscriptionClient) var subscriptionClient

    var body: some ReducerOf<Self> {
        Reduce { state, action in
            switch action {
            case .exportMarkdownTapped:
                // Pro限定チェック
                guard subscriptionClient.currentPlan() == .pro else {
                    state.showUpgradePrompt = true
                    return .none
                }

                let exporter = MarkdownExporter()
                let markdown = exporter.export(memo: state.memo)
                state.shareSheet = ShareSheetState(
                    items: [markdown],
                    filename: "\(state.memo.title).md"
                )
                return .none

            case .shareSheetDismissed:
                state.shareSheet = nil
                return .none

            case .upgradePromptDismissed:
                state.showUpgradePrompt = false
                return .none

            default:
                return .none
            }
        }
    }
}
```

### 4. UIActivityViewController統合 :large_blue_circle:

SwiftUIからUIActivityViewControllerを呼び出すラッパー。

```swift
// SharedUI/Components/ShareSheet.swift

import SwiftUI

struct ShareSheetState: Equatable, Identifiable {
    let id = UUID()
    let items: [Any]
    let filename: String?

    static func == (lhs: ShareSheetState, rhs: ShareSheetState) -> Bool {
        lhs.id == rhs.id
    }
}

struct ShareSheet: UIViewControllerRepresentable {
    let items: [Any]

    func makeUIViewController(context: Context) -> UIActivityViewController {
        UIActivityViewController(activityItems: items, applicationActivities: nil)
    }

    func updateUIViewController(_ uiViewController: UIActivityViewController, context: Context) {}
}

// View側での使用
extension View {
    func shareSheet(item: Binding<ShareSheetState?>) -> some View {
        self.sheet(item: item) { state in
            ShareSheet(items: state.items)
        }
    }
}
```

## テスト要件

### 正常系
- 全フィールド（タイトル、日時、録音時間、感情、タグ、AI要約、キーポイント、文字起こし）が含まれるMarkdownが生成される
- タイトルが空の場合「無題のメモ」が使用される
- タグが空の場合、タグセクションが省略される
- AI要約がない場合、AI要約セクションが省略される
- 感情分析がない場合、感情行が省略される
- 文字起こしがない場合、文字起こしセクションが省略される
- Proプランユーザーが共有シートを開ける

### 異常系
- Freeプランユーザーがエクスポートを試みた場合、アップグレード案内が表示される
- メモの全フィールドがnilの場合でもクラッシュしない

### TCA TestStore
- `.exportMarkdownTapped` アクション送信時、Proプランで `shareSheet` が設定される
- `.exportMarkdownTapped` アクション送信時、Freeプランで `showUpgradePrompt` がtrueになる
