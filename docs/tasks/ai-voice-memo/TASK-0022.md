### 1. サブスクリプション検証エンドポイント :large_blue_circle:

03-backend-proxy.mdセクション5.2準拠。appAccountToken検証あり。

```typescript
// src/routes/subscription.ts
import { Hono } from 'hono';
import { z } from 'zod';
import { zValidator } from '@hono/zod-validator';
import type { Env } from '../types/env';

const subscription = new Hono<{ Bindings: Env }>();

const verifySchema = z.object({
  transaction_jws: z.string(),
});

// POST /api/v1/subscription/verify
subscription.post('/verify',
  zValidator('json', verifySchema),
  async (c) => {
    const { transaction_jws } = c.req.valid('json');
    const user = c.get('user');

    // 1. JWS署名検証（Apple Root CA証明書チェーン）
    const transactionPayload = await verifyTransactionJWS(transaction_jws, c.env);

    // 2. appAccountToken検証（Critical #5）
    const expectedToken = generateAppAccountToken(user.id);
    if (transactionPayload.appAccountToken !== expectedToken) {
      return c.json({
        error: {
          code: 'SUBSCRIPTION_USER_MISMATCH',
          message: 'appAccountTokenがリクエストユーザーと一致しません',
        }
      }, 403);
    }

    // 3. Bundle ID検証
    if (transactionPayload.bundleId !== c.env.APPLE_CLIENT_ID) {
      return c.json({
        error: { code: 'INVALID_BUNDLE_ID', message: 'Bundle IDが一致しません' }
      }, 403);
    }

    // 4. 有効期限確認
    const now = Date.now();
    const isActive = transactionPayload.expiresDate > now;
    const plan = isActive ? 'pro' : 'free';

    // 5. D1にサブスクリプション情報を保存/更新
    const subId = generateId('sub');
    const nowISO = new Date().toISOString();

    await c.env.DB.prepare(
      `INSERT INTO subscriptions
       (id, user_id, product_id, app_account_token, original_transaction_id,
        status, expires_at, auto_renew, created_at, updated_at)
       VALUES (?, ?, ?, ?, ?, ?, ?, 1, ?, ?)
       ON CONFLICT(user_id, product_id) DO UPDATE SET
         status = excluded.status,
         expires_at = excluded.expires_at,
         updated_at = excluded.updated_at`
    ).bind(
      subId, user.id, transactionPayload.productId,
      expectedToken, transactionPayload.originalTransactionId,
      isActive ? 'active' : 'expired',
      new Date(transactionPayload.expiresDate).toISOString(),
      nowISO, nowISO
    ).run();

    // 6. ユーザーのプラン更新
    await c.env.DB.prepare(
      'UPDATE users SET plan = ?, updated_at = ? WHERE id = ?'
    ).bind(plan, nowISO, user.id).run();

    // 7. 新JWTを発行（plan更新済み）
    const accessToken = await generateJWT({
      sub: user.id,
      plan,
      auth_method: user.authMethod,
      device_id: user.deviceId,
    }, c.env);

    return c.json({
      access_token: accessToken,
      subscription: {
        status: isActive ? 'active' : 'expired',
        product_id: transactionPayload.productId,
        expires_at: new Date(transactionPayload.expiresDate).toISOString(),
      },
    });
  }
);
```

### 2. appAccountToken生成・検証 :large_blue_circle:

03-backend-proxy.mdセクション5.2準拠。UUIDv5による決定論的生成。

```typescript
// src/services/appAccountToken.ts
import { v5 as uuidv5 } from 'uuid';

const APP_ACCOUNT_TOKEN_NAMESPACE = '6ba7b810-9dad-11d1-80b4-00c04fd430c8';

export function generateAppAccountToken(userId: string): string {
  return uuidv5(userId, APP_ACCOUNT_TOKEN_NAMESPACE);
}

export async function resolveUserByAppAccountToken(
  appAccountToken: string,
  db: D1Database
): Promise<string | null> {
  const sub = await db.prepare(
    'SELECT user_id FROM subscriptions WHERE app_account_token = ?'
  ).bind(appAccountToken).first();
  return sub?.user_id as string | null;
}
```

### 3. Server Notification V2 Webhook :large_blue_circle:

03-backend-proxy.mdセクション5.3, 5.4準拠。冪等性 + signedDate順序制御。

```typescript
// POST /api/v1/webhook/appstore
subscription.post('/webhook',
  async (c) => {
    const { signedPayload } = await c.req.json();

    // 1. JWS署名検証（Apple Root CA）
    const payload = await verifyAppStoreNotification(signedPayload, c.env);

    // 2. notificationUUID冪等性チェック（Critical #6）
    const notificationUUID = payload.notificationUUID;
    const existing = await c.env.DB.prepare(
      'SELECT id FROM webhook_events WHERE notification_uuid = ?'
    ).bind(notificationUUID).first();

    if (existing) {
      return c.json({ status: 'already_processed' }, 200);
    }

    // 3. トランザクション情報デコード
    const transactionInfo = await decodeTransactionInfo(
      payload.data.signedTransactionInfo, c.env
    );
    const signedDate = payload.signedDate;

    // 4. signedDate順序チェック（High: 順序競合対策）
    const latestEvent = await c.env.DB.prepare(
      'SELECT MAX(signed_date) as max_signed_date FROM webhook_events WHERE original_transaction_id = ?'
    ).bind(transactionInfo.originalTransactionId).first();

    if (latestEvent?.max_signed_date && signedDate < (latestEvent.max_signed_date as number)) {
      await recordWebhookEvent(c.env.DB, {
        notificationUUID, notificationType: payload.notificationType,
        subtype: payload.subtype, originalTransactionId: transactionInfo.originalTransactionId,
        signedDate, status: 'skipped',
      });
      return c.json({ status: 'outdated_event_skipped' }, 200);
    }

    // 5. appAccountTokenでユーザー照合（Critical #5）
    const userId = await resolveUserByAppAccountToken(
      transactionInfo.appAccountToken, c.env.DB
    );
    if (!userId) {
      return c.json({ status: 'user_not_found' }, 200);
    }

    // 6. 通知タイプ別処理
    await processNotification(payload, userId, transactionInfo, c.env);

    // 7. Webhookイベント記録
    await recordWebhookEvent(c.env.DB, {
      notificationUUID, notificationType: payload.notificationType,
      subtype: payload.subtype, originalTransactionId: transactionInfo.originalTransactionId,
      signedDate, status: 'processed',
    });

    return c.json({ status: 'processed' }, 200);
  }
);
```

### 4. 通知タイプ別処理 :large_blue_circle:

03-backend-proxy.mdセクション5.3の通知タイプ一覧準拠。

```typescript
// src/services/notification.ts
async function processNotification(
  payload: NotificationPayload,
  userId: string,
  transactionInfo: TransactionInfo,
  env: Env
): Promise<void> {
  const now = new Date().toISOString();
  const { notificationType, subtype } = payload;

  switch (notificationType) {
    case 'SUBSCRIBED':
      // INITIAL_BUY / RESUBSCRIBE → plan を pro に更新
      await env.DB.batch([
        env.DB.prepare('UPDATE users SET plan = ?, updated_at = ? WHERE id = ?')
          .bind('pro', now, userId),
        env.DB.prepare(
          `UPDATE subscriptions SET status = 'active', expires_at = ?, updated_at = ? WHERE user_id = ?`
        ).bind(new Date(transactionInfo.expiresDate).toISOString(), now, userId),
      ]);
      break;

    case 'DID_RENEW':
      // 自動更新成功 → expires_at 更新
      await env.DB.prepare(
        `UPDATE subscriptions SET status = 'active', expires_at = ?, updated_at = ? WHERE user_id = ?`
      ).bind(new Date(transactionInfo.expiresDate).toISOString(), now, userId).run();
      break;

    case 'DID_FAIL_TO_RENEW':
      if (subtype === 'GRACE_PERIOD') {
        // 猶予期間: Pro維持（EC-015）
        await env.DB.prepare(
          `UPDATE subscriptions SET status = 'grace_period', updated_at = ? WHERE user_id = ?`
        ).bind(now, userId).run();
      } else {
        // 更新失敗: billing_retry状態
        await env.DB.prepare(
          `UPDATE subscriptions SET status = 'billing_retry', updated_at = ? WHERE user_id = ?`
        ).bind(now, userId).run();
      }
      break;

    case 'EXPIRED':
    case 'REVOKE':
    case 'REFUND':
      // plan を free に降格
      await env.DB.batch([
        env.DB.prepare('UPDATE users SET plan = ?, updated_at = ? WHERE id = ?')
          .bind('free', now, userId),
        env.DB.prepare(
          `UPDATE subscriptions SET status = ?, updated_at = ? WHERE user_id = ?`
        ).bind(notificationType === 'REFUND' ? 'refunded' : 'expired', now, userId),
      ]);
      break;

    case 'DID_CHANGE_RENEWAL_STATUS':
      const autoRenew = subtype !== 'AUTO_RENEW_DISABLED' ? 1 : 0;
      await env.DB.prepare(
        `UPDATE subscriptions SET auto_renew = ?, updated_at = ? WHERE user_id = ?`
      ).bind(autoRenew, now, userId).run();
      break;
  }
}
```

// GET /api/v1/usage
// src/routes/usage.ts
const usage = new Hono<{ Bindings: Env }>();

usage.get('/usage', async (c) => {
  const user = c.get('user');

  // 使用量取得（D1）
  const jstYearMonth = getJSTYearMonth();
  const usageRecord = await c.env.DB.prepare(
    'SELECT count FROM usage_counters WHERE user_id = ? AND year_month = ?'
  ).bind(user.id, jstYearMonth).first();

  const used = (usageRecord?.count as number) || 0;

  // サブスクリプション情報取得
  const sub = await c.env.DB.prepare(
    'SELECT product_id, status, expires_at, auto_renew FROM subscriptions WHERE user_id = ? ORDER BY updated_at DESC LIMIT 1'
  ).bind(user.id).first();

  if (user.plan === 'pro') {
    return c.json({
      plan: 'pro',
      ai_processing: {
        used,
        limit: null,
        resets_at: null,
      },
      cloud_stt: { available: true, reason: null },
      subscription: sub ? {
        status: sub.status,
        product_id: sub.product_id,
        expires_at: sub.expires_at,
        auto_renew: !!sub.auto_renew,
      } : null,
      rate_limit: {
        daily_remaining: 1000 - used,
        resets_at: getNextDayResetTime(),
      },
    });
  }

  return c.json({
    plan: 'free',
    ai_processing: {
      used,
      limit: 5,
      resets_at: getNextMonthResetTime(),
    },
    cloud_stt: { available: false, reason: 'Pro plan required' },
    subscription: null,
    rate_limit: { daily_remaining: null, resets_at: null },
  });
});

## テスト要件

### 正常系
- POST /api/v1/subscription/verify が有効なJWSを検証しサブスクリプション状態をactiveに更新する
- appAccountTokenが正しく生成されuser_idと一致確認される
- 検証成功後にplan=proの新JWTが発行される
- Webhook: SUBSCRIBED/INITIAL_BUYでplan=proに更新
- Webhook: DID_RENEWでexpires_at更新
- Webhook: DID_FAIL_TO_RENEW/GRACE_PERIODでPro維持
- Webhook: EXPIRED/REVOKE/REFUNDでplan=freeに降格
- Webhook: DID_CHANGE_RENEWAL_STATUSでauto_renewフラグ更新
- GET /api/v1/usage がFree/Proそれぞれ正しいレスポンスを返す

### 冪等性・順序制御（Critical #5, #6）
- 同じnotificationUUIDの重複通知が「already_processed」で返される
- 古いsignedDateの通知が「outdated_event_skipped」で処理スキップされる
- appAccountToken不一致で403 SUBSCRIPTION_USER_MISMATCHが返る
- appAccountTokenで不明ユーザーの場合に200 user_not_foundが返る（Apple再送防止）

### 異常系
- JWS署名検証失敗時のエラーハンドリング
- Bundle ID不一致で403が返る
- 有効期限切れトランザクションでplan=freeに設定される
- D1書き込み失敗時のロールバック
