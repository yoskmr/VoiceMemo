### 1. CloudLLMProvider実装 :large_blue_circle:

02-ai-pipeline.mdセクション3.3.5準拠。

```swift
// InfraLLM/Cloud/CloudLLMProvider.swift
import Foundation

final class CloudLLMProvider: LLMProviderProtocol {
    let providerType: LLMProviderType = .cloudGPT4oMini
    let isOnDevice: Bool = false
    let supportedTasks: Set<LLMTask> = Set(LLMTask.allCases) // 全タスク対応
    let maxInputTokens: Int = 16384

    private let baseURL: URL
    private let session: URLSession
    private let requestSigner: RequestSigner

    init(baseURL: URL, requestSigner: RequestSigner) {
        self.baseURL = baseURL
        self.requestSigner = requestSigner

        let config = URLSessionConfiguration.default
        config.timeoutIntervalForRequest = 30  // 30秒タイムアウト
        config.timeoutIntervalForResource = 60
        self.session = URLSession(configuration: config)
    }

    func isAvailable() async -> Bool {
        NetworkMonitor.shared.isConnected
    }

    func process(_ request: LLMRequest) async throws -> LLMResponse {
        let endpoint = baseURL.appendingPathComponent("/api/v1/ai/process")

        // リクエストボディ構築
        let payload = CloudAIRequest(
            text: request.text,
            language: request.language,
            options: CloudAIOptions(
                summary: request.tasks.contains(.summarize),
                tags: request.tasks.contains(.tagging),
                sentiment: request.tasks.contains(.sentimentAnalysis)
            )
        )

        let bodyData = try JSONEncoder().encode(payload)

        // リトライ（指数バックオフ、最大3回）
        return try await withRetry(maxAttempts: 3) {
            var urlRequest = URLRequest(url: endpoint)
            urlRequest.httpMethod = "POST"
            urlRequest.setValue("application/json", forHTTPHeaderField: "Content-Type")
            urlRequest.setValue(UUID().uuidString, forHTTPHeaderField: "X-Request-ID")
            urlRequest.setValue(Bundle.main.appVersion, forHTTPHeaderField: "X-App-Version")
            urlRequest.httpBody = bodyData

            // JWT認証 + App Attest assertion
            let authToken = try self.getAuthToken()
            try await self.requestSigner.sign(
                request: &urlRequest,
                body: bodyData,
                token: authToken
            )

            let startTime = CFAbsoluteTimeGetCurrent()
            let (data, response) = try await self.session.data(for: urlRequest)
            let elapsed = CFAbsoluteTimeGetCurrent() - startTime

            guard let httpResponse = response as? HTTPURLResponse else {
                throw LLMError.invalidResponse
            }

            switch httpResponse.statusCode {
            case 200:
                let cloudResponse = try JSONDecoder().decode(CloudAIResponse.self, from: data)
                return self.mapToLLMResponse(cloudResponse, processingTimeMs: Int(elapsed * 1000))
            case 429:
                throw LLMError.rateLimited
            case 402:
                throw LLMError.quotaExceeded
            case 401:
                throw LLMError.unauthorized
            default:
                throw LLMError.serverError(statusCode: httpResponse.statusCode)
            }
        }
    }

    private func getAuthToken() throws -> String {
        guard let tokenData = SecureKeychain.load(key: .accessToken),
              let token = String(data: tokenData, encoding: .utf8) else {
            throw LLMError.unauthorized
        }
        return token
    }
}
```

### 2. リトライ（指数バックオフ） :large_blue_circle:

02-ai-pipeline.mdセクション5.3リトライ戦略準拠。

```swift
// SharedUtil/Network/RetryHelper.swift

/// 指数バックオフリトライ
/// - Parameters:
///   - maxAttempts: 最大リトライ回数（初回含む）
///   - baseDelay: 初回リトライの待機秒数
///   - operation: リトライ対象の非同期操作
func withRetry<T>(
    maxAttempts: Int = 3,
    baseDelay: TimeInterval = 1.0,
    operation: @escaping () async throws -> T
) async throws -> T {
    var lastError: Error?

    for attempt in 0..<maxAttempts {
        do {
            return try await operation()
        } catch let error as LLMError {
            lastError = error

            // リトライ不可能なエラーは即座に伝播
            switch error {
            case .quotaExceeded, .unauthorized, .unsupportedTask, .inputTooLong:
                throw error
            case .rateLimited, .serverError, .invalidResponse, .networkError:
                // リトライ可能
                break
            default:
                throw error
            }

            // 指数バックオフ待機（1秒, 2秒, 4秒）
            if attempt < maxAttempts - 1 {
                let delay = baseDelay * pow(2.0, Double(attempt))
                let jitter = Double.random(in: 0...0.5)
                try await Task.sleep(nanoseconds: UInt64((delay + jitter) * 1_000_000_000))
            }
        }
    }

    throw lastError ?? LLMError.networkError
}
```

// InfraLLM/Cloud/CloudAIModels.swift

/// Backend Proxy AI処理リクエスト
struct CloudAIRequest: Codable {
    let text: String
    let language: String
    let options: CloudAIOptions
}

struct CloudAIOptions: Codable {
    let summary: Bool
    let tags: Bool
    let sentiment: Bool
}

/// Backend Proxy AI処理レスポンス
struct CloudAIResponse: Codable {
    let summary: CloudSummary?
    let tags: [CloudTag]?
    let sentiment: CloudSentiment?
    let usage: CloudUsage
    let metadata: CloudMetadata
}

struct CloudSummary: Codable {
    let title: String
    let brief: String
    let keyPoints: [String]?

    enum CodingKeys: String, CodingKey {
        case title, brief
        case keyPoints = "key_points"
    }
}

struct CloudTag: Codable {
    let label: String
    let confidence: Double
}

struct CloudSentiment: Codable {
    let primary: String  // EmotionCategory raw value
    let scores: [String: Double]
    let evidence: [CloudEvidence]?
}

struct CloudEvidence: Codable {
    let text: String
    let emotion: String
}

struct CloudUsage: Codable {
    let used: Int
    let limit: Int?
    let plan: String
    let resetsAt: String?

    enum CodingKeys: String, CodingKey {
        case used, limit, plan
        case resetsAt = "resets_at"
    }
}

struct CloudMetadata: Codable {
    let model: String
    let provider: String
    let processingTimeMs: Int
    let requestId: String

    enum CodingKeys: String, CodingKey {
        case model, provider
        case processingTimeMs = "processing_time_ms"
        case requestId = "request_id"
    }
}

### 4. レスポンスマッピング :large_blue_circle:

CloudAIResponseをDomain層のLLMResponseに変換。

```swift
// InfraLLM/Cloud/CloudLLMProvider+Mapping.swift
extension CloudLLMProvider {
    func mapToLLMResponse(_ cloud: CloudAIResponse, processingTimeMs: Int) -> LLMResponse {
        LLMResponse(
            summary: cloud.summary.map { summary in
                LLMSummaryResult(
                    title: summary.title,
                    brief: summary.brief,
                    keyPoints: summary.keyPoints ?? []
                )
            },
            tags: cloud.tags?.map { tag in
                LLMTagResult(label: tag.label, confidence: tag.confidence)
            },
            sentiment: cloud.sentiment.map { sentiment in
                let primary = EmotionCategory(rawValue: sentiment.primary) ?? .neutral
                var scores: [EmotionCategory: Double] = [:]
                for (key, value) in sentiment.scores {
                    if let category = EmotionCategory(rawValue: key) {
                        scores[category] = value
                    }
                }
                return LLMSentimentResult(
                    primary: primary,
                    scores: scores,
                    evidence: sentiment.evidence?.map { e in
                        SentimentEvidence(
                            text: e.text,
                            emotion: EmotionCategory(rawValue: e.emotion) ?? .neutral
                        )
                    } ?? []
                )
            },
            processingTimeMs: processingTimeMs,
            provider: cloud.metadata.provider
        )
    }
}
```

## テスト要件

### 正常系
- CloudLLMProviderがLLMProviderProtocolに準拠している
- isAvailable()がネットワーク接続時にtrue、不通時にfalseを返す
- process()が要約+タグ+感情分析の結果を返す
- 感情分析のみ（options.summary=false, options.tags=false）のリクエストが正常に処理される
- JWT認証ヘッダーが正しく設定される
- X-App-Attest-Assertionヘッダーが正しく設定される
- CloudAIResponseがLLMResponseに正しくマッピングされる

### リトライ
- 一時的な502エラーが指数バックオフリトライで回復する
- 2回失敗→3回目で成功のケースが正しく処理される
- リトライ間の待機時間が指数バックオフ（1秒, 2秒）に従う

### 異常系
- ネットワーク不通時にLLMError.networkErrorが発生する
- タイムアウト（30秒超過）時のエラーハンドリング
- 429 Rate Limited で LLMError.rateLimitedが発生する
- 402 Quota Exceeded で LLMError.quotaExceededが発生し、リトライされない
- 401 Unauthorized で LLMError.unauthorizedが発生し、リトライされない
- 不正なJSONレスポンスのデコードエラーハンドリング
