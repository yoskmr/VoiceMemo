## 実装詳細

### 1. KeychainManager :large_blue_circle:

統合仕様書セクション3.3 + セクション8.3準拠。全項目で `ThisDeviceOnly` を必須とする。

```swift
// InfraNetwork/Auth/KeychainManager.swift
import Security
import Foundation

/// Keychain保存属性（統合仕様書準拠）
/// 【正】全設計書でこのenumを使用すること
enum KeychainItemType: String, Sendable {
    case accessToken         = "access_token"
    case refreshToken        = "refresh_token"
    case appleUserID         = "apple_user_id"
    case subscriptionCache   = "subscription_cache"
    case appAttestKeyID      = "app_attest_key_id"

    /// アクセシビリティ属性
    /// 統合仕様書セクション8.3: 全項目で ThisDeviceOnly 必須
    var accessibility: CFString {
        switch self {
        case .appleUserID:
            // ロック解除時のみアクセス可能（最高セキュリティ）
            return kSecAttrAccessibleWhenUnlockedThisDeviceOnly
        case .accessToken, .refreshToken, .subscriptionCache, .appAttestKeyID:
            // 初回ロック解除後はアクセス可能（バックグラウンド処理対応）
            return kSecAttrAccessibleAfterFirstUnlockThisDeviceOnly
        }
    }
}

/// Keychain管理の統一実装
struct KeychainManager: Sendable {
    private let service = "io.murmurnote.api"

    /// データの保存
    func save(key: KeychainItemType, data: Data) throws {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrService as String: service,
            kSecAttrAccount as String: key.rawValue,
            kSecValueData as String: data,
            kSecAttrAccessible as String: key.accessibility,
        ]
        // 既存を削除してから追加（upsert相当）
        SecItemDelete(query as CFDictionary)
        let status = SecItemAdd(query as CFDictionary, nil)
        guard status == errSecSuccess else {
            throw KeychainError.saveFailed(status)
        }
    }

    /// データの読み込み
    func load(key: KeychainItemType) -> Data? {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrService as String: service,
            kSecAttrAccount as String: key.rawValue,
            kSecReturnData as String: true,
            kSecMatchLimit as String: kSecMatchLimitOne,
        ]
        var result: AnyObject?
        let status = SecItemCopyMatching(query as CFDictionary, &result)
        guard status == errSecSuccess else { return nil }
        return result as? Data
    }

    /// データの削除
    func delete(key: KeychainItemType) {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrService as String: service,
            kSecAttrAccount as String: key.rawValue,
        ]
        SecItemDelete(query as CFDictionary)
    }

    /// 全データの削除（サインアウト時）
    func deleteAll() {
        for key in [KeychainItemType.accessToken, .refreshToken, .appleUserID, .subscriptionCache, .appAttestKeyID] {
            delete(key: key)
        }
    }

    enum KeychainError: Error {
        case saveFailed(OSStatus)
    }
}
```

### 2. Data Protection検証ユーティリティ :large_blue_circle:

ディレクトリ/ファイルのData Protection設定を確認するユーティリティ。

```swift
// SharedUtil/Security/DataProtectionValidator.swift
struct DataProtectionValidator {
    /// ファイル/ディレクトリの保護レベルを確認
    static func validateProtection(
        at url: URL,
        expected: FileProtectionType
    ) -> Bool {
        guard let attributes = try? FileManager.default.attributesOfItem(atPath: url.path),
              let protection = attributes[.protectionKey] as? FileProtectionType else {
            return false
        }
        return protection == expected
    }

    /// 全ディレクトリの保護レベルを一括検証
    /// 統合仕様書セクション8.1準拠
    static func validateAllProtections() -> [String: Bool] {
        let documents = FileManager.default.urls(for: .documentDirectory, in: .userDomainMask).first!
        let appSupport = FileManager.default.urls(for: .applicationSupportDirectory, in: .userDomainMask).first!

        return [
            "Documents/Audio/": validateProtection(
                at: documents.appendingPathComponent("Audio"),
                expected: .complete
            ),
            "SecureStore/": validateProtection(
                at: appSupport.appendingPathComponent("SecureStore"),
                expected: .complete
            ),
        ]
    }
}
```

### 4. Info.plist 権限設定 :large_blue_circle:

```xml

<key>NSMicrophoneUsageDescription</key>
<string>音声メモを録音するためにマイクへのアクセスが必要です。録音データはデバイス内にのみ保存されます。</string>

<key>NSSpeechRecognitionUsageDescription</key>
<string>録音した音声をテキストに変換するために音声認識へのアクセスが必要です。認識処理はデバイス上で行われます。</string>
```

### 5. Keychain属性の統一確認表 :large_blue_circle:

統合仕様書セクション8.3準拠。

| データ項目 | Keychainキー | アクセシビリティ |
|:---|:---|:---|
| アクセストークン | `access_token` | `kSecAttrAccessibleAfterFirstUnlockThisDeviceOnly` |
| リフレッシュトークン | `refresh_token` | `kSecAttrAccessibleAfterFirstUnlockThisDeviceOnly` |
| Apple Sign Inユーザー識別子 | `apple_user_id` | `kSecAttrAccessibleWhenUnlockedThisDeviceOnly` |
| サブスクリプション検証キャッシュ | `subscription_cache` | `kSecAttrAccessibleAfterFirstUnlockThisDeviceOnly` |
| App AttestキーID | `app_attest_key_id` | `kSecAttrAccessibleAfterFirstUnlockThisDeviceOnly` |

**重要**: `kSecAttrAccessibleAfterFirstUnlock`（`ThisDeviceOnly`なし）は使用禁止。01-Archの旧仕様から修正済み。
