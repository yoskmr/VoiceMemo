### 1. UsageManager :large_blue_circle:

03-backend-proxy.mdセクション5.5, 3.4.6準拠。ローカル + サーバー同期。

```swift
// FeatureAI/Usage/UsageManager.swift
import Foundation
import SwiftData

final class UsageManager: ObservableObject {
    @Published var currentUsage: UsageInfo = .init()
    @Published var isLimitReached: Bool = false

    private let modelContext: ModelContext
    private let networkClient: NetworkClient
    private let networkMonitor: NetworkMonitor

    struct UsageInfo: Equatable {
        var used: Int = 0
        var limit: Int? = 5         // nil = 無制限（Pro）
        var plan: String = "free"
        var resetsAt: Date?
        var source: UsageSource = .local

        enum UsageSource { case local, server }

        var remaining: Int? {
            guard let limit else { return nil }
            return max(0, limit - used)
        }
    }

    /// AI処理前のチェック（ローカル判定）
    func canProcess() async -> Bool {
        await syncIfNeeded()

        if currentUsage.plan == "pro" { return true }

        guard let limit = currentUsage.limit else { return true }
        return currentUsage.used < limit
    }

    /// 使用量のインクリメント
    func recordUsage() async {
        // ローカルカウンタのインクリメント
        incrementLocalCounter()

        // 非同期でサーバーにも反映（失敗してもローカルは更新済み）
        if networkMonitor.isConnected {
            await syncWithServer()
        }
    }

    /// サーバーとの同期（オンライン時）
    func syncWithServer() async {
        guard networkMonitor.isConnected else { return }

        do {
            let response: ServerUsageResponse = try await networkClient.get("/api/v1/usage")

            await MainActor.run {
                currentUsage = UsageInfo(
                    used: response.aiProcessing.used,
                    limit: response.aiProcessing.limit,
                    plan: response.plan,
                    resetsAt: ISO8601DateFormatter().date(from: response.aiProcessing.resetsAt ?? ""),
                    source: .server
                )
                isLimitReached = currentUsage.remaining == 0
            }
        } catch {
            // サーバー同期失敗時はローカルカウンタを維持
        }
    }
}
```

### 2. ローカルカウンタ管理 :large_blue_circle:

UserSettings（SwiftData）のaiProcessingCountThisMonthを使用。

```swift
// FeatureAI/Usage/UsageManager+Local.swift
extension UsageManager {
    /// ローカルカウンタのインクリメント
    func incrementLocalCounter() {
        let settings = fetchUserSettings()

        // 月次リセット判定（JST基準）
        if shouldResetMonthlyCount(lastResetDate: settings.lastAICountResetDate) {
            settings.aiProcessingCountThisMonth = 0
            settings.lastAICountResetDate = Date()
        }

        settings.aiProcessingCountThisMonth += 1
        try? modelContext.save()

        currentUsage.used = settings.aiProcessingCountThisMonth
        isLimitReached = (currentUsage.remaining ?? 1) <= 0
    }

    /// ローカルカウンタの取得
    func getLocalUsage() -> Int {
        let settings = fetchUserSettings()

        // 月次リセット判定
        if shouldResetMonthlyCount(lastResetDate: settings.lastAICountResetDate) {
            settings.aiProcessingCountThisMonth = 0
            settings.lastAICountResetDate = Date()
            try? modelContext.save()
        }

        return settings.aiProcessingCountThisMonth
    }

    /// 必要に応じてサーバーと同期
    func syncIfNeeded() async {
        // ローカルカウンタをまず読み込み
        let localCount = getLocalUsage()
        currentUsage.used = localCount

        // オンラインかつ最終同期から5分以上経過していれば同期
        if networkMonitor.isConnected {
            await syncWithServer()
        }
    }

    private func fetchUserSettings() -> UserSettings {
        let descriptor = FetchDescriptor<UserSettings>()
        let results = try? modelContext.fetch(descriptor)
        return results?.first ?? UserSettings.default()
    }
}
```

### 3. 月次リセットロジック（JST基準） :large_blue_circle:

EC-014準拠。JST 0:00でリセット。

```swift
// SharedUtil/Time/JSTDateHelper.swift

/// 月次リセットが必要かどうかを判定（EC-014: JST基準）
func shouldResetMonthlyCount(lastResetDate: Date) -> Bool {
    let calendar = Calendar(identifier: .gregorian)
    let jst = TimeZone(identifier: "Asia/Tokyo")!

    var calendarJST = calendar
    calendarJST.timeZone = jst

    let lastResetComponents = calendarJST.dateComponents([.year, .month], from: lastResetDate)
    let nowComponents = calendarJST.dateComponents([.year, .month], from: Date())

    // 年または月が異なればリセット
    return lastResetComponents.year != nowComponents.year
        || lastResetComponents.month != nowComponents.month
}

/// 次の月次リセット日時を取得（JST基準）
func getNextMonthlyResetDate() -> Date {
    let calendar = Calendar(identifier: .gregorian)
    let jst = TimeZone(identifier: "Asia/Tokyo")!

    var calendarJST = calendar
    calendarJST.timeZone = jst

    let now = Date()
    let components = calendarJST.dateComponents([.year, .month], from: now)

    var nextMonthComponents = DateComponents()
    if components.month == 12 {
        nextMonthComponents.year = (components.year ?? 2026) + 1
        nextMonthComponents.month = 1
    } else {
        nextMonthComponents.year = components.year
        nextMonthComponents.month = (components.month ?? 1) + 1
    }
    nextMonthComponents.day = 1
    nextMonthComponents.hour = 0
    nextMonthComponents.minute = 0
    nextMonthComponents.second = 0

    return calendarJST.date(from: nextMonthComponents) ?? now
}
```

// InfraNetwork/RateLimit/RateLimitHeaderParser.swift

struct RateLimitInfo: Equatable {
    let limit: Int?       // X-RateLimit-Limit
    let remaining: Int?   // X-RateLimit-Remaining
    let resetsAt: Date?   // X-RateLimit-Reset

    init(from headers: [String: String]) {
        if let limitStr = headers["X-RateLimit-Limit"], limitStr != "unlimited" {
            limit = Int(limitStr)
        } else {
            limit = nil
        }

        if let remainingStr = headers["X-RateLimit-Remaining"], remainingStr != "unlimited" {
            remaining = Int(remainingStr)
        } else {
            remaining = nil
        }

        if let resetStr = headers["X-RateLimit-Reset"] {
            resetsAt = ISO8601DateFormatter().date(from: resetStr)
        } else {
            resetsAt = nil
        }
    }
}

// SharedUI/Components/UsageLimitBanner.swift
import SwiftUI

struct UsageLimitBanner: View {
    let usage: UsageManager.UsageInfo
    var onUpgradeTapped: (() -> Void)?

    var body: some View {
        if let remaining = usage.remaining, remaining <= 1 && usage.plan == "free" {
            VStack(spacing: 8) {
                HStack {
                    Image(systemName: remaining == 0 ? "exclamationmark.triangle.fill" : "info.circle.fill")
                        .foregroundColor(remaining == 0 ? .orange : .accentColor)

                    if remaining == 0 {
                        Text("今月のAI処理回数の上限に達しました")
                            .font(.subheadline)
                            .fontWeight(.medium)
                    } else {
                        Text("今月のAI処理: 残り\(remaining)回")
                            .font(.subheadline)
                    }
                    Spacer()
                }

                if remaining == 0 {
                    Button(action: { onUpgradeTapped?() }) {
                        HStack {
                            Image(systemName: "star.fill")
                            Text("Proプランにアップグレード")
                        }
                        .font(.subheadline)
                        .fontWeight(.semibold)
                        .frame(maxWidth: .infinity)
                        .padding(.vertical, 10)
                        .background(Color.accentColor)
                        .foregroundColor(.white)
                        .cornerRadius(8)
                    }
                }

                // リセット日時の表示
                if let resetsAt = usage.resetsAt {
                    let formatter = RelativeDateTimeFormatter()
                    Text("リセット: \(formatter.localizedString(for: resetsAt, relativeTo: Date()))")
                        .font(.caption)
                        .foregroundColor(.secondary)
                }
            }
            .padding()
            .background(remaining == 0 ? Color.orange.opacity(0.1) : Color.accentColor.opacity(0.05))
            .cornerRadius(12)
        }
    }
}

## テスト要件

### ローカルカウンタ
- incrementLocalCounter()でaiProcessingCountThisMonthが+1される
- getLocalUsage()が現在月の使用量を返す
- 月が変わった場合にカウンタがリセットされる

### サーバー同期
- syncWithServer()でGET /api/v1/usageが呼ばれ、currentUsageが更新される
- サーバーの値がローカルの値より大きい場合にサーバー値が優先される
- サーバー同期失敗時にローカルカウンタが維持される

### 月次リセット（EC-014 JST基準）
- 3月31日 23:59 JSTでリセット不要
- 4月1日 00:00 JSTでリセット必要
- 12月31日→1月1日の年跨ぎでリセットが正しく動作する
- shouldResetMonthlyCount()がJSTタイムゾーンで正しく判定する

### オフライン優先
- オフライン時にローカルカウンタが使用される
- オフライン時にsyncWithServer()がスキップされる
- オンライン復帰時にサーバーと同期される

### 上限到達UI
- 残り1回でUsageLimitBannerの警告が表示される
- 残り0回でアップグレードボタンが表示される
- アップグレードボタンタップでサブスクリプション画面に遷移する
- リセット日時が相対時間形式で表示される

### レスポンスヘッダー
- X-RateLimit-Limit, X-RateLimit-Remaining, X-RateLimit-Reset が正しくパースされる
- "unlimited" の場合にnilが返される
