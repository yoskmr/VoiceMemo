### 1. RecordingReducer（TCA） :large_blue_circle:

01-Archセクション2.2準拠。`@Dependency` でSTTエンジンとAudioRecorderを注入。

```swift
// FeatureRecording/RecordingReducer.swift
import ComposableArchitecture

@Reducer
struct RecordingReducer {
    @ObservableState
    struct State: Equatable {
        var recordingStatus: RecordingStatus = .idle
        var elapsedTime: TimeInterval = 0
        var partialTranscription: String = ""
        var audioLevel: Float = 0        // 0.0-1.0（波形表示用）
        var confidenceLevel: ConfidenceLevel = .high
        var isPermissionGranted: Bool = false

        enum RecordingStatus: Equatable {
            case idle
            case recording
            case paused
        }
    }

    enum Action: Equatable {
        // ユーザーアクション
        case recordButtonTapped
        case pauseButtonTapped
        case resumeButtonTapped
        case stopButtonTapped
        case permissionRequested
        case permissionResponse(Bool)

        // 内部アクション（Effect からの通知）
        case timerTicked
        case audioLevelUpdated(Float)
        case transcriptionUpdated(String, Double)  // text, confidence
        case recordingCompleted(URL)                // 音声ファイルURL
        case recordingFailed(String)                // エラーメッセージ
    }

    @Dependency(\.audioRecorder) var audioRecorder
    @Dependency(\.sttEngineFactory) var sttEngineFactory
    @Dependency(\.continuousClock) var clock

    var body: some ReducerOf<Self> {
        Reduce { state, action in
            switch action {
            case .recordButtonTapped:
                guard state.isPermissionGranted else {
                    return .send(.permissionRequested)
                }
                state.recordingStatus = .recording
                state.elapsedTime = 0
                return .merge(
                    startRecording(),
                    startTimer(),
                    startSTT()
                )

            case .stopButtonTapped:
                state.recordingStatus = .idle
                return stopRecording()

            case .pauseButtonTapped:
                state.recordingStatus = .paused
                return pauseRecording()

            case .resumeButtonTapped:
                state.recordingStatus = .recording
                return resumeRecording()

            case .timerTicked:
                state.elapsedTime += 1
                return .none

            case let .audioLevelUpdated(level):
                state.audioLevel = level
                return .none

            case let .transcriptionUpdated(text, confidence):
                state.partialTranscription = text
                state.confidenceLevel = ConfidenceLevel(confidence: confidence)
                return .none

            // ... 他のアクション
            }
        }
    }
}
```

### 2. RecordButtonコンポーネント :large_blue_circle:

3状態のアニメーション付き録音ボタン。

```swift
// SharedUI/Components/RecordButton.swift
struct RecordButton: View {
    let status: RecordingReducer.State.RecordingStatus
    let action: () -> Void

    var body: some View {
        Button(action: action) {
            ZStack {
                Circle()
                    .fill(Color.red.opacity(0.15))
                    .frame(width: 80, height: 80)

                switch status {
                case .idle:
                    // 赤い円（録音開始ボタン）
                    Circle()
                        .fill(Color.red)
                        .frame(width: 56, height: 56)

                case .recording:
                    // 赤い正方形（録音停止ボタン）
                    RoundedRectangle(cornerRadius: 4)
                        .fill(Color.red)
                        .frame(width: 28, height: 28)

                case .paused:
                    // 角丸正方形（一時停止中）
                    RoundedRectangle(cornerRadius: 8)
                        .fill(Color.orange)
                        .frame(width: 36, height: 36)
                }
            }
        }
        .animation(.easeInOut(duration: 0.2), value: status)
    }
}
```

### 3. リアルタイム波形アニメーション :large_blue_circle:

60fpsの波形描画。音量レベル（audioLevel）から波形の振幅を算出。

```swift
// SharedUI/Components/WaveformView.swift
struct WaveformView: View {
    let audioLevel: Float
    let isRecording: Bool

    @State private var wavePhase: Double = 0
    private let barCount = 40
    private let timer = Timer.publish(every: 1.0/60.0, on: .main, in: .common).autoconnect()

    var body: some View {
        HStack(spacing: 2) {
            ForEach(0..<barCount, id: \.self) { index in
                RoundedRectangle(cornerRadius: 2)
                    .fill(Color.orange.opacity(0.8))
                    .frame(
                        width: 4,
                        height: barHeight(for: index)
                    )
            }
        }
        .frame(height: 60)
        .onReceive(timer) { _ in
            if isRecording {
                wavePhase += 0.1
            }
        }
    }

    private func barHeight(for index: Int) -> CGFloat {
        let normalizedLevel = CGFloat(audioLevel)
        let wave = sin(Double(index) * 0.3 + wavePhase)
        return max(4, normalizedLevel * 50 * CGFloat(wave + 1) / 2 + 4)
    }
}
```

### 4. 録音経過時間表示 :large_blue_circle:

```swift
// FeatureRecording/Views/TimerView.swift
struct TimerView: View {
    let elapsedTime: TimeInterval

    var body: some View {
        Text(formattedTime)
            .font(.system(size: 48, weight: .light, design: .monospaced))
            .foregroundColor(.primary)
    }

    private var formattedTime: String {
        let minutes = Int(elapsedTime) / 60
        let seconds = Int(elapsedTime) % 60
        return String(format: "%02d:%02d", minutes, seconds)
    }
}
```

### 6. RecordingOverlayView（メイン録音画面） :large_blue_circle:

01-Archセクション2.3準拠。fullScreenCoverとして表示される。暖色系ジャーナル風デザイン。

```swift
// FeatureRecording/Views/RecordingOverlayView.swift
struct RecordingOverlayView: View {
    @Bindable var store: StoreOf<RecordingReducer>

    var body: some View {
        VStack(spacing: 24) {
            Spacer()

            // 録音時間
            TimerView(elapsedTime: store.elapsedTime)

            // 波形アニメーション
            WaveformView(
                audioLevel: store.audioLevel,
                isRecording: store.recordingStatus == .recording
            )

            // リアルタイム文字起こし
            RealtimeTranscriptionView(
                text: store.partialTranscription,
                confidenceLevel: store.confidenceLevel
            )

            Spacer()

            // 録音コントロール
            HStack(spacing: 40) {
                if store.recordingStatus == .recording {
                    // 一時停止ボタン
                    Button { store.send(.pauseButtonTapped) } label: {
                        Image(systemName: "pause.circle.fill")
                            .font(.system(size: 44))
                    }
                }

                RecordButton(status: store.recordingStatus) {
                    switch store.recordingStatus {
                    case .idle:     store.send(.recordButtonTapped)
                    case .recording: store.send(.stopButtonTapped)
                    case .paused:   store.send(.resumeButtonTapped)
                    }
                }

                if store.recordingStatus != .idle {
                    // 停止ボタン
                    Button { store.send(.stopButtonTapped) } label: {
                        Image(systemName: "stop.circle.fill")
                            .font(.system(size: 44))
                            .foregroundColor(.red)
                    }
                }
            }
            .padding(.bottom, 40)
        }
        .background(Color(.systemBackground))
    }
}
```
