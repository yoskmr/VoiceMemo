### 1. SubscriptionManager :large_blue_circle:

03-backend-proxy.mdセクション5準拠。

```swift
// FeatureSubscription/SubscriptionManager.swift
import StoreKit

final class SubscriptionManager: ObservableObject {
    @Published var products: [Product] = []
    @Published var currentPlan: SubscriptionPlan = .free
    @Published var subscriptionStatus: SubscriptionStatus = .none
    @Published var isPurchasing = false

    private let productIds = [
        "app.soyoka.pro.monthly",
        "app.soyoka.pro.yearly",
    ]

    private var updateTask: Task<Void, Never>?
    private let networkClient: NetworkClient

    init(networkClient: NetworkClient) {
        self.networkClient = networkClient
        // Transaction.updates の監視開始
        updateTask = Task { await listenForTransactionUpdates() }
    }

    deinit {
        updateTask?.cancel()
    }

    /// Product一覧を取得
    func loadProducts() async throws {
        products = try await Product.products(for: Set(productIds))
            .sorted { $0.price < $1.price }
    }

    /// 購入実行（appAccountToken付き）
    func purchase(_ product: Product, userId: String) async throws -> PurchaseResult {
        isPurchasing = true
        defer { isPurchasing = false }

        // appAccountToken生成（Critical #5: user_idから決定論的に生成）
        let appAccountToken = generateAppAccountToken(userId: userId)

        let result = try await product.purchase(options: [
            .appAccountToken(appAccountToken)
        ])

        switch result {
        case .success(let verification):
            let transaction = try checkVerified(verification)

            // Backend Proxyでサーバー側検証
            let serverResult = try await verifyWithServer(transaction: transaction)

            await transaction.finish()

            currentPlan = serverResult.plan == "pro" ? .pro : .free
            subscriptionStatus = .active(expiresAt: serverResult.expiresAt)

            return .success
        case .userCancelled:
            return .cancelled
        case .pending:
            return .pending
        @unknown default:
            return .unknown
        }
    }

    /// 購入復元
    func restorePurchases() async throws {
        try await AppStore.sync()

        // 最新のTransactionを確認
        for await result in Transaction.currentEntitlements {
            let transaction = try checkVerified(result)
            if productIds.contains(transaction.productID) {
                try await verifyWithServer(transaction: transaction)
                await transaction.finish()
                currentPlan = .pro
                return
            }
        }

        // 有効なサブスクリプションなし
        currentPlan = .free
    }
}
```

### 2. appAccountToken生成 :large_blue_circle:

03-backend-proxy.mdセクション5.2準拠。サーバーと同じロジック。

```swift
// FeatureSubscription/AppAccountToken.swift
import Foundation
import CryptoKit

/// appAccountTokenの生成（クライアント・サーバー共通ロジック）
/// user_idから決定論的にUUIDを生成
func generateAppAccountToken(userId: String) -> UUID {
    // UUIDv5: DNS namespace + userId
    let namespace = UUID(uuidString: "6ba7b810-9dad-11d1-80b4-00c04fd430c8")!
    return UUID.v5(namespace: namespace, name: userId)
}

extension UUID {
    /// UUIDv5生成（SHA1ベース）
    static func v5(namespace: UUID, name: String) -> UUID {
        var namespaceBytes = withUnsafeBytes(of: namespace.uuid) { Array($0) }
        let nameBytes = Array(name.utf8)
        let hashInput = namespaceBytes + nameBytes
        let hashData = Data(hashInput)

        // SHA1ハッシュ
        var hash = Insecure.SHA1()
        hash.update(data: hashData)
        let digest = hash.finalize()

        var uuidBytes = Array(digest.prefix(16))
        uuidBytes[6] = (uuidBytes[6] & 0x0F) | 0x50 // version 5
        uuidBytes[8] = (uuidBytes[8] & 0x3F) | 0x80 // variant

        return UUID(uuid: (
            uuidBytes[0], uuidBytes[1], uuidBytes[2], uuidBytes[3],
            uuidBytes[4], uuidBytes[5], uuidBytes[6], uuidBytes[7],
            uuidBytes[8], uuidBytes[9], uuidBytes[10], uuidBytes[11],
            uuidBytes[12], uuidBytes[13], uuidBytes[14], uuidBytes[15]
        ))
    }
}
```

### 3. Transaction検証 + サーバー連携 :large_blue_circle:

```swift
// FeatureSubscription/SubscriptionManager+Verify.swift
extension SubscriptionManager {
    /// JWSトランザクションのVerification
    func checkVerified<T>(_ result: VerificationResult<T>) throws -> T {
        switch result {
        case .unverified(_, let error):
            throw SubscriptionError.verificationFailed(error)
        case .verified(let value):
            return value
        }
    }

    /// Backend Proxyでサーバー側検証
    @discardableResult
    func verifyWithServer(transaction: Transaction) async throws -> ServerVerifyResponse {
        let transactionJWS = transaction.jwsRepresentation

        let response: ServerVerifyResponse = try await networkClient.post(
            "/api/v1/subscription/verify",
            body: ["transaction_jws": transactionJWS]
        )

        // 新しいJWTトークンを保存（plan=proに更新済み）
        if let newToken = response.accessToken {
            try SecureKeychain.save(
                key: .accessToken,
                data: Data(newToken.utf8)
            )
        }

        return response
    }

    /// Transaction.updates の監視（バックグラウンド更新対応）
    func listenForTransactionUpdates() async {
        for await result in Transaction.updates {
            guard let transaction = try? checkVerified(result) else { continue }

            if productIds.contains(transaction.productID) {
                do {
                    try await verifyWithServer(transaction: transaction)
                    await transaction.finish()

                    // UIの更新
                    await MainActor.run {
                        if transaction.revocationDate != nil {
                            currentPlan = .free
                            subscriptionStatus = .expired
                        } else if let expiresDate = transaction.expirationDate,
                                  expiresDate > Date() {
                            currentPlan = .pro
                            subscriptionStatus = .active(expiresAt: expiresDate)
                        }
                    }
                } catch {
                    // サーバー検証失敗時はローカル状態を維持
                    await transaction.finish()
                }
            }
        }
    }
}
```

### 4. Pro/Free判定ロジック :large_blue_circle:

```swift
// Domain/Models/SubscriptionTypes.swift
enum SubscriptionPlan: String, Codable {
    case free
    case pro
}

enum SubscriptionStatus: Equatable {
    case none
    case active(expiresAt: Date?)
    case expired
    case gracePeriod(expiresAt: Date?)
    case billingRetry
}

struct ServerVerifyResponse: Codable {
    let accessToken: String?
    let subscription: SubscriptionInfo?

    struct SubscriptionInfo: Codable {
        let status: String
        let productId: String
        let expiresAt: String?
    }

    var plan: String {
        subscription?.status == "active" ? "pro" : "free"
    }

    var expiresAt: Date? {
        guard let dateStr = subscription?.expiresAt else { return nil }
        return ISO8601DateFormatter().date(from: dateStr)
    }
}
```

## テスト要件

### 正常系
- loadProducts()が月額/年額の2つのProductを取得する
- purchase()がappAccountToken付きで購入を実行する
- appAccountTokenがuser_idから決定論的に生成される（サーバーと一致）
- 購入成功後にBackend Proxy verifyが呼ばれJWTが更新される
- Transaction.finish()が呼ばれる
- restorePurchases()が有効なサブスクリプションを復元する
- Transaction.updates監視がバックグラウンドで動作する

### appAccountToken（Critical #5）
- 同一user_idから常に同じappAccountTokenが生成される
- 異なるuser_idからは異なるappAccountTokenが生成される
- UUIDv5フォーマットに準拠している

### 解約・降格
- 解約後にcurrentPlan=freeに変更される
- 解約後にAI処理結果（AISummary, EmotionAnalysis, Tag）が削除されない
- 猶予期間（gracePeriod）中はPro機能が維持される

### 異常系
- ユーザーキャンセル時にPurchaseResult.cancelledが返る
- StoreKit検証失敗時にSubscriptionError.verificationFailedが発生する
- サーバー検証失敗時のエラーハンドリング（ローカル状態維持）
- ネットワーク不通時の購入フローのエラーハンドリング

### UI
- 現在のプラン（Free/Pro）が正しく表示される
- プラン比較表が表示される
- 購入中のローディング状態が表示される
- 復元ボタンが機能する
