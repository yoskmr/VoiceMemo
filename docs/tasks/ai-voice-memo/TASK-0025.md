### 2. OnDeviceLLMProvider実装 :large_blue_circle:

02-ai-pipeline.mdセクション3.2.3準拠。

```swift
// InfraLLM/OnDevice/OnDeviceLLMProvider.swift
import Foundation

final class OnDeviceLLMProvider: LLMProviderProtocol {
    let providerType: LLMProviderType = .onDeviceLlamaCpp
    let isOnDevice: Bool = true
    let supportedTasks: Set<LLMTask> = [.summarize, .tagging]
    let maxInputTokens: Int = 650  // ~500日本語文字（日本語: 1文字≒1.3トークン）

    /// トークン見積もり: 日本語1文字 ≒ 1.3トークン（GPT-4oトークナイザ基準）
    static let japaneseTokenRatio: Double = 1.3
    static let maxInputCharacters: Int = 500

    private var llamaContext: LlamaContext?
    private let modelPath: URL
    private let modelManager: LLMModelManager

    init(modelManager: LLMModelManager) {
        self.modelManager = modelManager
        self.modelPath = modelManager.modelFilePath(for: .phi3MiniQ4KM)
    }

    func isAvailable() async -> Bool {
        DeviceCapabilityChecker.shared.supportsOnDeviceLLM && llamaContext != nil
    }

    /// モデルの初期化（llama.cppコンテキスト作成）
    func initialize() async throws {
        guard DeviceCapabilityChecker.shared.supportsOnDeviceLLM else {
            throw LLMError.deviceNotSupported
        }
        guard DeviceCapabilityChecker.shared.hasMemoryHeadroomForLLM else {
            throw LLMError.insufficientMemory
        }

        let params = LlamaModelParams.default()
        params.n_gpu_layers = 99  // Metal GPU全レイヤーオフロード

        llamaContext = try LlamaContext(
            modelPath: modelPath.path,
            params: params
        )
    }

    func process(_ request: LLMRequest) async throws -> LLMResponse {
        guard let llamaContext else { throw LLMError.deviceNotSupported }
        guard request.text.count <= Self.maxInputCharacters else {
            throw LLMError.inputTooLong
        }

        // 要求タスクがsupportedTasksに含まれることを検証
        let effectiveTasks = request.tasks.intersection(supportedTasks)
        guard !effectiveTasks.isEmpty else { throw LLMError.unsupportedTask }

        let prompt = buildOnDevicePrompt(text: request.text, tasks: effectiveTasks)
        let startTime = CFAbsoluteTimeGetCurrent()

        let response = try await llamaContext.generate(
            prompt: prompt,
            maxTokens: 512,
            temperature: 0.3,
            stopTokens: ["```", "</json>"]
        )
        let elapsed = CFAbsoluteTimeGetCurrent() - startTime

        return parseOnDeviceResponse(response, processingTimeMs: Int(elapsed * 1000))
    }

    /// モデルリソースの明示的解放（メモリ排他制御用）
    func unloadModel() async {
        llamaContext = nil
    }
}
```

### 3. デバイス能力チェック :large_blue_circle:

02-ai-pipeline.mdセクション3.2.1準拠。A16+ / 6GB RAM要件。

```swift
// SharedUtil/Device/DeviceCapabilityChecker.swift
final class DeviceCapabilityChecker {
    static let shared = DeviceCapabilityChecker()

    /// REQ-021: A16 Bionic以降かつメモリ6GB以上
    var supportsOnDeviceLLM: Bool {
        let chipGeneration = getChipGeneration()
        let totalMemoryGB = ProcessInfo.processInfo.physicalMemory / (1024 * 1024 * 1024)
        return chipGeneration >= 16 && totalMemoryGB >= 6
    }

    /// STT実行中のメモリ余裕チェック
    var hasMemoryHeadroomForLLM: Bool {
        let availableMemory = os_proc_available_memory()
        return availableMemory > 2 * 1024 * 1024 * 1024  // 2GB以上の空きが必要
    }

    private func getChipGeneration() -> Int {
        var systemInfo = utsname()
        uname(&systemInfo)
        let machine = withUnsafePointer(to: &systemInfo.machine) {
            $0.withMemoryRebound(to: CChar.self, capacity: 1) {
                String(cString: $0)
            }
        }
        // iPhone15,x = A16, iPhone16,x = A17 Pro, iPhone17,x = A18 ...
        return parseChipGeneration(from: machine)
    }

    private func parseChipGeneration(from machine: String) -> Int {
        // iPhone15,1-4 → A16 (generation 16)
        // iPhone16,1-2 → A17 Pro (generation 17)
        // iPhone17,x → A18 (generation 18)
        guard machine.hasPrefix("iPhone") else { return 0 }
        let versionPart = machine.replacingOccurrences(of: "iPhone", with: "")
        guard let majorVersion = Int(versionPart.split(separator: ",").first ?? "") else { return 0 }
        // iPhone major version - modelOffset ≒ chip generation
        // iPhone15 = A16, iPhone16 = A17, ...
        return majorVersion - 15 + 16
    }
}
```

### 4. メモリ排他制御 :large_blue_circle:

02-ai-pipeline.mdセクション3.2.3準拠。STTモデルとLLMモデルの同時ロードを回避。

```swift
// InfraLLM/OnDevice/MemoryExclusionController.swift

/// STTモデルとLLMモデルのメモリ排他制御
/// - STTモデル (~600MB) + LLMモデル (~2.5GB) = ~3.1GB
/// - iPhone 15 (6GB) ではアプリ全体で約4GBまで使用可能
/// - 同時ロードするとメモリ不足のリスクがあるため排他制御
actor MemoryExclusionController {
    static let shared = MemoryExclusionController()

    private var activeModel: ActiveModel = .none

    enum ActiveModel {
        case none
        case stt
        case llm
    }

    /// LLMモデルのロード（STTモデルをアンロード後）
    func acquireLLM(
        sttEngine: STTEngineProtocol?,
        llmProvider: OnDeviceLLMProvider
    ) async throws {
        // 1. STTモデルのアンロード
        if let whisperEngine = sttEngine as? WhisperKitEngine {
            await whisperEngine.unloadModel()
        }
        activeModel = .llm

        // 2. LLMモデルのロード
        try await llmProvider.initialize()
    }

    /// LLM処理完了後のリソース解放 + STT再ロード
    func releaseLLM(
        llmProvider: OnDeviceLLMProvider,
        sttEngine: STTEngineProtocol?
    ) async {
        // 1. LLMモデルのアンロード
        await llmProvider.unloadModel()
        activeModel = .none

        // 2. STTモデルの再ロード（次回録音に備える）
        if let whisperEngine = sttEngine as? WhisperKitEngine {
            try? await whisperEngine.initialize(model: .small)
        }
        activeModel = .stt
    }
}
```

### 5. モデルダウンロード・管理 :large_blue_circle:

02-ai-pipeline.mdセクション3.2.1準拠。Library/Caches/Models/ に配置。

```swift
// InfraLLM/OnDevice/LLMModelManager.swift
enum LLMModel: String {
    case phi3MiniQ4KM = "phi-3-mini-q4_k_m"

    var fileSize: Int64 { 2_000_000_000 }  // 約2GB
    var downloadURL: URL {
        URL(string: "https://huggingface.co/microsoft/Phi-3-mini-4k-instruct-gguf/resolve/main/Phi-3-mini-4k-instruct-q4.gguf")!
    }
}

final class LLMModelManager {
    private let modelsDirectory: URL

    init() {
        let caches = FileManager.default.urls(for: .cachesDirectory, in: .userDomainMask).first!
        modelsDirectory = caches.appendingPathComponent("Models", isDirectory: true)

        try? FileManager.default.createDirectory(
            at: modelsDirectory,
            withIntermediateDirectories: true
        )
    }

    func modelFilePath(for model: LLMModel) -> URL {
        modelsDirectory.appendingPathComponent("\(model.rawValue).gguf")
    }

    func isModelDownloaded(_ model: LLMModel) -> Bool {
        FileManager.default.fileExists(atPath: modelFilePath(for: model).path)
    }

    /// モデルダウンロード（バックグラウンド対応）
    func downloadModel(_ model: LLMModel, progress: @escaping (Double) -> Void) async throws {
        guard !isModelDownloaded(model) else { return }

        let (tempURL, _) = try await URLSession.shared.download(from: model.downloadURL)
        try FileManager.default.moveItem(at: tempURL, to: modelFilePath(for: model))
    }
}
```

// InfraLLM/OnDevice/OnDevicePromptBuilder.swift
func buildOnDevicePrompt(text: String, tasks: Set<LLMTask>) -> String {
    var instructions: [String] = []

    if tasks.contains(.summarize) {
        instructions.append("""
        - "summary": {"title": "20文字以内のタイトル", "brief": "1-2行の要約", "key_points": ["ポイント1", "ポイント2"]}
        """)
    }

    if tasks.contains(.tagging) {
        instructions.append("""
        - "tags": [{"label": "タグ名", "confidence": 0.9}]（最大3つ、15文字以内）
        """)
    }

    return """
    以下のテキストを分析し、JSON形式で結果を返してください。
    \(instructions.joined(separator: "\n"))

    テキスト:
    \(text)

    JSON:
    """
}

- process()が要約+タグのJSON結果を返す

### メモリ排他制御
- MemoryExclusionControllerがSTTアンロード → LLMロードの順序を保証する

- 同時ロード要求が排他制御される

### 異常系
- 非対応デバイスでLLMError.deviceNotSupportedが発生する
- メモリ不足時にLLMError.insufficientMemoryが発生する
- 500文字超のテキストでLLMError.inputTooLongが発生する
- 感情分析タスク（.sentimentAnalysis）のみの要求でLLMError.unsupportedTaskが発生する
- モデル未ダウンロード状態でのエラーハンドリング

### デバイス能力チェック
- iPhone 15（A16, 6GB）でsupportsOnDeviceLLM=true
- iPhone 14（A15, 6GB）でsupportsOnDeviceLLM=false（A16未満）
- iPhone SE 3（A15, 4GB）でsupportsOnDeviceLLM=false
