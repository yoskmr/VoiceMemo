```swift
import SwiftUI

struct TagChipView: View {
    let tag: LLMTagResult
    var onTap: (() -> Void)?

    var body: some View {
        Button(action: { onTap?() }) {
            HStack(spacing: 4) {
                Image(systemName: "tag.fill")
                    .font(.caption2)
                Text(tag.label)
                    .font(.caption)
                    .fontWeight(.medium)
            }
            .padding(.horizontal, 10)
            .padding(.vertical, 5)
            .background(Color.accentColor.opacity(confidenceOpacity))
            .foregroundColor(.accentColor)
            .cornerRadius(16)
        }
        .buttonStyle(.plain)
    }

    private var confidenceOpacity: Double {
        0.1 + (tag.confidence * 0.15) // 0.1〜0.25
    }
}

/// タグチップの横並びフローレイアウト
struct TagFlowLayout: View {
    let tags: [LLMTagResult]
    var onTagTap: ((LLMTagResult) -> Void)?

    var body: some View {
        FlowLayout(spacing: 8) {
            ForEach(tags, id: \.label) { tag in
                TagChipView(tag: tag) {
                    onTagTap?(tag)
                }
            }
        }
    }
}
```

```swift
// SharedUI/Components/EmotionBadgeView.swift
import SwiftUI

struct EmotionBadgeView: View {
    let emotion: EmotionCategory
    let score: Double
    var size: BadgeSize = .medium

    enum BadgeSize {
        case small, medium, large
        var fontSize: Font { switch self { case .small: .caption2; case .medium: .caption; case .large: .subheadline } }
        var iconSize: CGFloat { switch self { case .small: 12; case .medium: 16; case .large: 20 } }
        var padding: CGFloat { switch self { case .small: 4; case .medium: 6; case .large: 8 } }
    }

    var body: some View {
        HStack(spacing: 4) {
            Image(systemName: emotion.iconName)
                .font(.system(size: size.iconSize))
            Text(emotion.localizedLabel)
                .font(size.fontSize)
                .fontWeight(.medium)
            if score > 0 {
                Text("\(Int(score * 100))%")
                    .font(size.fontSize)
                    .foregroundColor(.secondary)
            }
        }
        .padding(.horizontal, size.padding + 4)
        .padding(.vertical, size.padding)
        .background(emotion.color.opacity(0.2))
        .foregroundColor(emotion.color)
        .cornerRadius(size.padding + 4)
    }
}
```

### 5. 感情分析結果ビュー :large_blue_circle:

```swift
// SharedUI/Components/SentimentResultView.swift
import SwiftUI

struct SentimentResultView: View {
    let sentiment: LLMSentimentResult

    var body: some View {
        VStack(alignment: .leading, spacing: 12) {
            // ヘッダー + 主要感情バッジ
            HStack {
                Image(systemName: "brain.head.profile")
                    .foregroundColor(.accentColor)
                Text("感情分析")
                    .font(.headline)
                Spacer()
                EmotionBadgeView(
                    emotion: sentiment.primary,
                    score: sentiment.scores[sentiment.primary] ?? 0,
                    size: .medium
                )
            }

            // 感情スコアバー（上位4カテゴリ）
            let sortedScores = sentiment.scores
                .sorted { $0.value > $1.value }
                .prefix(4)

            ForEach(Array(sortedScores), id: \.key) { category, score in
                EmotionScoreBar(emotion: category, score: score)
            }

            // 根拠テキスト
            if !sentiment.evidence.isEmpty {
                VStack(alignment: .leading, spacing: 6) {
                    Text("根拠")
                        .font(.subheadline)
                        .fontWeight(.medium)
                        .foregroundColor(.secondary)

                    ForEach(sentiment.evidence, id: \.text) { evidence in
                        HStack(alignment: .top, spacing: 8) {
                            EmotionBadgeView(
                                emotion: evidence.emotion,
                                score: 0,
                                size: .small
                            )
                            Text("「\(evidence.text)」")
                                .font(.caption)
                                .foregroundColor(.secondary)
                        }
                    }
                }
            }
        }
        .padding()
        .background(Color(.systemBackground))
        .cornerRadius(12)
        .shadow(color: .black.opacity(0.05), radius: 4, y: 2)
    }
}

struct EmotionScoreBar: View {
    let emotion: EmotionCategory
    let score: Double

    var body: some View {
        HStack(spacing: 8) {
            Text(emotion.localizedLabel)
                .font(.caption)
                .frame(width: 40, alignment: .leading)
            GeometryReader { geometry in
                ZStack(alignment: .leading) {
                    RoundedRectangle(cornerRadius: 4)
                        .fill(Color.gray.opacity(0.1))
                    RoundedRectangle(cornerRadius: 4)
                        .fill(emotion.color)
                        .frame(width: geometry.size.width * score)
                }
            }
            .frame(height: 8)
            Text("\(Int(score * 100))%")
                .font(.caption2)
                .foregroundColor(.secondary)
                .frame(width: 32, alignment: .trailing)
        }
    }
}
```

### 6. スケルトンローディング :large_blue_circle:

```swift
// SharedUI/Components/AIProcessingSkeletonView.swift
import SwiftUI

struct AIProcessingSkeletonView: View {
    @State private var isAnimating = false

    var body: some View {
        VStack(alignment: .leading, spacing: 12) {
            HStack {
                Image(systemName: "sparkles")
                    .foregroundColor(.accentColor)
                Text("AI分析中...")
                    .font(.headline)
                Spacer()
                ProgressView()
            }

            // スケルトンライン
            ForEach(0..<3, id: \.self) { _ in
                RoundedRectangle(cornerRadius: 4)
                    .fill(Color.gray.opacity(0.15))
                    .frame(height: 16)
                    .shimmer(isAnimating: isAnimating)
            }

            // タグスケルトン
            HStack(spacing: 8) {
                ForEach(0..<3, id: \.self) { _ in
                    RoundedRectangle(cornerRadius: 12)
                        .fill(Color.gray.opacity(0.15))
                        .frame(width: 60, height: 24)
                        .shimmer(isAnimating: isAnimating)
                }
            }

            // 感情バッジスケルトン
            RoundedRectangle(cornerRadius: 8)
                .fill(Color.gray.opacity(0.15))
                .frame(width: 100, height: 28)
                .shimmer(isAnimating: isAnimating)
        }
        .padding()
        .background(Color(.systemBackground))
        .cornerRadius(12)
        .shadow(color: .black.opacity(0.05), radius: 4, y: 2)
        .onAppear { isAnimating = true }
    }
}

extension View {
    func shimmer(isAnimating: Bool) -> some View {
        self.overlay(
            GeometryReader { geometry in
                LinearGradient(
                    colors: [.clear, .white.opacity(0.3), .clear],
                    startPoint: .leading,
                    endPoint: .trailing
                )
                .frame(width: geometry.size.width * 2)
                .offset(x: isAnimating ? geometry.size.width : -geometry.size.width)
                .animation(
                    .linear(duration: 1.5).repeatForever(autoreverses: false),
                    value: isAnimating
                )
            }
            .clipped()
        )
    }
}
```

### 7. TCA Reducer統合 :large_blue_circle:

```swift
// FeatureMemo/MemoDetailReducer.swift (AI処理部分)
extension MemoDetailReducer {
    @ObservableState
    struct AIState: Equatable {
        var processingStatus: AIProcessingStatus = .idle
        var summary: LLMSummaryResult?
        var tags: [LLMTagResult] = []
        var sentiment: LLMSentimentResult?
        var error: String?
    }

    enum AIProcessingStatus: Equatable {
        case idle
        case processing
        case completed
        case failed
    }

    enum AIAction {
        case startAIProcessing
        case aiProcessingCompleted(LLMResponse)
        case aiProcessingFailed(String)
        case retryAIProcessing
        case tagTapped(LLMTagResult)
    }
}
```
