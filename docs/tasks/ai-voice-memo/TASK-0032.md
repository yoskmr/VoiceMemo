### 2. タイポグラフィトークン :large_blue_circle:

04-ui-design-system.md セクション3準拠。全Font.custom定義にrelativeTo:パラメータを追加しDynamic Type対応。

```swift
// SharedUI/DesignSystem/VMDesignTokens+Typography.swift

import SwiftUI

extension Font {
    // MARK: - 本文・日記テキスト用（Hiragino Maru Gothic ProN）
    // relativeTo: を指定しDynamic Typeスケーリングに対応
    static func vmBody(_ size: CGFloat = 17) -> Font {
        .custom("HiraMaruProN-W4", size: size, relativeTo: .body)
    }

    static func vmBodyBold(_ size: CGFloat = 17) -> Font {
        .custom("HiraMaruProN-W4", size: size, relativeTo: .body).bold()
    }

    // MARK: - UI要素用（SF Pro Rounded）
    // system fontは自動的にDynamic Typeに対応
    static func vmUI(_ size: CGFloat = 17, weight: Font.Weight = .regular) -> Font {
        .system(size: size, weight: weight, design: .rounded)
    }

    // MARK: - タイマー表示用（SF Mono Rounded、固定サイズ）
    static func vmTimer(_ size: CGFloat = 48) -> Font {
        .system(size: size, weight: .light, design: .monospaced)
    }

    // MARK: - Dynamic Type対応カスタムフォント生成ヘルパー
    static func vmCustomScaled(
        _ name: String = "HiraMaruProN-W4",
        size: CGFloat,
        relativeTo textStyle: Font.TextStyle = .body
    ) -> Font {
        .custom(name, size: size, relativeTo: textStyle)
    }

    // MARK: - プリセット（04-ui-design-system.md セクション3.2準拠）
    static let vmLargeTitle = Font.vmUI(34, weight: .bold)       // 画面タイトル
    static let vmTitle1 = Font.vmUI(28, weight: .bold)           // セクションヘッダー
    static let vmTitle2 = Font.vmUI(22, weight: .bold)           // カードタイトル
    static let vmTitle3 = Font.vmUI(20, weight: .semibold)       // サブセクション
    static let vmHeadline = Font.vmUI(17, weight: .semibold)     // 強調テキスト
    static let vmCallout = Font.custom("HiraMaruProN-W4", size: 16, relativeTo: .callout)
    static let vmSubheadline = Font.custom("HiraMaruProN-W4", size: 15, relativeTo: .subheadline)
    static let vmFootnote = Font.custom("HiraMaruProN-W4", size: 13, relativeTo: .footnote)
    static let vmCaption1 = Font.vmUI(12, weight: .regular)      // ラベル、バッジ
    static let vmCaption2 = Font.vmUI(11, weight: .regular)      // 最小テキスト
}
```

### 3. スペーシングトークン :large_blue_circle:

04-ui-design-system.md セクション10準拠。4ptグリッドベース。

```swift
// SharedUI/DesignSystem/VMDesignTokens+Spacing.swift

import SwiftUI

enum VMSpacing {
    /// 2pt - アイコンとテキスト間（タイト）
    static let xxs: CGFloat = 2
    /// 4pt - バッジ内パディング
    static let xs: CGFloat = 4
    /// 8pt - カード内要素間
    static let sm: CGFloat = 8
    /// 12pt - カード間マージン
    static let md: CGFloat = 12
    /// 16pt - カード内パディング、セクション内
    static let lg: CGFloat = 16
    /// 24pt - セクション間
    static let xl: CGFloat = 24
    /// 32pt - 画面内主要ブロック間
    static let xxl: CGFloat = 32
    /// 48pt - 画面上部余白
    static let xxxl: CGFloat = 48
}

// MARK: - 角丸基準値（NFR-012: 12pt以上）
enum VMCornerRadius {
    /// 12pt - タグチップ、感情バッジ、テキスト入力フィールド、カレンダーセル
    static let small: CGFloat = 12
    /// 16pt - ボタン（大）、カード、検索バー
    static let medium: CGFloat = 16
    /// 20pt - シート（モーダル）上部
    static let large: CGFloat = 20
    /// 40pt - 録音ボタン（完全な円）
    static let circle: CGFloat = 40
}

// MARK: - シャドウ基準値
enum VMShadow {
    struct Definition {
        let color: Color
        let radius: CGFloat
        let y: CGFloat
    }

    /// Level 1: タグチップ、バッジ
    static let level1 = Definition(color: .black.opacity(0.03), radius: 4, y: 1)
    /// Level 2: カード
    static let level2 = Definition(color: .black.opacity(0.05), radius: 8, y: 2)
    /// Level 3: フローティングボタン
    static let level3 = Definition(color: .black.opacity(0.08), radius: 16, y: 4)
    /// Level 4: モーダルシート
    static let level4 = Definition(color: .black.opacity(0.12), radius: 24, y: 8)
}

// MARK: - Shadow View Extension
extension View {
    func vmShadow(_ level: VMShadow.Definition) -> some View {
        self.shadow(color: level.color, radius: level.radius, y: level.y)
    }
}
```
