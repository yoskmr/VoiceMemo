```typescript
// src/routes/ai.ts
import { Hono } from 'hono';
import { z } from 'zod';
import { zValidator } from '@hono/zod-validator';
import type { Env } from '../types/env';

// リクエストスキーマ（03-backend-proxy.md JSON Schema準拠）
const aiProcessSchema = z.object({
  text: z.string().min(1).max(30000),
  language: z.enum(['ja', 'en']).default('ja'),
  options: z.object({
    summary: z.boolean().default(true),
    tags: z.boolean().default(true),
    sentiment: z.boolean().default(true),
  }).default({}),
  context: z.object({
    memo_local_id: z.string().optional(),
    recording_duration_sec: z.number().int().min(0).optional(),
  }).optional(),
});

const ai = new Hono<{ Bindings: Env }>();

ai.post('/process',
  zValidator('json', aiProcessSchema),
  async (c) => {
    const body = c.req.valid('json');
    const user = c.get('user');
    const requestId = c.req.header('X-Request-ID') || crypto.randomUUID();
    const startTime = Date.now();

    try {
      // OpenAI GPT-4o mini 呼び出し
      const aiResult = await callOpenAI(body, c.env);

      const processingTimeMs = Date.now() - startTime;

      return c.json({
        summary: aiResult.summary,
        tags: aiResult.tags,
        sentiment: aiResult.sentiment,
        usage: {
          used: 0,   // TASK-0023で実装
          limit: user.plan === 'free' ? 5 : null,
          plan: user.plan,
          resets_at: null,
        },
        metadata: {
          model: 'gpt-4o-mini',
          provider: 'cloud_gpt4o_mini',
          processing_time_ms: processingTimeMs,
          request_id: requestId,
        },
      });
    } catch (error) {
      // テキストデータのメモリからの即時解放を保証
      // (JavaScript GCに委ねるが、参照を明示的にnull化)
      if (error instanceof OpenAIError) {
        return c.json({
          error: { code: 'UPSTREAM_ERROR', message: 'AI処理に失敗しました', request_id: requestId }
        }, 502);
      }
      throw error;
    }
  }
);

export { ai };
```

### 2. OpenAI API呼び出し :large_blue_circle:

02-ai-pipeline.mdセクション3.3.2, 6準拠。統合プロンプトで要約+タグ+感情分析を1回で処理。

```typescript
// src/services/openai.ts
interface OpenAIRequest {
  text: string;
  language: string;
  options: { summary: boolean; tags: boolean; sentiment: boolean };
}

interface AIProcessResult {
  summary: {
    title: string;          // 20文字以内
    brief: string;          // 1-2行の要約
    key_points: string[];   // 最大5つ
  } | null;
  tags: Array<{
    label: string;          // 15文字以内
    confidence: number;     // 0.0-1.0
  }> | null;
  sentiment: {
    primary: EmotionCategory;
    scores: Record<EmotionCategory, number>;  // 合計1.0
    evidence: Array<{ text: string; emotion: EmotionCategory }>;  // 最大3件
  } | null;
}

type EmotionCategory = 'joy' | 'calm' | 'anticipation' | 'sadness'
                     | 'anxiety' | 'anger' | 'surprise' | 'neutral';

async function callOpenAI(request: OpenAIRequest, env: Env): Promise<AIProcessResult> {
  const systemPrompt = buildSystemPrompt(request.language, request.options);
  const userPrompt = `以下の音声メモのテキストを分析してください:\n\n${request.text}`;

  const response = await fetch('https://api.openai.com/v1/chat/completions', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${env.OPENAI_API_KEY}`,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      model: 'gpt-4o-mini',
      messages: [
        { role: 'system', content: systemPrompt },
        { role: 'user', content: userPrompt },
      ],
      response_format: { type: 'json_object' },
      temperature: 0.3,
      max_tokens: 1024,
    }),
  });

  if (!response.ok) {
    throw new OpenAIError(`OpenAI API error: ${response.status}`);
  }

  const data = await response.json();
  const content = JSON.parse(data.choices[0].message.content);

  return validateAndNormalizeResponse(content, request.options);
}
```

```typescript
// src/services/prompt.ts
function buildSystemPrompt(language: string, options: { summary: boolean; tags: boolean; sentiment: boolean }): string {
  const tasks: string[] = [];

  if (options.summary) {
    tasks.push(`## 要約
- "title": メモの内容を表すタイトル（20文字以内）
- "brief": 1-2行の要約文（200文字以内）
- "key_points": 重要ポイント（最大5つ、各ポイントは1文）`);
  }

  if (options.tags) {
    tasks.push(`## タグ
- "tags": 自動付与タグの配列（最大5つ）
  - "label": タグラベル（15文字以内、日本語）
  - "confidence": 信頼度（0.0-1.0）`);
  }

  if (options.sentiment) {
    tasks.push(`## 感情分析
- "sentiment":
  - "primary": 最も支配的な感情（joy/calm/anticipation/sadness/anxiety/anger/surprise/neutral）
  - "scores": 8カテゴリの感情スコア（合計1.0）
    - joy(喜び), calm(安心), anticipation(期待), sadness(悲しみ), anxiety(不安), anger(怒り), surprise(驚き), neutral(中立)
  - "evidence": 感情判定の根拠テキスト箇所（最大3件）
    - "text": 根拠テキスト, "emotion": 対応する感情カテゴリ`);
  }

  return `あなたは音声メモの分析アシスタントです。${language === 'ja' ? '日本語' : '英語'}で応答してください。

以下のJSON形式で結果を返してください:

${tasks.join('\n\n')}

必ず有効なJSONで応答してください。指定されていないフィールドはnullとしてください。`;
}
```

### 4. レスポンスバリデーション :large_blue_circle:

03-backend-proxy.mdレスポンスJSON Schema準拠。

```typescript
// src/services/validator.ts
const emotionCategories = ['joy', 'calm', 'anticipation', 'sadness',
                           'anxiety', 'anger', 'surprise', 'neutral'] as const;

function validateAndNormalizeResponse(
  raw: unknown,
  options: { summary: boolean; tags: boolean; sentiment: boolean }
): AIProcessResult {
  const result: AIProcessResult = { summary: null, tags: null, sentiment: null };

  if (options.summary && raw.summary) {
    result.summary = {
      title: String(raw.summary.title || '').slice(0, 20),
      brief: String(raw.summary.brief || '').slice(0, 200),
      key_points: Array.isArray(raw.summary.key_points)
        ? raw.summary.key_points.slice(0, 5).map(String)
        : [],
    };
  }

  if (options.tags && Array.isArray(raw.tags)) {
    result.tags = raw.tags.slice(0, 10).map((t: any) => ({
      label: String(t.label || '').slice(0, 15),
      confidence: Math.max(0, Math.min(1, Number(t.confidence) || 0)),
    }));
  }

  if (options.sentiment && raw.sentiment) {
    const scores = normalizeEmotionScores(raw.sentiment.scores);
    result.sentiment = {
      primary: validateEmotionCategory(raw.sentiment.primary),
      scores,
      evidence: Array.isArray(raw.sentiment.evidence)
        ? raw.sentiment.evidence.slice(0, 3).map((e: any) => ({
            text: String(e.text || ''),
            emotion: validateEmotionCategory(e.emotion),
          }))
        : [],
    };
  }

  return result;
}

function normalizeEmotionScores(scores: any): Record<EmotionCategory, number> {
  const normalized: Record<string, number> = {};
  let total = 0;

  for (const category of emotionCategories) {
    const val = Math.max(0, Number(scores?.[category]) || 0);
    normalized[category] = val;
    total += val;
  }

  // 合計が1.0になるよう正規化
  if (total > 0) {
    for (const category of emotionCategories) {
      normalized[category] = Math.round((normalized[category] / total) * 100) / 100;
    }
  } else {
    normalized['neutral'] = 1.0;
  }

  return normalized as Record<EmotionCategory, number>;
}
```

```typescript
// src/utils/errors.ts
class AppError extends Error {
  constructor(
    public code: string,
    message: string,
    public statusCode: number,
    public details?: Record<string, unknown>
  ) {
    super(message);
  }
}

class OpenAIError extends AppError {
  constructor(message: string) {
    super('UPSTREAM_ERROR', message, 502);
  }
}

// エラーコード一覧（03-backend-proxy.md準拠）
// 400: INVALID_REQUEST - リクエストパラメータ不正
// 401: UNAUTHORIZED - 認証トークン無効
// 429: RATE_LIMITED / USAGE_LIMIT_EXCEEDED - 制限超過
// 500: INTERNAL_ERROR - サーバー内部エラー
// 502: UPSTREAM_ERROR - OpenAI APIエラー
```

## テスト要件

### 正常系
- 要約+タグ+感情分析の統合処理が正常に完了する
- 要約のみ（options.tags=false, options.sentiment=false）が正常に処理される
- 日本語テキスト / 英語テキストの両方が処理可能
- レスポンスがJSON Schema準拠（summary.title <= 20文字、tags.maxItems=10、sentimentの8カテゴリ）
- 感情スコアの合計が1.0に正規化される
- metadataにprocessing_time_ms, model, provider, request_idが含まれる

### 異常系
- text未指定時に400 INVALID_REQUESTが返る
- textが30000文字超の場合に400 INVALID_REQUESTが返る
- 不正なlanguage値（"zh"等）で400 INVALID_REQUESTが返る
- OpenAI API障害時に502 UPSTREAM_ERRORが返る
- OpenAI APIレスポンスが不正JSONの場合のフォールバック処理
- 認証なしリクエストに401 UNAUTHORIZEDが返る

### プライバシー検証
- リクエストのテキストデータがログに記録されないこと
- テキストデータがD1/KVに永続化されないこと
