### 1. Challenge発行エンドポイント :large_blue_circle:

03-backend-proxy.mdセクション3.4.5a準拠。

```typescript
// src/routes/auth.ts
import { Hono } from 'hono';
import type { Env } from '../types/env';

const auth = new Hono<{ Bindings: Env }>();

// GET /api/v1/auth/challenge - App Attest用チャレンジ発行
auth.get('/challenge', async (c) => {
  // 32バイトのランダムnonce生成
  const challengeBytes = new Uint8Array(32);
  crypto.getRandomValues(challengeBytes);
  const challenge = btoa(String.fromCharCode(...challengeBytes))
    .replace(/\+/g, '-').replace(/\//g, '_').replace(/=+$/, '');

  // KVに保存（TTL 300秒）
  await c.env.CHALLENGE_KV.put(
    `challenge:${challenge}`,
    JSON.stringify({ created_at: new Date().toISOString() }),
    { expirationTtl: 300 }
  );

  return c.json({
    challenge,
    expires_in: 300,
  });
});
```

// POST /api/v1/auth/device - デバイストークン認証 + App Attest attestation
const deviceAuthSchema = z.object({
  device_id: z.string().uuid(),
  device_info: z.object({
    model: z.string(),
    os_version: z.string(),
    app_version: z.string(),
  }),
  attestation: z.string().optional(),       // Base64 CBOR attestation object
  key_id: z.string().optional(),            // App Attest keyId
  challenge: z.string().optional(),         // サーバー発行のchallenge
  attest_supported: z.boolean().default(true),
});

auth.post('/device',
  zValidator('json', deviceAuthSchema),
  async (c) => {
    const body = c.req.valid('json');

    if (body.attest_supported && body.attestation && body.key_id && body.challenge) {
      // App Attest対応端末: attestation検証
      return handleAttestDevice(c, body);
    } else {
      // App Attest非対応端末: 制限付き認証
      return handleFallbackDevice(c, body);
    }
  }
);

async function handleAttestDevice(c: Context, body: DeviceAuthBody) {
  // 1. challenge有効性検証（使い捨てnonce）
  const challengeKey = `challenge:${body.challenge}`;
  const challengeData = await c.env.CHALLENGE_KV.get(challengeKey);
  if (!challengeData) {
    return c.json({
      error: { code: 'INVALID_REQUEST', message: 'Challenge が無効または期限切れです' }
    }, 400);
  }
  await c.env.CHALLENGE_KV.delete(challengeKey); // 使い捨て

  // 2. Attestation検証（Apple App Attest）
  const { publicKey, receipt } = await verifyAttestation({
    keyId: body.key_id!,
    attestation: body.attestation!,
    challenge: body.challenge!,
  }, body.device_id, c.env);

  // 3. ユーザー作成 / 取得
  const userId = generateId('usr');
  const now = new Date().toISOString();

  await c.env.DB.batch([
    c.env.DB.prepare(
      `INSERT OR IGNORE INTO users (id, auth_method, plan, created_at, updated_at)
       VALUES (?, 'device', 'free', ?, ?)`
    ).bind(userId, now, now),
    c.env.DB.prepare(
      `INSERT INTO user_devices
       (id, user_id, device_id, device_model, os_version, app_version,
        attest_verified, attest_key_id, attest_public_key, attest_receipt, attest_counter,
        created_at, updated_at)
       VALUES (?, ?, ?, ?, ?, ?, 1, ?, ?, ?, 0, ?, ?)`
    ).bind(
      generateId('dev'), userId, body.device_id,
      body.device_info.model, body.device_info.os_version, body.device_info.app_version,
      body.key_id, exportPublicKey(publicKey), encodeReceipt(receipt),
      now, now
    ),
  ]);

  // 4. JWT生成
  const accessToken = await generateJWT({
    sub: userId,
    plan: 'free',
    auth_method: 'device',
    device_id: body.device_id,
  }, c.env);

  return c.json({
    access_token: accessToken,
    expires_in: 86400,
    token_type: 'Bearer',
    user: {
      id: userId,
      plan: 'free',
      auth_method: 'device',
      attest_verified: true,
      created_at: now,
    },
  });
}

### 3. Attestation検証ロジック :large_blue_circle:

03-backend-proxy.mdセクション4.4.3、05-security.mdセクション4.6.2準拠。

```typescript
// src/services/appAttest.ts
interface AttestationData {
  keyId: string;
  attestation: string;  // Base64 CBOR
  challenge: string;
}

async function verifyAttestation(
  data: AttestationData,
  deviceId: string,
  env: Env
): Promise<{ publicKey: CryptoKey; receipt: Uint8Array }> {
  // 1. CBOR デコード
  const attestObj = decodeCBOR(base64Decode(data.attestation));

  // 2. 証明書チェーン検証（Apple App Attest Root CA）
  const leafCert = attestObj.attStmt.x5c[0];
  const intermediateCert = attestObj.attStmt.x5c[1];
  await verifyCertChain(leafCert, intermediateCert, env.APPLE_ROOT_CA);

  // 3. nonce検証
  const clientDataHash = await sha256(data.challenge + deviceId);
  const expectedNonce = await sha256(
    concatBuffers(attestObj.authData, clientDataHash)
  );
  const certNonce = extractNonceFromCert(leafCert);
  if (!timingSafeEqual(certNonce, expectedNonce)) {
    throw new AppError('ATTEST_FAILED', 'Attestation nonce mismatch', 403);
  }

  // 4. App ID検証
  const expectedAppId = `${env.APPLE_TEAM_ID}.${env.APPLE_CLIENT_ID}`;
  const rpIdHash = attestObj.authData.slice(0, 32);
  const expectedRpIdHash = await sha256(expectedAppId);
  if (!timingSafeEqual(rpIdHash, expectedRpIdHash)) {
    throw new AppError('ATTEST_FAILED', 'App ID mismatch', 403);
  }

  // 5. counter == 0（初回attestation）
  const counter = new DataView(attestObj.authData.buffer).getUint32(33);
  if (counter !== 0) {
    throw new AppError('ATTEST_FAILED', 'Counter must be 0 for initial attestation', 403);
  }

  // 6. credentialId == keyId
  const credentialId = extractCredentialId(attestObj.authData);
  if (credentialId !== data.keyId) {
    throw new AppError('ATTEST_FAILED', 'Credential ID mismatch', 403);
  }

  const publicKey = await extractPublicKey(leafCert);
  return { publicKey, receipt: attestObj.attStmt.receipt };
}
```

### 4. JWT生成・検証 :large_blue_circle:

03-backend-proxy.mdセクション4.3準拠。ES256署名。

```typescript
// src/services/jwt.ts
import { sign, verify } from 'hono/jwt';

interface JWTPayload {
  sub: string;         // ユーザーID (usr_xxx)
  plan: 'free' | 'pro';
  auth_method: 'device' | 'apple';
  device_id: string;
}

async function generateJWT(payload: JWTPayload, env: Env): Promise<string> {
  const now = Math.floor(Date.now() / 1000);
  return sign(
    {
      ...payload,
      iss: 'voicememo-proxy',
      aud: 'voicememo-ios',
      iat: now,
      exp: now + 86400, // 24時間
    },
    env.JWT_PRIVATE_KEY,
    'ES256'
  );
}

async function verifyJWT(token: string, env: Env): Promise<JWTPayload> {
  const payload = await verify(token, env.JWT_PUBLIC_KEY, 'ES256');

  if (payload.exp < Math.floor(Date.now() / 1000)) {
    throw new AppError('UNAUTHORIZED', 'トークンが期限切れです', 401);
  }

  return {
    sub: payload.sub as string,
    plan: payload.plan as 'free' | 'pro',
    auth_method: payload.auth_method as 'device' | 'apple',
    device_id: payload.device_id as string,
  };
}
```

```typescript
// src/middleware/auth.ts
const authMiddleware = async (c: Context, next: Next) => {
  const authHeader = c.req.header('Authorization');
  if (!authHeader?.startsWith('Bearer ')) {
    return c.json({
      error: { code: 'UNAUTHORIZED', message: '認証トークンが必要です' }
    }, 401);
  }

  const token = authHeader.substring(7);

  try {
    const payload = await verifyJWT(token, c.env);
    c.set('user', {
      id: payload.sub,
      plan: payload.plan,
      authMethod: payload.auth_method,
      deviceId: payload.device_id,
    });
    await next();
  } catch (e) {
    return c.json({
      error: { code: 'UNAUTHORIZED', message: '無効なトークンです' }
    }, 401);
  }
};
```

### 6. App Attest Assertionミドルウェア :large_blue_circle:

03-backend-proxy.mdセクション4.4.2、4.4.3準拠。

```typescript
// src/middleware/appAttest.ts
const appAttestMiddleware = async (c: Context, next: Next) => {
  const user = c.get('user');
  const assertionHeader = c.req.header('X-App-Attest-Assertion');

  if (!assertionHeader) {
    // D1でユーザーのattest_verified状態を確認
    const device = await c.env.DB.prepare(
      'SELECT attest_verified FROM user_devices WHERE user_id = ? AND device_id = ?'
    ).bind(user.id, user.deviceId).first();

    if (device?.attest_verified) {
      return c.json({
        error: { code: 'ATTEST_REQUIRED', message: 'App Attest assertionが必要です' }
      }, 403);
    }

    // 未attest端末は制限付きで許可
    c.set('attestVerified', false);
    await next();
    return;
  }

  try {
    const body = await c.req.raw.clone().arrayBuffer();
    const clientDataHash = new Uint8Array(
      await crypto.subtle.digest('SHA-256', body)
    );

    const device = await c.env.DB.prepare(
      'SELECT attest_public_key, attest_counter FROM user_devices WHERE user_id = ? AND device_id = ?'
    ).bind(user.id, user.deviceId).first();

    if (!device) {
      return c.json({
        error: { code: 'DEVICE_NOT_FOUND', message: 'デバイスが登録されていません' }
      }, 403);
    }

    const assertion = base64Decode(assertionHeader);
    const publicKey = await importPublicKey(device.attest_public_key);

    const { newCounter } = await verifyAssertion(
      assertion, clientDataHash, publicKey, device.attest_counter, c.env
    );

    // counter更新
    await c.env.DB.prepare(
      'UPDATE user_devices SET attest_counter = ?, updated_at = ? WHERE user_id = ? AND device_id = ?'
    ).bind(newCounter, new Date().toISOString(), user.id, user.deviceId).run();

    c.set('attestVerified', true);
    await next();
  } catch (e) {
    return c.json({
      error: { code: 'ATTEST_FAILED', message: 'App Attest検証に失敗しました' }
    }, 403);
  }
};
```

## テスト要件

### 正常系
- GET /api/v1/auth/challenge がchallenge + expires_in=300を返す
- challengeがKVに保存されTTL 300秒が設定される
- POST /api/v1/auth/device (attest付き) がJWT + userオブジェクトを返す
- attestation検証の全項目（証明書チェーン、nonce、App ID、counter、credentialId）が通過する
- 生成されたJWTがES256で署名され、payload(sub, plan, auth_method, device_id)を含む
- D1にusers + user_devicesレコードが作成される
- 認証ミドルウェアが有効なJWTを検証しユーザー情報をコンテキストに設定する
- assertionミドルウェアがcounter単調増加を検証する

### 異常系
- 無効なchallengeでのattestation失敗（400）
- 期限切れchallengeの拒否（400）
- attestation nonce不一致（403 ATTEST_FAILED）
- App ID不一致（403 ATTEST_FAILED）
- 初回attestationでcounter != 0（403 ATTEST_FAILED）
- 無効なJWTトークンの拒否（401 UNAUTHORIZED）
- 期限切れJWTの拒否（401 UNAUTHORIZED）
- attest済み端末がassertionなしでリクエスト（403 ATTEST_REQUIRED）
- assertionのcounterリプレイ検出（403 ATTEST_FAILED）

### フォールバック検証
- App Attest非対応端末（attest_supported: false）が制限付きで登録される
- attest_verified=false の端末がassertionなしでAPI利用可能（制限付き）
