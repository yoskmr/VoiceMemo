### 1. LLMOrchestrator :large_blue_circle:

02-ai-pipeline.mdセクション3.5準拠。Critical #3修正のタスク分割ロジック。

```swift
// FeatureAI/LLMOrchestrator.swift
import Foundation

final class LLMOrchestrator {
    private let onDeviceProvider: OnDeviceLLMProvider
    private let cloudProvider: CloudLLMProvider
    private let quotaManager: AIQuotaManager
    private let deviceChecker: DeviceCapabilityChecker
    private let networkMonitor: NetworkMonitor
    private let memoryController: MemoryExclusionController
    private let offlineQueue: OfflineAIQueue

    /// 最適なプロバイダを選択してLLM処理を実行
    func processText(_ text: String, plan: SubscriptionPlan) async throws -> LLMResponse {
        // 1. クォータチェック（REQ-011）
        if plan == .free {
            guard try await quotaManager.canProcess() else {
                throw LLMError.quotaExceeded
            }
        }

        // 2. タスク判定（オンデバイス/クラウドに分割）
        let isOnDeviceCapable = deviceChecker.supportsOnDeviceLLM
            && deviceChecker.hasMemoryHeadroomForLLM
        let isNetworkAvailable = networkMonitor.isConnected
        let (onDeviceTasks, cloudTasks) = determineTasks(
            text: text,
            plan: plan,
            isOnDeviceCapable: isOnDeviceCapable,
            isNetworkAvailable: isNetworkAvailable
        )

        var mergedResponse = LLMResponse(
            summary: nil, tags: nil, sentiment: nil,
            processingTimeMs: 0, provider: ""
        )

        // 3. オンデバイス処理（要約+タグ）
        if !onDeviceTasks.isEmpty {
            let onDeviceResponse = try await executeOnDevice(text: text, tasks: onDeviceTasks)
            mergedResponse = mergeResponses(base: mergedResponse, overlay: onDeviceResponse)
        }

        // 4. クラウド処理（感情分析、または全タスク一括）
        if !cloudTasks.isEmpty {
            do {
                let cloudResponse = try await executeCloud(text: text, tasks: cloudTasks)
                mergedResponse = mergeResponses(base: mergedResponse, overlay: cloudResponse)
            } catch {
                // クラウド失敗時: オンデバイスフォールバック（感情分析は省略）
                if onDeviceTasks.isEmpty && isOnDeviceCapable {
                    let fallbackTasks: Set<LLMTask> = [.summarize, .tagging]
                    let fallbackResponse = try await executeOnDevice(text: text, tasks: fallbackTasks)
                    mergedResponse = mergeResponses(base: mergedResponse, overlay: fallbackResponse)
                }
                // 感情分析のみの場合はキューに入れて後でリトライ
                if cloudTasks == [.sentimentAnalysis] {
                    await offlineQueue.enqueue(text: text, tasks: cloudTasks)
                }
            }
        }

        // 5. クォータ消費記録（無料プラン）
        if plan == .free {
            try await quotaManager.recordUsage()
        }

        return mergedResponse
    }
}
```

### 2. determineTasks() — タスク分割ロジック :large_blue_circle:

02-ai-pipeline.mdセクション3.4、3.5準拠。Critical #3修正版。

```swift
// FeatureAI/LLMOrchestrator+Tasks.swift
extension LLMOrchestrator {
    /// タスク判定ロジック ※統合仕様書 v1.0 準拠
    /// オンデバイス対応デバイスでは、要約+タグのみをオンデバイスで処理し、
    /// 感情分析はクラウドで別途処理する（または省略する）
    func determineTasks(
        text: String,
        plan: SubscriptionPlan,
        isOnDeviceCapable: Bool,
        isNetworkAvailable: Bool
    ) -> (onDeviceTasks: Set<LLMTask>, cloudTasks: Set<LLMTask>) {
        let allTasks: Set<LLMTask> = [.summarize, .tagging, .sentimentAnalysis]
        let onDeviceSupportedTasks: Set<LLMTask> = [.summarize, .tagging]

        // ケース1: クラウド利用可能 → 全タスクをクラウドで一括処理
        //   条件: ネットワーク接続中 かつ (Proプラン or 長文 or 非対応端末)
        if isNetworkAvailable && (plan == .pro || text.count > 500 || !isOnDeviceCapable) {
            return (onDeviceTasks: [], cloudTasks: allTasks)
        }

        // ケース2: オンデバイス対応 + オフライン → 要約+タグのみ、感情分析は省略
        if isOnDeviceCapable && !isNetworkAvailable {
            return (onDeviceTasks: onDeviceSupportedTasks, cloudTasks: [])
        }

        // ケース3: オンデバイス対応 + オンライン + 短文(<=500字)
        //   → 要約+タグはオンデバイス、感情分析のみクラウド
        if isOnDeviceCapable && isNetworkAvailable && text.count <= 500 {
            return (onDeviceTasks: onDeviceSupportedTasks, cloudTasks: [.sentimentAnalysis])
        }

        // デフォルト: クラウドで全処理
        return (onDeviceTasks: [], cloudTasks: allTasks)
    }
}
```

```swift
// FeatureAI/LLMOrchestrator+Merge.swift
extension LLMOrchestrator {
    /// 2段階処理の結果をマージ
    func mergeResponses(base: LLMResponse, overlay: LLMResponse) -> LLMResponse {
        LLMResponse(
            summary: overlay.summary ?? base.summary,
            tags: overlay.tags ?? base.tags,
            sentiment: overlay.sentiment ?? base.sentiment,
            processingTimeMs: base.processingTimeMs + overlay.processingTimeMs,
            provider: [base.provider, overlay.provider]
                .filter { !$0.isEmpty }
                .joined(separator: "+")
        )
    }
}
```

### 4. オンデバイス実行（メモリ排他制御付き） :large_blue_circle:

02-ai-pipeline.mdセクション3.2.3準拠。

```swift
// FeatureAI/LLMOrchestrator+Execute.swift
extension LLMOrchestrator {
    func executeOnDevice(text: String, tasks: Set<LLMTask>) async throws -> LLMResponse {
        // メモリ排他制御: STTアンロード → LLMロード
        try await memoryController.acquireLLM(
            sttEngine: nil, // STTエンジン参照は外部から注入
            llmProvider: onDeviceProvider
        )

        defer {
            Task {
                await memoryController.releaseLLM(
                    llmProvider: onDeviceProvider,
                    sttEngine: nil
                )
            }
        }

        let request = LLMRequest(
            text: text,
            tasks: tasks,
            language: "ja",
            maxTokens: onDeviceProvider.maxInputTokens
        )
        return try await onDeviceProvider.process(request)
    }

    func executeCloud(text: String, tasks: Set<LLMTask>) async throws -> LLMResponse {
        let request = LLMRequest(
            text: text,
            tasks: tasks,
            language: "ja",
            maxTokens: cloudProvider.maxInputTokens
        )
        return try await cloudProvider.process(request)
    }
}
```

### 5. オフラインキューイング（BGTaskScheduler） :large_blue_circle:

02-ai-pipeline.mdセクション5準拠。

```swift
// FeatureAI/Queue/OfflineAIQueue.swift
import BackgroundTasks

final class OfflineAIQueue {
    private let modelContext: ModelContext  // SwiftData

    /// オフライン処理をキューに追加
    func enqueue(text: String, tasks: Set<LLMTask>) async {
        let task = AIProcessingTask(
            id: UUID(),
            memoId: UUID(), // 関連メモID
            status: .queued,
            priority: .normal,
            createdAt: Date(),
            retryCount: 0,
            maxRetries: 3
        )

        modelContext.insert(task)
        try? modelContext.save()

        // BGTaskSchedulerに登録
        scheduleBackgroundProcessing()
    }

    /// バックグラウンドタスクのスケジューリング
    func scheduleBackgroundProcessing() {
        let request = BGProcessingTaskRequest(
            identifier: "io.murmurnote.ai-processing"
        )
        request.requiresNetworkConnectivity = true
        request.requiresExternalPower = false
        request.earliestBeginDate = Date(timeIntervalSinceNow: 60)

        try? BGTaskScheduler.shared.submit(request)
    }

    /// バックグラウンドタスクの実行
    func processQueuedTasks(cloudProvider: CloudLLMProvider) async {
        let descriptor = FetchDescriptor<AIProcessingTask>(
            predicate: #Predicate { $0.status == .queued },
            sortBy: [SortDescriptor(\.priority), SortDescriptor(\.createdAt)]
        )

        guard let pendingTasks = try? modelContext.fetch(descriptor) else { return }

        for task in pendingTasks {
            task.status = .processing
            task.startedAt = Date()

            do {
                // クラウドLLMで処理
                let request = LLMRequest(
                    text: task.inputText ?? "",
                    tasks: Set(LLMTask.allCases),
                    language: "ja",
                    maxTokens: cloudProvider.maxInputTokens
                )
                let response = try await cloudProvider.process(request)

                task.status = .completed
                task.completedAt = Date()
                task.providerUsed = response.provider
            } catch {
                task.retryCount += 1
                if task.retryCount >= task.maxRetries {
                    task.status = .failed
                    task.errorMessage = error.localizedDescription
                } else {
                    task.status = .queued
                    task.nextRetryAt = Date(timeIntervalSinceNow: pow(2, Double(task.retryCount)) * 60)
                }
            }

            try? modelContext.save()
        }
    }
}
```

### 6. AIQuotaManager :large_blue_circle:

```swift
// FeatureAI/Quota/AIQuotaManager.swift

/// AI処理クォータの管理（ローカルチェック）
/// サーバー側カウンタとの同期はTASK-0030で実装
final class AIQuotaManager {
    private let userSettings: UserSettings

    /// 処理可能かチェック（ローカル判定）
    func canProcess() async throws -> Bool {
        let settings = userSettings
        // 月次リセット判定
        if settings.shouldResetMonthlyCount() {
            settings.aiProcessingCountThisMonth = 0
            settings.lastAICountResetDate = Date()
        }
        return settings.aiProcessingCountThisMonth < 5
    }

    /// 使用量を記録（ローカルインクリメント）
    func recordUsage() async throws {
        userSettings.aiProcessingCountThisMonth += 1
    }
}
```
