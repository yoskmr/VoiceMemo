```swift
// VoiceMemoPerformanceTests/PerformanceTests.swift

import XCTest
@testable import Domain
@testable import InfraStorage

final class PerformanceTests: XCTestCase {

    /// NFR-001: 録音開始レイテンシ 500ms以内（p95）
    func testRecordingStartLatency() throws {
        let metrics: [XCTMetric] = [
            XCTClockMetric(),
            XCTMemoryMetric(),
        ]

        let options = XCTMeasureOptions()
        options.iterationCount = 20  // p95を得るために20回計測

        measure(metrics: metrics, options: options) {
            let recorder = AudioRecorder()
            let start = CFAbsoluteTimeGetCurrent()
            try? recorder.startRecording()
            let elapsed = CFAbsoluteTimeGetCurrent() - start
            XCTAssertLessThan(elapsed, 0.5, "録音開始が500msを超過")
            recorder.stopRecording()
        }
    }

    /// NFR-002: 検索レスポンス 1000件1秒以内（p95）
    func testSearchPerformanceWith1000Memos() throws {
        // テストデータ準備: 1000件のメモをFTS5インデックスに登録
        let indexManager = FTS5IndexManager.testInstance()
        for i in 0..<1000 {
            try indexManager.index(
                memoID: UUID(),
                text: "テストメモ \(i) 音声メモアプリのアイデアについて記録しました。"
            )
        }

        measure {
            let results = try? indexManager.search(query: "アイデア")
            XCTAssertNotNil(results)
        }
    }

    /// NFR-003: メモ一覧スクロールパフォーマンス（60fps維持）
    func testMemoListScrollPerformance() throws {
        let app = XCUIApplication()
        app.launch()

        // 100件のメモが表示された一覧画面
        let memoList = app.collectionViews.firstMatch

        let metrics: [XCTMetric] = [
            XCTOSSignpostMetric.scrollDecelerationMetric,
        ]

        measure(metrics: metrics) {
            memoList.swipeUp(velocity: .fast)
            memoList.swipeDown(velocity: .fast)
        }
    }

    /// NFR-004: アプリ起動時間（Cold Start 2秒以内）
    func testAppLaunchPerformance() throws {
        measure(metrics: [XCTApplicationLaunchMetric()]) {
            XCUIApplication().launch()
        }
    }

    /// NFR-005: AI処理中のメモリ使用量（ピーク500MB以内）
    func testAIProcessingMemoryUsage() throws {
        let metrics: [XCTMetric] = [
            XCTMemoryMetric(),
        ]

        measure(metrics: metrics) {
            // AI処理のシミュレーション
            let processor = AIProcessingUseCase.testInstance()
            let longText = String(repeating: "テスト文章です。", count: 500) // 長文テスト
            _ = try? processor.process(text: longText)
        }
    }
}
```

```swift
// VoiceMemoPerformanceTests/MemoryLeakTests.swift

final class MemoryLeakTests: XCTestCase {

    /// TCA Storeのメモリリーク検出
    func testStoreDoesNotLeak() {
        var store: StoreOf<AppReducer>? = Store(
            initialState: AppReducer.State()
        ) {
            AppReducer()
        }

        weak var weakStore = store
        store = nil

        // Storeが解放されることを確認
        XCTAssertNil(weakStore, "AppReducer Storeがメモリリークしています")
    }

    /// Recording Effectのメモリリーク検出
    func testRecordingEffectDoesNotLeak() async {
        var recorder: AudioRecorder? = AudioRecorder()

        weak var weakRecorder = recorder
        try? recorder?.startRecording()
        recorder?.stopRecording()
        recorder = nil

        XCTAssertNil(weakRecorder, "AudioRecorderがメモリリークしています")
    }

    /// SwiftData ModelContextのメモリリーク検出
    func testModelContextDoesNotLeak() throws {
        var context: ModelContext? = try ModelContext(
            ModelContainerConfiguration.create()
        )

        weak var weakContext = context
        context = nil

        XCTAssertNil(weakContext, "ModelContextがメモリリークしています")
    }

    /// NavigationStack遷移でのメモリリーク検出
    func testNavigationDoesNotLeak() {
        var detailStore: StoreOf<MemoDetailReducer>? = Store(
            initialState: MemoDetailReducer.State(memoID: UUID())
        ) {
            MemoDetailReducer()
        }

        weak var weakStore = detailStore
        detailStore = nil

        XCTAssertNil(weakStore, "MemoDetailReducer Storeがメモリリークしています")
    }
}
```

### 5. パフォーマンス最適化チェックリスト :large_blue_circle:

NFR-001〜006に対応する最適化項目。

| NFR | 項目 | 目標 | 計測方法 | 最適化手法 |
|:----|:-----|:-----|:---------|:-----------|
| NFR-001 | 録音開始レイテンシ | 500ms以内(p95) | XCTClockMetric | AudioSession事前準備、録音バッファプリアロケーション |
| NFR-002 | 検索レスポンス（1000件） | 1秒以内(p95) | XCTest measure | FTS5インデックス最適化、クエリキャッシュ |
| NFR-003 | 一覧スクロール | 60fps | scrollDecelerationMetric | LazyVStack、画像キャッシュ、セルリサイクル |
| NFR-004 | Cold Start | 2秒以内 | XCTApplicationLaunchMetric | 遅延初期化、モジュール分割による起動時ロード最小化 |
| NFR-005 | AI処理メモリ | ピーク500MB以内 | XCTMemoryMetric | ストリーミング処理、中間結果の即時解放 |
| NFR-006 | バッテリー消費 | 録音1時間で5%以内 | Energy Gaugeログ | 効率的な音声バッファサイズ、不要なUI更新抑制 |

```swift
// パフォーマンス最適化の実装例

// 1. AudioSession事前準備（録音開始レイテンシ削減）
// AppReducer.onAppear で事前にAudioSessionを設定しておく
case .appDidBecomeActive:
    return .run { _ in
        try? AVAudioSession.sharedInstance().setCategory(.playAndRecord, mode: .default)
        try? AVAudioSession.sharedInstance().setActive(true)
    }

// 2. LazyVStack + prefetch（スクロールパフォーマンス）
LazyVStack(spacing: VMSpacing.md) {
    ForEach(store.memos) { memo in
        MemoCard(memo: memo)
            .onAppear {
                // プリフェッチ: 次の10件を先読み
                if memo == store.memos.last {
                    store.send(.loadMoreMemos)
                }
            }
    }
}

// 3. FTS5クエリ最適化
// 検索クエリをデバウンス（300ms）して不要なクエリを抑制
case .searchQueryChanged(let query):
    return .run { send in
        try await Task.sleep(for: .milliseconds(300))
        let results = try await searchIndex.search(query: query)
        await send(.searchResultsLoaded(results))
    }
    .cancellable(id: CancelID.search, cancelInFlight: true)
```
