### 3.1 TCA State / Action / Reducer

```swift
// FeatureMemo/MemoList/MemoListReducer.swift
import ComposableArchitecture
import Foundation

@Reducer
struct MemoListReducer {
    @ObservableState
    struct State: Equatable {
        var memos: IdentifiedArrayOf<MemoItem> = []
        var sections: [MemoSection] = []
        var isLoading: Bool = false
        var hasMorePages: Bool = true
        var currentPage: Int = 0
        var errorMessage: String?

        // ページネーション設定（NFR-005: 1,000件一覧 1秒以内）
        static let pageSize = 50
    }

    struct MemoItem: Equatable, Identifiable {
        let id: UUID
        var title: String
        var createdAt: Date
        var durationSeconds: Double
        var transcriptPreview: String  // 最大2行分
        var emotion: EmotionCategory?
        var tags: [String]
        var audioFilePath: String
    }

    struct MemoSection: Equatable, Identifiable {
        var id: String { label }
        let label: String       // "今日", "昨日", "2026年3月14日" 等
        let date: Date
        var memoIDs: [UUID]
    }

    enum Action {
        case onAppear
        case loadNextPage
        case memosLoaded(Result<[MemoItem], Error>)
        case memoTapped(id: UUID)
        case swipeToDelete(id: UUID)
        case deleteConfirmed(id: UUID)
        case deleteCancelled
        case memoDeleted(Result<UUID, Error>)
        case searchIconTapped
        case trendIconTapped
        case refreshRequested
        case refreshCompleted(Result<[MemoItem], Error>)
    }

    @Dependency(\.memoRepository) var memoRepository
    @Dependency(\.date.now) var now
    @Dependency(\.calendar) var calendar

    var body: some ReducerOf<Self> {
        Reduce { state, action in
            switch action {
            case .onAppear:
                guard state.memos.isEmpty else { return .none }
                state.isLoading = true
                state.currentPage = 0
                return .run { send in
                    let result = await Result {
                        try await memoRepository.fetchMemos(
                            page: 0,
                            pageSize: State.pageSize
                        )
                    }
                    await send(.memosLoaded(result))
                }

            case .loadNextPage:
                guard !state.isLoading, state.hasMorePages else { return .none }
                state.isLoading = true
                let nextPage = state.currentPage + 1
                return .run { send in
                    let result = await Result {
                        try await memoRepository.fetchMemos(
                            page: nextPage,
                            pageSize: State.pageSize
                        )
                    }
                    await send(.memosLoaded(result))
                }

            case let .memosLoaded(.success(newMemos)):
                state.isLoading = false
                state.currentPage += 1
                state.hasMorePages = newMemos.count >= State.pageSize
                for memo in newMemos {
                    state.memos.updateOrAppend(memo)
                }
                state.sections = buildSections(from: state.memos, now: now, calendar: calendar)
                return .none

            case let .memosLoaded(.failure(error)):
                state.isLoading = false
                state.errorMessage = error.localizedDescription
                return .none

            case let .swipeToDelete(id):
                // 確認ダイアログの表示はTASK-0017で実装
                return .send(.deleteConfirmed(id: id))

            case .refreshRequested:
                state.isLoading = true
                state.currentPage = 0
                return .run { send in
                    let result = await Result {
                        try await memoRepository.fetchMemos(
                            page: 0,
                            pageSize: State.pageSize
                        )
                    }
                    await send(.refreshCompleted(result))
                }

            case let .refreshCompleted(.success(memos)):
                state.isLoading = false
                state.currentPage = 1
                state.memos = IdentifiedArrayOf(uniqueElements: memos)
                state.hasMorePages = memos.count >= State.pageSize
                state.sections = buildSections(from: state.memos, now: now, calendar: calendar)
                return .none

            default:
                return .none
            }
        }
    }

    // 日付セクションの構築
    private func buildSections(
        from memos: IdentifiedArrayOf<MemoItem>,
        now: Date,
        calendar: Calendar
    ) -> [MemoSection] {
        let grouped = Dictionary(grouping: memos.elements) { memo in
            calendar.startOfDay(for: memo.createdAt)
        }

        return grouped.keys.sorted(by: >).map { date in
            let label = sectionLabel(for: date, now: now, calendar: calendar)
            let ids = grouped[date]!.sorted { $0.createdAt > $1.createdAt }.map(\.id)
            return MemoSection(label: label, date: date, memoIDs: ids)
        }
    }

    private func sectionLabel(for date: Date, now: Date, calendar: Calendar) -> String {
        if calendar.isDateInToday(date) { return "今日" }
        if calendar.isDateInYesterday(date) { return "昨日" }
        let formatter = DateFormatter()
        formatter.locale = Locale(identifier: "ja_JP")
        formatter.dateFormat = "yyyy年M月d日"
        return formatter.string(from: date)
    }
}
```

### 3.2 MemoCardコンポーネント

```swift
// SharedUI/Components/MemoCard.swift
// 設計書 04-ui-design-system.md セクション4.2 準拠

struct MemoCard: View {
    let memo: MemoListReducer.MemoItem

    var body: some View {
        VStack(alignment: .leading, spacing: 8) {
            // ヘッダー: 日時 + 録音時間
            HStack {
                Label(formattedDate, systemImage: "calendar")
                    .font(.vmCaption1)
                    .foregroundColor(.vmTextSecondary)
                Spacer()
                Label(formattedDuration, systemImage: "mic.fill")
                    .font(.vmCaption1)
                    .foregroundColor(.vmTextTertiary)
            }

            // タイトル
            Text(memo.title)
                .font(.vmTitle3)
                .foregroundColor(.vmTextPrimary)
                .lineLimit(1)

            // プレビューテキスト
            Text(memo.transcriptPreview)
                .font(.vmCallout)
                .foregroundColor(.vmTextSecondary)
                .lineLimit(2)

            // フッター: 感情バッジ + タグ
            HStack(spacing: 8) {
                if let emotion = memo.emotion {
                    EmotionBadge(emotion: emotion)
                }
                ForEach(memo.tags.prefix(3), id: \.self) { tag in
                    TagChip(text: tag)
                }
            }
        }
        .padding(16)
        .background(Color.vmSurface)
        .cornerRadius(16)                   // NFR-012: 12pt以上
        .shadow(color: .black.opacity(0.05), radius: 8, y: 2)
        // アクセシビリティ（設計書 セクション8.1）
        .accessibilityElement(children: .combine)
        .accessibilityLabel("\(memo.title)、\(formattedDate)、\(formattedDuration)の録音")
        .accessibilityHint("ダブルタップで詳細を表示します")
        .accessibilityValue(memo.emotion?.label ?? "感情分析なし")
    }
}
```

### 3.3 MemoListView（SwiftUI）

```swift
// FeatureMemo/MemoList/MemoListView.swift

struct MemoListView: View {
    @Bindable var store: StoreOf<MemoListReducer>

    var body: some View {
        NavigationStack {
            ScrollView {
                LazyVStack(spacing: 12, pinnedViews: [.sectionHeaders]) {
                    ForEach(store.sections) { section in
                        Section {
                            ForEach(section.memoIDs, id: \.self) { memoID in
                                if let memo = store.memos[id: memoID] {
                                    MemoCard(memo: memo)
                                        .onTapGesture {
                                            store.send(.memoTapped(id: memoID))
                                        }
                                        .swipeActions(edge: .trailing) {
                                            Button(role: .destructive) {
                                                store.send(.swipeToDelete(id: memoID))
                                            } label: {
                                                Label("削除", systemImage: "trash")
                                            }
                                        }
                                        .padding(.horizontal, 16)
                                }
                            }

                            // ページネーション: 最後のセクションの最後のアイテム付近でトリガー
                            if section.id == store.sections.last?.id,
                               let lastID = section.memoIDs.last,
                               store.hasMorePages {
                                Color.clear
                                    .frame(height: 1)
                                    .onAppear {
                                        store.send(.loadNextPage)
                                    }
                            }
                        } header: {
                            SectionHeader(label: section.label)
                        }
                    }

                    if store.isLoading {
                        ProgressView()
                            .padding()
                    }
                }
            }
            .background(Color.vmBackground)
            .navigationTitle("メモ")
            .toolbar {
                ToolbarItem(placement: .topBarTrailing) {
                    HStack(spacing: 16) {
                        Button { store.send(.trendIconTapped) } label: {
                            Image(systemName: "chart.line.uptrend.xyaxis")
                        }
                        Button { store.send(.searchIconTapped) } label: {
                            Image(systemName: "magnifyingglass")
                        }
                    }
                }
            }
            .refreshable {
                store.send(.refreshRequested)
            }
        }
        .onAppear {
            store.send(.onAppear)
        }
    }
}

// 日付セクションヘッダー
struct SectionHeader: View {
    let label: String

    var body: some View {
        HStack {
            Text(label)
                .font(.vmSubheadline)
                .foregroundColor(.vmTextSecondary)
            Spacer()
        }
        .padding(.horizontal, 16)
        .padding(.vertical, 8)
        .background(Color.vmBackground)
    }
}
```
