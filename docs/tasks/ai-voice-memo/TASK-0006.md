### 1. WhisperKitEngine :large_blue_circle:

統合仕様書セクション3.1のSTTEngineProtocol準拠。enum値は `.whisperKit`（旧 `.whisperCpp` / `.whisperOnDevice` を統一）。

```swift
// InfraSTT/Engines/WhisperKitEngine.swift
import WhisperKit
import AVFoundation

final class WhisperKitEngine: STTEngineProtocol, @unchecked Sendable {
    let engineType: STTEngineType = .whisperKit
    let supportedLanguages: [String] = ["ja", "en", "zh", "ko"]

    private var whisperKit: WhisperKit?
    private let modelDirectory: URL  // Library/Caches/Models/
    private let modelName = "whisper-small"

    func startTranscription(
        audioStream: AsyncStream<AVAudioPCMBuffer>,
        language: String
    ) -> AsyncStream<TranscriptionResult> {
        AsyncStream { continuation in
            Task {
                // モデルがロードされていなければロード
                if whisperKit == nil {
                    try await loadModel()
                }

                // 3秒間隔のチャンク処理
                var audioBuffer: [Float] = []
                let chunkDuration: Double = 3.0
                let sampleRate = 16000

                for await pcmBuffer in audioStream {
                    let floatData = extractFloatData(from: pcmBuffer)
                    audioBuffer.append(contentsOf: floatData)

                    // 3秒分のバッファが溜まったら認識実行
                    if audioBuffer.count >= Int(chunkDuration) * sampleRate {
                        let result = try await whisperKit?.transcribe(
                            audioArray: audioBuffer,
                            decodeOptions: DecodingOptions(
                                language: language,
                                wordTimestamps: true
                            )
                        )

                        if let transcription = result?.first {
                            let partialResult = TranscriptionResult(
                                text: transcription.text,
                                confidence: Double(transcription.avgLogprob).normalized(),
                                isFinal: false,
                                language: language,
                                segments: transcription.segments.map { seg in
                                    TranscriptionSegment(
                                        text: seg.text,
                                        startTime: TimeInterval(seg.start),
                                        endTime: TimeInterval(seg.end),
                                        confidence: Double(seg.avgLogprob).normalized()
                                    )
                                }
                            )
                            continuation.yield(partialResult)
                        }
                        audioBuffer.removeAll()
                    }
                }

                // 残りのバッファを処理して最終結果
                if !audioBuffer.isEmpty {
                    // ... 最終チャンクの処理
                }
                continuation.finish()
            }
        }
    }

    func finishTranscription() async throws -> TranscriptionResult {
        // 最後のバッファを処理して最終結果を返す
        // ...
    }

    func isAvailable() async -> Bool {
        // モデルファイルの存在チェック + デバイス性能チェック
        return isModelDownloaded() || canDownloadModel()
    }

    func setCustomDictionary(_ dictionary: [String: String]) async {
        // WhisperKitのinitial_promptにカスタム辞書の用語を設定
        // REQ-025対応
    }
}
```

final class WhisperKitModelManager {

### 3. メモリ管理（モデルロード/アンロード） :large_blue_circle:

WhisperKit Smallモデルは実行時に約600MB-1GBのメモリを使用する。録音/STT処理完了後にモデルをアンロードしてメモリを解放する。

```swift
extension WhisperKitEngine {
    /// モデルのロード
    func loadModel() async throws {
        let config = WhisperKitConfig(
            model: modelName,
            modelFolder: modelDirectory.path
        )
        whisperKit = try await WhisperKit(config)
    }

    /// モデルのアンロード（メモリ解放）
    func unloadModel() {
        whisperKit = nil
        // メモリ解放を促進
    }

    /// メモリ使用量の確認
    func currentMemoryUsage() -> UInt64 {
        var info = mach_task_basic_info()
        var count = mach_msg_type_number_t(MemoryLayout<mach_task_basic_info>.size) / 4
        let result = withUnsafeMutablePointer(to: &info) {
            $0.withMemoryRebound(to: integer_t.self, capacity: 1) {
                task_info(mach_task_self_, task_flavor_t(MACH_TASK_BASIC_INFO), $0, &count)
            }
        }
        return result == KERN_SUCCESS ? info.resident_size : 0
    }
}
```

### 4. カスタム辞書サポート（REQ-025） :large_blue_circle:

WhisperKitの `initial_prompt` パラメータを活用し、カスタム辞書の用語をプロンプトに含めることで認識精度を向上させる。

```swift
func setCustomDictionary(_ dictionary: [String: String]) async {
    let customTerms = dictionary.values.joined(separator: "、")
    self.initialPrompt = "以下の用語を含む可能性があります: \(customTerms)"
}
```

## テスト要件

### 正常系
- WhisperKit認識で日本語テキストが正しく返されること
- `startTranscription` でAsyncStreamが返却され、部分結果が流れること
- `finishTranscription` で最終確定結果が返されること
- `engineType` が `.whisperKit` であること
- カスタム辞書設定後の認識で設定した用語が優先されること

### 異常系
- モデル未ダウンロード時に `isAvailable()` がfalseを返すこと（ダウンロード不可の場合）
- モデルロード失敗時のエラーハンドリング
- メモリ不足時のフォールバック（モデルアンロード -> Apple Speechへ切替通知）

### メモリ管理
- `loadModel()` 後にWhisperKitインスタンスが非nilであること
- `unloadModel()` 後にWhisperKitインスタンスがnilになりメモリが解放されること
- メモリ使用量が妥当な範囲（< 1.5GB）であること

### モデル管理
- `downloadModel()` でモデルファイルが `Library/Caches/Models/` に保存されること
- `isModelDownloaded()` がダウンロード後にtrueを返すこと
- `deleteModel()` でモデルファイルが削除されること
