```swift
// Domain/Models/DailyEmotionData.swift

/// 日次感情データ（集計結果）
struct DailyEmotionData: Identifiable, Equatable {
    let id = UUID()
    let date: Date
    let emotions: [EmotionRatio]
    let memoCount: Int

    /// 主要感情（最大比率のカテゴリ）
    var primaryEmotion: EmotionCategory? {
        emotions.max(by: { $0.ratio < $1.ratio })?.category
    }
}

struct EmotionRatio: Identifiable, Equatable {
    let id = UUID()
    let category: EmotionCategory
    let ratio: Double  // 0.0 - 1.0
}

// Domain/UseCases/EmotionAggregationUseCase.swift

protocol EmotionAggregationUseCaseProtocol {
    /// 指定期間の日次感情データを集計
    func aggregateDaily(from: Date, to: Date) async throws -> [DailyEmotionData]
    /// 週次サマリーを集計
    func aggregateWeekly(weekOf: Date) async throws -> [DailyEmotionData]
    /// 月次サマリーを集計
    func aggregateMonthly(monthOf: Date) async throws -> [DailyEmotionData]
}
```

```swift
struct DailyEmotionSummaryView: View {
```

```swift
// FeatureMemo/EmotionVisualization/EmotionTrendChart.swift

import Charts

struct EmotionTrendChart: View {
    let data: [DailyEmotionData]
    @State private var selectedDate: Date?
    @State private var timeRange: TimeRange = .weekly

    enum TimeRange: String, CaseIterable {
        case weekly = "週"
        case monthly = "月"
    }

    var body: some View {
        VStack(alignment: .leading, spacing: VMSpacing.lg) {
            // ヘッダー + 切替コントロール
            HStack {
                Text("感情トレンド")
                    .font(.vmHeadline)
                    .foregroundColor(.vmTextPrimary)
                Spacer()
                Picker("期間", selection: $timeRange) {
                    ForEach(TimeRange.allCases, id: \.self) { range in
                        Text(range.rawValue).tag(range)
                    }
                }
                .pickerStyle(.segmented)
                .frame(width: 120)
            }

            // 100%積み上げエリアチャート
            Chart(data) { daily in
                ForEach(daily.emotions, id: \.category) { emotion in
                    AreaMark(
                        x: .value("日付", daily.date),
                        y: .value("比率", emotion.ratio)
                    )
                    .foregroundStyle(emotion.category.color)
                    .interpolationMethod(.catmullRom)
                }
            }
            .chartYScale(domain: 0...1)
            .chartYAxis {
                AxisMarks(format: .percent)
            }
            .chartXAxis {
                AxisMarks(values: .stride(by: timeRange == .weekly ? .day : .weekOfYear)) { value in
                    AxisValueLabel(format: timeRange == .weekly ? .dateTime.weekday(.abbreviated) : .dateTime.day())
                }
            }
            .frame(height: 200)
            .chartOverlay { proxy in
                GeometryReader { geometry in
                    Rectangle()
                        .fill(.clear)
                        .contentShape(Rectangle())
                        .gesture(
                            DragGesture(minimumDistance: 0)
                                .onChanged { value in
                                    let x = value.location.x
                                    if let date: Date = proxy.value(atX: x) {
                                        selectedDate = date
                                    }
                                }
                        )
                }
            }

            // 凡例
            LazyVGrid(columns: Array(repeating: GridItem(.flexible()), count: 4), spacing: VMSpacing.xs) {
                ForEach(EmotionCategory.allCases, id: \.self) { category in
                    HStack(spacing: VMSpacing.xxs) {
                        Circle()
                            .fill(category.color)
                            .frame(width: 8, height: 8)
                        Text(category.label)
                            .font(.vmCaption2)
                            .foregroundColor(.vmTextSecondary)
                    }
                }
            }
        }
        .padding(VMSpacing.lg)
        .background(Color.vmSurface)
        .cornerRadius(VMCornerRadius.medium)
        .vmShadow(VMShadow.level2)
    }
}
```

### 4. カレンダーヒートマップ :large_blue_circle:

04-ui-design-system.md セクション9.4準拠。月間カレンダー形式で各日の主要感情を色で表示。

```swift
// FeatureMemo/EmotionVisualization/EmotionCalendarView.swift

struct EmotionCalendarView: View {
    let month: Date
    let dailyEmotions: [Date: EmotionCategory]
    @State private var selectedDate: Date?
    let onDateSelected: (Date) -> Void
    let onMonthChanged: (Int) -> Void  // -1: 前月, +1: 翌月

    private let weekdayLabels = ["月", "火", "水", "木", "金", "土", "日"]

    var body: some View {
        VStack(spacing: VMSpacing.lg) {
            // ヘッダー: 月表示 + 前後ナビゲーション
            HStack {
                Text(monthTitle)
                    .font(.vmTitle3)
                    .foregroundColor(.vmTextPrimary)
                Spacer()
                HStack(spacing: VMSpacing.lg) {
                    Button(action: { onMonthChanged(-1) }) {
                        Image(systemName: "chevron.left")
                            .foregroundColor(.vmPrimary)
                    }
                    Button(action: { onMonthChanged(1) }) {
                        Image(systemName: "chevron.right")
                            .foregroundColor(.vmPrimary)
                    }
                }
            }

            // 曜日ラベル
            LazyVGrid(columns: Array(repeating: GridItem(.flexible()), count: 7), spacing: VMSpacing.sm) {
                ForEach(weekdayLabels, id: \.self) { label in
                    Text(label)
                        .font(.vmCaption1)
                        .foregroundColor(.vmTextTertiary)
                        .frame(height: 20)
                }
            }

            // カレンダーセル
            LazyVGrid(columns: Array(repeating: GridItem(.flexible()), count: 7), spacing: VMSpacing.sm) {
                ForEach(daysInMonth, id: \.self) { date in
                    CalendarCell(
                        date: date,
                        emotion: dailyEmotions[date],
                        isToday: Calendar.current.isDateInToday(date),
                        isSelected: selectedDate == date
                    )
                    .onTapGesture {
                        selectedDate = date
                        onDateSelected(date)
                    }
                }
            }

            // 凡例
            EmotionLegend()
        }
        .padding(VMSpacing.lg)
        .background(Color.vmSurface)
        .cornerRadius(VMCornerRadius.medium)
        .vmShadow(VMShadow.level2)
    }

    private var monthTitle: String {
        let formatter = DateFormatter()
        formatter.dateFormat = "yyyy年M月"
        formatter.locale = Locale(identifier: "ja_JP")
        return formatter.string(from: month)
    }

    private var daysInMonth: [Date] {
        // カレンダー計算: 月の全日を返す（前月末の空白セルを含む）
        let calendar = Calendar.current
        guard let range = calendar.range(of: .day, in: .month, for: month),
              let firstDay = calendar.date(from: calendar.dateComponents([.year, .month], from: month))
        else { return [] }

        let weekday = calendar.component(.weekday, from: firstDay)
        let offset = (weekday + 5) % 7  // 月曜始まりへの変換

        return (0..<(offset + range.count)).compactMap { index in
            calendar.date(byAdding: .day, value: index - offset, to: firstDay)
        }
    }
}

// MARK: - カレンダーセル（40x40pt、角丸12pt）
struct CalendarCell: View {
    let date: Date
    let emotion: EmotionCategory?
    let isToday: Bool
    let isSelected: Bool

    var body: some View {
        VStack(spacing: VMSpacing.xxs) {
            Text("\(Calendar.current.component(.day, from: date))")
                .font(.vmCaption1)
                .foregroundColor(isCurrentMonth ? .vmTextPrimary : .vmTextTertiary)
        }
        .frame(width: 40, height: 40)
        .background(
            RoundedRectangle(cornerRadius: VMCornerRadius.small)
                .fill(emotion?.color.opacity(0.6) ?? Color.vmSurfaceVariant)
        )
        .overlay(
            RoundedRectangle(cornerRadius: VMCornerRadius.small)
                .stroke(isToday ? Color.vmPrimary : .clear, lineWidth: 2)
        )
        .overlay(
            RoundedRectangle(cornerRadius: VMCornerRadius.small)
                .stroke(isSelected ? Color.vmAccent : .clear, lineWidth: 1)
        )
        .accessibilityLabel(accessibilityDescription)
    }

    private var isCurrentMonth: Bool {
        // 実装時に月の判定ロジックを追加
        true
    }

    private var accessibilityDescription: String {
        let formatter = DateFormatter()
        formatter.dateFormat = "M月d日"
        formatter.locale = Locale(identifier: "ja_JP")
        let dateStr = formatter.string(from: date)
        let emotionStr = emotion?.label ?? "メモなし"
        return "\(dateStr)、\(emotionStr)"
    }
}
```

### 5. TCA Reducer統合 :large_blue_circle:

```swift
// FeatureMemo/EmotionVisualization/EmotionVisualizationReducer.swift

@Reducer
struct EmotionVisualizationReducer {
    @ObservableState
    struct State: Equatable {
        var dailySummary: DailyEmotionData?
        var trendData: [DailyEmotionData] = []
        var calendarData: [Date: EmotionCategory] = [:]
        var currentMonth: Date = Date()
        var timeRange: EmotionTrendChart.TimeRange = .weekly
        var selectedDate: Date?
        var isLoading: Bool = false
    }

    enum Action {
        case onAppear
        case loadDailySummary
        case loadTrendData
        case loadCalendarData
        case dailySummaryLoaded(DailyEmotionData)
        case trendDataLoaded([DailyEmotionData])
        case calendarDataLoaded([Date: EmotionCategory])
        case timeRangeChanged(EmotionTrendChart.TimeRange)
        case monthChanged(Int)
        case dateSelected(Date)
    }

    @Dependency(\.emotionAggregation) var emotionAggregation

    var body: some ReducerOf<Self> {
        Reduce { state, action in
            switch action {
            case .onAppear:
                state.isLoading = true
                return .merge(
                    .send(.loadDailySummary),
                    .send(.loadTrendData),
                    .send(.loadCalendarData)
                )

            case .timeRangeChanged(let range):
                state.timeRange = range
                return .send(.loadTrendData)

            case .monthChanged(let offset):
                let calendar = Calendar.current
                if let newMonth = calendar.date(byAdding: .month, value: offset, to: state.currentMonth) {
                    state.currentMonth = newMonth
                }
                return .send(.loadCalendarData)

            case .dateSelected(let date):
                state.selectedDate = date
                return .none

            default:
                return .none
            }
        }
    }
}
```

## テスト要件

### 正常系
- 日次サマリーのスタックドバーチャートが感情比率に応じた幅で描画される
- 週次トレンドデータが7日分の日次感情データとして集計される
- 月次トレンドデータが当月の全日数分で集計される
- カレンダーヒートマップが正しい月の日数を表示する
- 感情がある日のセルが対応する感情カラー（opacity 0.6）で着色される
- 感情がない日のセルがSurfaceVariant色で表示される
- 今日のセルにPrimary色のボーダーが表示される
- 8カテゴリ全てがEmotionColor準拠のカラーで表示される
- 週次↔月次の切替がセグメンテッドコントロールで動作する

### 異常系
- 感情データが0件の場合、空状態が適切に表示される
- 月のデータが部分的にない場合でもクラッシュしない

### TCA TestStore
- `.onAppear` で3つのロード処理が並行実行される
- `.timeRangeChanged(.monthly)` でトレンドデータが再ロードされる
- `.monthChanged(1)` で `currentMonth` が翌月に更新される
- `.dateSelected` で `selectedDate` が設定される

### アニメーション
- 感情バッジ表示時にEmotionRevealAnimation（vmMicroInteraction: 250ms）が適用される
