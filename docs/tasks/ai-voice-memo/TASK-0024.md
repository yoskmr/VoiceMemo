### 1. AppAttestManager :large_blue_circle:

05-security.mdセクション4.6.4準拠。DCAppAttestServiceのSwiftラッパー。

```swift
// InfraNetwork/AppAttest/AppAttestManager.swift
import DeviceCheck
import CryptoKit

/// App Attestの管理を行うマネージャー
/// ※統合仕様書 v1.0 準拠: Critical #7/#8 対応
final class AppAttestManager: Sendable {
    private let service = DCAppAttestService.shared

    /// App Attest の利用可否
    var isSupported: Bool {
        service.isSupported
    }

    /// キーの生成と登録（初回起動時に1回）
    /// - Parameter challenge: サーバーから取得したchallenge nonce
    /// - Returns: keyIdとattestation objectのペア
    func generateAndAttestKey(challenge: Data) async throws -> (keyId: String, attestation: Data) {
        // 1. キー生成（Secure Enclaveで生成）
        let keyId = try await service.generateKey()

        // 2. チャレンジのハッシュ化
        let clientDataHash = Data(SHA256.hash(data: challenge))

        // 3. Attestation 生成（Apple Attest Serverと通信）
        let attestation = try await service.attestKey(keyId, clientDataHash: clientDataHash)

        // 4. KeychainにキーIDを保存
        try SecureKeychain.save(key: .appAttestKeyID, data: Data(keyId.utf8))

        return (keyId, attestation)
    }

    /// APIリクエスト用のAssertion生成
    /// - Parameter requestData: タイムスタンプ + nonce + リクエストボディの結合データ
    func generateAssertion(for requestData: Data) async throws -> Data {
        guard let keyIdData = SecureKeychain.load(key: .appAttestKeyID),
              let keyId = String(data: keyIdData, encoding: .utf8) else {
            throw AppAttestError.keyNotFound
        }

        let clientDataHash = Data(SHA256.hash(data: requestData))
        return try await service.generateAssertion(keyId, clientDataHash: clientDataHash)
    }

    /// 保存済みキーIDの取得
    func getStoredKeyId() -> String? {
        guard let keyIdData = SecureKeychain.load(key: .appAttestKeyID) else {
            return nil
        }
        return String(data: keyIdData, encoding: .utf8)
    }

    /// キーIDの削除（再登録用）
    func clearStoredKeyId() throws {
        try SecureKeychain.delete(key: .appAttestKeyID)
    }
}
```

// InfraNetwork/AppAttest/AppAttestError.swift
enum AppAttestError: Error, LocalizedError {
    case notSupported          // App Attest非対応端末
    case keyNotFound           // KeychainにkeyIdが存在しない
    case attestationFailed     // attestation生成失敗
    case assertionFailed       // assertion生成失敗
    case challengeExpired      // challenge nonceの期限切れ
    case serverRejected        // サーバーがattestationを拒否

    var errorDescription: String? {
        switch self {
        case .notSupported: return "このデバイスはApp Attestに対応していません"
        case .keyNotFound: return "App Attestキーが見つかりません。再登録が必要です"
        case .attestationFailed: return "デバイス認証に失敗しました"
        case .assertionFailed: return "リクエスト署名に失敗しました"
        case .challengeExpired: return "認証チャレンジの有効期限が切れました"
        case .serverRejected: return "サーバーがデバイス認証を拒否しました"
        }
    }
}

### 3. RequestSigner :large_blue_circle:

05-security.mdセクション3.5準拠。App Attest assertionによるリクエスト署名。

```swift
// InfraNetwork/AppAttest/RequestSigner.swift
struct RequestSigner {
    private let appAttestManager: AppAttestManager

    init(appAttestManager: AppAttestManager) {
        self.appAttestManager = appAttestManager
    }

    /// リクエストに認証ヘッダとApp Attestアサーションを付加
    func sign(
        request: inout URLRequest,
        body: Data,
        token: String
    ) async throws {
        let timestamp = String(Int(Date().timeIntervalSince1970))
        let nonce = UUID().uuidString

        // Bearerトークンによる認証
        request.setValue("Bearer \(token)", forHTTPHeaderField: "Authorization")
        request.setValue(timestamp, forHTTPHeaderField: "X-Timestamp")
        request.setValue(nonce, forHTTPHeaderField: "X-Nonce")

        if appAttestManager.isSupported {
            // App Attestアサーションでリクエストの真正性を検証
            // clientDataHash = SHA256(timestamp + nonce + body)
            let clientData = "\(timestamp).\(nonce).\(body.base64EncodedString())"
            let assertion = try await appAttestManager.generateAssertion(
                for: Data(clientData.utf8)
            )
            request.setValue(
                assertion.base64EncodedString(),
                forHTTPHeaderField: "X-App-Attest-Assertion"
            )
        }
        // App Attest非対応端末ではassertion省略（サーバー側で制限付き許可）
    }
}
```

### 4. デバイス登録フロー :large_blue_circle:

03-backend-proxy.mdセクション4.4.1準拠。初回起動時の完全なフロー。

```swift
// InfraNetwork/Auth/DeviceRegistrationService.swift
final class DeviceRegistrationService {
    private let appAttestManager: AppAttestManager
    private let networkClient: NetworkClient

    /// 初回デバイス登録（App Attest attestation付き）
    func registerDevice() async throws -> AuthToken {
        let deviceId = UIDevice.current.identifierForVendor?.uuidString
            ?? UUID().uuidString

        if appAttestManager.isSupported {
            return try await registerWithAppAttest(deviceId: deviceId)
        } else {
            return try await registerWithFallback(deviceId: deviceId)
        }
    }

    private func registerWithAppAttest(deviceId: String) async throws -> AuthToken {
        // 1. Challenge取得
        let challengeResponse = try await networkClient.get("/api/v1/auth/challenge")
        let challenge = challengeResponse.challenge

        // 2. App Attest キー生成 + Attestation
        let (keyId, attestation) = try await appAttestManager.generateAndAttestKey(
            challenge: Data(challenge.utf8)
        )

        // 3. Backend APIにデバイス登録
        let request = DeviceAuthRequest(
            deviceId: deviceId,
            deviceInfo: DeviceInfo.current(),
            attestation: attestation.base64EncodedString(),
            keyId: keyId,
            challenge: challenge,
            attestSupported: true
        )

        let response = try await networkClient.post("/api/v1/auth/device", body: request)

        // 4. トークンをKeychainに保存
        try SecureKeychain.save(key: .accessToken, data: Data(response.accessToken.utf8))

        return AuthToken(
            accessToken: response.accessToken,
            expiresIn: response.expiresIn
        )
    }

    private func registerWithFallback(deviceId: String) async throws -> AuthToken {
        // App Attest非対応端末: 制限付き登録
        let request = DeviceAuthRequest(
            deviceId: deviceId,
            deviceInfo: DeviceInfo.current(),
            attestSupported: false
        )

        let response = try await networkClient.post("/api/v1/auth/device", body: request)
        try SecureKeychain.save(key: .accessToken, data: Data(response.accessToken.utf8))

        return AuthToken(
            accessToken: response.accessToken,
            expiresIn: response.expiresIn
        )
    }
}
```

## テスト要件

### 正常系
- AppAttestManager.isSupported が対応端末でtrueを返す
- generateAndAttestKey が keyId + attestation を返す
- attestation後にKeychainにkeyIdが保存される
- generateAssertion が保存済みkeyIdを使用してassertionを生成する
- RequestSignerがX-App-Attest-Assertionヘッダーを正しく設定する
- clientDataHash が SHA256(timestamp.nonce.body) で計算される
- デバイス登録フローが正常に完了しトークンがKeychainに保存される

### 異常系
- App Attest非対応端末でisSupported=false、フォールバック登録が実行される
- Keychain にkeyIdがない状態でgenerateAssertion呼び出し → AppAttestError.keyNotFound
- attestKey失敗時（ネットワークエラー等）のエラーハンドリング
- challenge期限切れ時のリトライロジック
- サーバーがattestationを拒否した場合のエラーハンドリング

### セキュリティ検証
- keyIdがkSecAttrAccessibleAfterFirstUnlockThisDeviceOnlyで保存される
- clearStoredKeyId()でKeychainからkeyIdが完全に削除される
