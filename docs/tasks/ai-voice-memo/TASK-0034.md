## タスク概要

WidgetKitを使用したホーム画面ウィジェット（Small/Medium）を実装する。Smallウィジェットはワンタップでアプリを起動し即座に録音を開始するショートカットとして機能する。Mediumウィジェットは録音ショートカットに加え、直近のメモプレビュー（最大2件）を表示する。Deep Link（`voicememo://record`）によるアプリ連携と、App Intent連携によるSiriショートカット対応を含む。

### 2. TimelineProvider :large_blue_circle:

直近メモの取得と15分間隔の更新。

```swift
// VoiceMemoWidget/Provider/VoiceMemoTimelineProvider.swift

import WidgetKit
import SwiftData

struct VoiceMemoEntry: TimelineEntry {
    let date: Date
    let recentMemos: [MemoPreview]
}

struct MemoPreview: Identifiable {
    let id: UUID
    let title: String
    let createdAt: Date
    let emotion: EmotionCategory?
    let durationSeconds: Double
}

struct VoiceMemoTimelineProvider: TimelineProvider {
    // App Groups経由でSwiftDataにアクセス
    private let modelContainer: ModelContainer? = {
        let groupURL = FileManager.default
            .containerURL(forSecurityApplicationGroupIdentifier: "group.io.murmurnote.app")
        guard let groupURL else { return nil }
        let config = ModelConfiguration(url: groupURL.appendingPathComponent("VoiceMemoStore.sqlite"))
        return try? ModelContainer(for: VoiceMemo.self, configurations: [config])
    }()

    func placeholder(in context: Context) -> VoiceMemoEntry {
        VoiceMemoEntry(date: Date(), recentMemos: [
            MemoPreview(id: UUID(), title: "サンプルメモ", createdAt: Date(), emotion: .joy, durationSeconds: 120),
        ])
    }

    func getSnapshot(in context: Context, completion: @escaping (VoiceMemoEntry) -> Void) {
        let entry = VoiceMemoEntry(date: Date(), recentMemos: fetchRecentMemos())
        completion(entry)
    }

    func getTimeline(in context: Context, completion: @escaping (Timeline<VoiceMemoEntry>) -> Void) {
        let entry = VoiceMemoEntry(date: Date(), recentMemos: fetchRecentMemos())
        // 15分後に次の更新をスケジュール
        let nextUpdate = Calendar.current.date(byAdding: .minute, value: 15, to: Date())!
        let timeline = Timeline(entries: [entry], policy: .after(nextUpdate))
        completion(timeline)
    }

    private func fetchRecentMemos() -> [MemoPreview] {
        guard let container = modelContainer else { return [] }
        let context = ModelContext(container)
        var descriptor = FetchDescriptor<VoiceMemo>(
            predicate: #Predicate { $0.status == .completed },
            sortBy: [SortDescriptor(\.createdAt, order: .reverse)]
        )
        descriptor.fetchLimit = 2
        guard let memos = try? context.fetch(descriptor) else { return [] }
        return memos.map { memo in
            MemoPreview(
                id: memo.id,
                title: memo.title,
                createdAt: memo.createdAt,
                emotion: memo.emotionAnalysis?.primaryEmotion,
                durationSeconds: memo.durationSeconds
            )
        }
    }
}
```

import SwiftUI

struct MediumWidgetView: View {

                            .font(.system(size: 20, weight: .medium))
                            .foregroundColor(.white)
                    }
                    Text("録音")
                        .font(.vmCaption1)
                        .foregroundColor(.vmTextPrimary)
                }
                .frame(width: 80)
            }

            // 右: 直近メモプレビュー
            VStack(alignment: .leading, spacing: VMSpacing.sm) {
                if entry.recentMemos.isEmpty {
                    Text("メモはまだありません")
                        .font(.vmCaption1)
                        .foregroundColor(.vmTextTertiary)
                        .frame(maxWidth: .infinity, maxHeight: .infinity)
                } else {
                    ForEach(entry.recentMemos) { memo in
                        Link(destination: URL(string: "voicememo://memo/\(memo.id.uuidString)")!) {
                            WidgetMemoRow(memo: memo)
                        }
                    }
                }
            }
            .frame(maxWidth: .infinity)
        }
        .padding(VMSpacing.md)
        .background(Color.vmBackground)
    }
}

struct WidgetMemoRow: View {
    let memo: MemoPreview

    var body: some View {
        HStack(spacing: VMSpacing.sm) {
            // 感情カラーインジケータ
            Circle()
                .fill(memo.emotion?.color ?? Color.vmSurfaceVariant)
                .frame(width: 8, height: 8)

            VStack(alignment: .leading, spacing: VMSpacing.xxs) {
                Text(memo.title.isEmpty ? "無題のメモ" : memo.title)
                    .font(.vmCaption1)
                    .foregroundColor(.vmTextPrimary)
                    .lineLimit(1)

                Text(memo.createdAt.relativeDescription)
                    .font(.vmCaption2)
                    .foregroundColor(.vmTextTertiary)
            }
        }
        .padding(.vertical, VMSpacing.xs)
    }
}

### 6. Deep Link受信（アプリ側） :large_blue_circle:

01-system-architecture.md セクション2.4準拠。`onOpenURL` でカスタムURLスキームを受信。

```swift
// VoiceMemoApp/VoiceMemoApp.swift（追加部分）

@main
struct VoiceMemoApp: App {
    var body: some Scene {
        WindowGroup {
            AppView(store: store)
                .onOpenURL { url in
                    store.send(.deepLinkReceived(url))
                }
        }
    }
}

// AppReducer（Deep Link処理）
case .deepLinkReceived(let url):
    guard let components = URLComponents(url: url, resolvingAgainstBaseURL: false) else {
        return .none
    }
    switch components.host {
    case "record":
        // 即座に録音開始
        state.selectedTab = .home
        return .send(.recording(.recordButtonTapped))

    case "memo":
        // メモ詳細画面を開く
        if let memoIDString = components.path.dropFirst().description as String?,
           let memoID = UUID(uuidString: memoIDString) {
            return .send(.navigateToMemoDetail(memoID))
        }
        return .none

    default:
        return .none
    }
```
