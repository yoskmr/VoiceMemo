### 1. 使用量カウント（D1トランザクション方式） :large_blue_circle:

03-backend-proxy.mdセクション5.5準拠。KVベストエフォートからD1トランザクション方式に改善。

```typescript
// src/services/usageCounter.ts

const FREE_MONTHLY_LIMIT = 5;
const PRO_DAILY_LIMIT = 1000;

interface UsageCheckResult {
  allowed: boolean;
  used: number;
  limit: number | null;
  resetsAt: string;
  plan: 'free' | 'pro';
}

/**
 * 月次使用量のチェックとインクリメント（D1トランザクション）
 * EC-014: 毎月1日 JST 0:00 (UTC+9 00:00:00) にリセット
 */
async function checkAndIncrementUsage(
  userId: string,
  plan: 'free' | 'pro',
  db: D1Database
): Promise<UsageCheckResult> {
  const jstYearMonth = getJSTYearMonth();
  const now = new Date().toISOString();

  if (plan === 'free') {
    return checkFreeUsage(userId, jstYearMonth, now, db);
  } else {
    return checkProUsage(userId, jstYearMonth, now, db);
  }
}

async function checkFreeUsage(
  userId: string,
  yearMonth: string,
  now: string,
  db: D1Database
): Promise<UsageCheckResult> {
  // D1トランザクションでアトミックにカウントチェック + インクリメント
  const existing = await db.prepare(
    'SELECT count FROM usage_counters WHERE user_id = ? AND year_month = ?'
  ).bind(userId, yearMonth).first();

  const currentCount = (existing?.count as number) || 0;

  if (currentCount >= FREE_MONTHLY_LIMIT) {
    return {
      allowed: false,
      used: currentCount,
      limit: FREE_MONTHLY_LIMIT,
      resetsAt: getNextMonthResetTime(),
      plan: 'free',
    };
  }

  // アトミックインクリメント（INSERT OR UPDATE）
  await db.prepare(
    `INSERT INTO usage_counters (id, user_id, year_month, count, last_used_at, created_at, updated_at)
     VALUES (?, ?, ?, 1, ?, ?, ?)
     ON CONFLICT(user_id, year_month) DO UPDATE SET
       count = count + 1,
       last_used_at = excluded.last_used_at,
       updated_at = excluded.updated_at`
  ).bind(generateId('usg'), userId, yearMonth, now, now, now).run();

  return {
    allowed: true,
    used: currentCount + 1,
    limit: FREE_MONTHLY_LIMIT,
    resetsAt: getNextMonthResetTime(),
    plan: 'free',
  };
}

async function checkProUsage(
  userId: string,
  yearMonth: string,
  now: string,
  db: D1Database
): Promise<UsageCheckResult> {
  // Proプランは日次制限（1000回/日）
  const jstDate = getJSTDate();
  const dayKey = `${yearMonth}-${jstDate}`;

  // ※ 日次カウントはKV利用（ベストエフォート許容）
  // D1は月次集計のみ
  const existing = await db.prepare(
    'SELECT count FROM usage_counters WHERE user_id = ? AND year_month = ?'
  ).bind(userId, yearMonth).first();

  const currentCount = (existing?.count as number) || 0;

  // Proプランは月次制限なし。日次制限のみ（別途KVで管理）
  await db.prepare(
    `INSERT INTO usage_counters (id, user_id, year_month, count, last_used_at, created_at, updated_at)
     VALUES (?, ?, ?, 1, ?, ?, ?)
     ON CONFLICT(user_id, year_month) DO UPDATE SET
       count = count + 1,
       last_used_at = excluded.last_used_at,
       updated_at = excluded.updated_at`
  ).bind(generateId('usg'), userId, yearMonth, now, now, now).run();

  return {
    allowed: true,
    used: currentCount + 1,
    limit: null,
    resetsAt: getNextMonthResetTime(),
    plan: 'pro',
  };
}
```

### 2. JST基準の月次リセットロジック :large_blue_circle:

03-backend-proxy.mdセクション5.5、EC-014準拠。

```typescript
// src/utils/jstTime.ts

/**
 * JST基準で現在の年月を取得
 * EC-014: 毎月1日 JST 0:00 にリセット
 */
function getJSTYearMonth(): string {
  const now = new Date();
  const jstOffset = 9 * 60 * 60 * 1000;
  const jstNow = new Date(now.getTime() + jstOffset);
  return `${jstNow.getUTCFullYear()}-${String(jstNow.getUTCMonth() + 1).padStart(2, '0')}`;
}

function getJSTDate(): string {
  const now = new Date();
  const jstOffset = 9 * 60 * 60 * 1000;
  const jstNow = new Date(now.getTime() + jstOffset);
  return String(jstNow.getUTCDate()).padStart(2, '0');
}

function getNextMonthResetTime(): string {
  const now = new Date();
  const jstOffset = 9 * 60 * 60 * 1000;
  const jstNow = new Date(now.getTime() + jstOffset);
  const year = jstNow.getUTCFullYear();
  const month = jstNow.getUTCMonth() + 1;
  const nextMonth = month === 12 ? 1 : month + 1;
  const nextYear = month === 12 ? year + 1 : year;
  return `${nextYear}-${String(nextMonth).padStart(2, '0')}-01T00:00:00+09:00`;
}

function getNextDayResetTime(): string {
  const now = new Date();
  const jstOffset = 9 * 60 * 60 * 1000;
  const jstNow = new Date(now.getTime() + jstOffset);
  const tomorrow = new Date(jstNow);
  tomorrow.setUTCDate(tomorrow.getUTCDate() + 1);
  tomorrow.setUTCHours(0, 0, 0, 0);
  // JSTで翌日0:00 → UTC換算で -9時間
  return new Date(tomorrow.getTime() - jstOffset).toISOString().replace('Z', '+09:00');
}
```

### 3. レート制限ミドルウェア :large_blue_circle:

03-backend-proxy.mdセクション6準拠。

```typescript
// src/middleware/rateLimit.ts
const usageLimitMiddleware = async (c: Context, next: Next) => {
  const user = c.get('user');

  const usageResult = await checkAndIncrementUsage(
    user.id,
    user.plan,
    c.env.DB
  );

  if (!usageResult.allowed) {
    // レスポンスヘッダー設定
    c.header('X-RateLimit-Limit', String(usageResult.limit));
    c.header('X-RateLimit-Remaining', '0');
    c.header('X-RateLimit-Reset', usageResult.resetsAt);

    return c.json({
      error: {
        code: 'USAGE_LIMIT_EXCEEDED',
        message: '月次AI処理上限に達しました',
        details: {
          used: usageResult.used,
          limit: usageResult.limit,
          plan: usageResult.plan,
          resets_at: usageResult.resetsAt,
          upgrade_url: 'voicememo://settings/upgrade',
        },
      },
    }, 429);
  }

  // 正常時もレスポンスヘッダー設定
  c.header('X-RateLimit-Limit', String(usageResult.limit ?? 'unlimited'));
  c.header('X-RateLimit-Remaining', String(
    usageResult.limit ? usageResult.limit - usageResult.used : 'unlimited'
  ));
  c.header('X-RateLimit-Reset', usageResult.resetsAt);

  // usageResultをコンテキストに保存（レスポンスのusageフィールド用）
  c.set('usage', usageResult);

  await next();
};
```

### 4. Pro日次レート制限（KV） :large_blue_circle:

```typescript
// src/services/dailyRateLimit.ts
async function checkDailyRateLimit(
  userId: string,
  kv: KVNamespace
): Promise<{ allowed: boolean; remaining: number }> {
  const jstDate = getJSTYearMonth() + '-' + getJSTDate();
  const key = `daily:${userId}:${jstDate}`;
  const raw = await kv.get(key);
  const count = raw ? parseInt(raw, 10) : 0;

  if (count >= PRO_DAILY_LIMIT) {
    return { allowed: false, remaining: 0 };
  }

  await kv.put(key, String(count + 1), { expirationTtl: 86400 + 3600 });
  return { allowed: true, remaining: PRO_DAILY_LIMIT - count - 1 };
}
```

## テスト要件

### 正常系
- Freeプランで1回目〜5回目のAI処理が許可される
- Freeプランで5回目のレスポンスにX-RateLimit-Remaining: 0が含まれる
- Proプランで1000回/日のAI処理が許可される
- 月が変わった（JST基準）際にカウントがリセットされる
- レスポンスヘッダー（X-RateLimit-Limit, X-RateLimit-Remaining, X-RateLimit-Reset）が正しく設定される
- usageフィールド（used, limit, plan, resets_at）がレスポンスに含まれる

### 月次リセット（EC-014）
- 3月31日 23:59 JSTの処理と4月1日 00:00 JSTの処理が異なる月として集計される
- UTC基準ではなくJST基準でリセットされる
- 12月→1月の年跨ぎでリセットが正しく動作する

### 異常系
- Freeプラン6回目のAI処理で429 USAGE_LIMIT_EXCEEDEDが返る
- エラーレスポンスにupgrade_urlが含まれる
- D1書き込み失敗時のエラーハンドリング
- 同時リクエストでのアトミックカウンタの整合性

### 境界値
- ちょうど5回目（Free上限ジャスト）でallowed=trueかつremaining=0
- ちょうど6回目でallowed=false
- JSTで月境界（3月31日 23:59:59 → 4月1日 00:00:00）
